64  	public void arithmeticTaskTest() throws ExertionException,
65  			SignatureException, ContextException, RemoteException {
66  		// to test tracing of execution enable ServiceExertion.debug
67  		ParModel pm = new ParModel("par-model");
68  
69  		Task task = task(
70  				"add",
71  				sig("add", AdderImpl.class),
72  				context(in("arg/x1", 20.0), in("arg/x2", 80.0),
73  						result("result/y")));
74  
75  		OptExertion ift = opt("ift", condition(pm,
76  				"{ x, y -> x > y }", "x", "y"), task);
77  
78  		add(pm, ift);
79  		put(pm, "x", 10.0);
80  		put(pm, "y", 20.0);
81  
82  		task = exert(ift);
83  		// logger.info("task: " + task);
84  		assertEquals(false, get(task, Condition.CONDITION_VALUE));
85  		assertEquals("add", get(task, Condition.CONDITION_TARGET));
86  
87  		pm.putValue("x", 300.0);
88  		pm.putValue("y", 200.0);
89  		task = exert(ift);
90  		// logger.info("opt value: " + exert(ift));
91  		assertEquals(true, get(task, Condition.CONDITION_VALUE));
92  		assertEquals("add", get(task, Condition.CONDITION_TARGET));
93  		assertEquals(100.0, get(task, "result/y"));
94  	}
