28  /**
29   * Class used to configure a RootLoader from a stream or by using
30   * it's methods.
31   * <p/>
32   * The stream can be for example a FileInputStream from a file with
33   * the following format:
34   * <p/>
35   * <pre>
36   * # comment
37   * main is classname
38   * load path
39   * load file
40   * load pathWith${property}
41   * load pathWith!{required.property}
42   * load path/*.jar
43   * load path/&#42;&#42;/&#42;.jar
44   * </pre>
45   * <ul>
46   * <li>All lines starting with "#" are ignored.</li>
47   * <li>The "main is" part may only be once in the file. The String
48   * afterwards is the name of a class with a main method. </li>
49   * <li>The "load" command will add the given file or path to the
50   * classpath in this configuration object. If the path does not
51   * exist, the path will be ignored.
52   * </li>
53   * <li>properties referenced using !{x} are required.</li>
54   * <li>properties referenced using ${x} are not required. If the
55   * property does not exist the whole load instruction line will
56   * be ignored.</li>
57   * <li>* is used to match zero or more characters in a file.</li>
58   * <li>** is used to match zero or more directories.</li>
59   * <li>Loading paths with <code>load ./*.jar</code> or <code>load *.jar</code> are not supported.</li>
60   * </ul>
61   * <p/>
62   * Defining the main class is required unless setRequireMain(boolean) is
63   * called with false, before reading the configuration.
64   * You can use the wildcard "*" to filter the path, but only for files, not
65   * directories. To match directories use "**". The ${propertyname} is replaced by the value of the system's
66   * property name. You can use user.home here for example. If the property does
67   * not exist, an empty string will be used. If the path or file after the load
68   * command does not exist, the path will be ignored.
69   *
70   * @author Jochen Theodorou
71   * @version $Revision$
72   * @see RootLoader
73   */
74  public class LoaderConfiguration {

77      private List classPath = new ArrayList();

81      private static final String ALL_WILDCARD = "" + WILDCARD + WILDCARD;
82      private static final String MATCH_FILE_NAME = "\\\\E[^/]+?\\\\Q";
83      private static final String MATCH_ALL = "\\\\E.+?\\\\Q";

92      /**
93       * configures this loader with a stream
94       *
95       * @param is stream used to read the configuration
96       * @throws IOException if reading or parsing the contents of the stream fails
97       */
98      public void configure(InputStream is) throws IOException {
99          BufferedReader reader = new BufferedReader(new InputStreamReader(is));
100         int lineNumber = 0;
101 
102         while (true) {
103             String line = reader.readLine();
104             if (line == null) break;
105 
106             line = line.trim();
107             lineNumber++;
108 
109             if (line.startsWith("#") || line.length() == 0) continue;
110 
111             if (line.startsWith(LOAD_PREFIX)) {
112                 String loadPath = line.substring(LOAD_PREFIX.length()).trim();
113                 loadPath = assignProperties(loadPath);
114                 loadFilteredPath(loadPath);
115             } else if (line.startsWith(MAIN_PREFIX)) {
116                 if (main != null)
117                     throw new IOException("duplicate definition of main in line " + lineNumber + " : " + line);
118                 main = line.substring(MAIN_PREFIX.length()).trim();
119             } else if (line.startsWith(PROP_PREFIX)) {
120                 String params = line.substring(PROP_PREFIX.length()).trim();
121                 int index = params.indexOf('=');
122                 if (index == -1) {
123                     throw new IOException("unexpected property format - expecting name=value" + lineNumber + " : " + line);
124                 }
125                 String propName = params.substring(0, index);
126                 String propValue= assignProperties(params.substring(index+1));
127                 System.setProperty(propName, propValue);
128             } else {
129                 throw new IOException("unexpected line in " + lineNumber + " : " + line);
130             }
131         }
132 
133         if (requireMain && main == null) throw new IOException("missing main class definition in config file");
134     }

136     /*
137     * Expands the properties inside the given string to it's values.
138     */
139     private String assignProperties(String str) {
140         int propertyIndexStart = 0, propertyIndexEnd = 0;
141         boolean requireProperty = false;
142         String result = "";
143 
144         while (propertyIndexStart < str.length()) {
145             {
146                 int i1 = str.indexOf("${", propertyIndexStart);
147                 int i2 = str.indexOf("!{", propertyIndexStart);
148                 if (i1 == -1) {
149                     propertyIndexStart = i2;
150                 } else if (i2 == -1) {
151                     propertyIndexStart = i1;
152                 } else {
153                     propertyIndexStart = Math.min(i1, i2);
154                 }
155                 requireProperty = propertyIndexStart == i2;
156             }
157             if (propertyIndexStart == -1) break;
158             result += str.substring(propertyIndexEnd, propertyIndexStart);
159 
160             propertyIndexEnd = str.indexOf("}", propertyIndexStart);
161             if (propertyIndexEnd == -1) break;
162 
163             String propertyKey = str.substring(propertyIndexStart + 2, propertyIndexEnd);
164             String propertyValue = null;
165             // Interpret Maven Artifact coordinates
166             if (propertyKey.startsWith("mvn:")) {            	
167             	propertyValue = Resolver.resolveAbsolute(propertyKey.substring(4));
168             }  else {         
169             	propertyValue = System.getProperty(propertyKey);
170             }
171             // assume properties contain paths
172             if (propertyValue == null) {
173                 if (requireProperty) {
174                     throw new IllegalArgumentException("Variable " + propertyKey + " in nsh.config references a non-existent System property! Try passing the property to the VM using -D" + propertyKey + "=myValue in JAVA_OPTS");
175                 } else {
176                     return null;
177                 }
178             }
179             propertyValue = getSlashyPath(propertyValue);
180             propertyValue = correctDoubleSlash(propertyValue,propertyIndexEnd,str);
181             result += propertyValue;
182 
183             propertyIndexEnd++;
184             propertyIndexStart = propertyIndexEnd;
185         }
186 
187         if (propertyIndexStart == -1 || propertyIndexStart >= str.length()) {
188             result += str.substring(propertyIndexEnd);
189         } else if (propertyIndexEnd == -1) {
190             result += str.substring(propertyIndexStart);
191         }
192 
193         return result;
194     }

197     private String correctDoubleSlash(String propertyValue, int propertyIndexEnd, String str) {
198         int index = propertyIndexEnd+1;
199         if ( index<str.length() && str.charAt(index)=='/' && 
200              propertyValue.endsWith("/") &&
201              propertyValue.length()>0)
202         {
203             propertyValue = propertyValue.substring(0,propertyValue.length()-1);
204         } 
205         return propertyValue;
206     }

208     /*
209      * Load a possibly filtered path. Filters are defined
210      * by using the * wildcard like in any shell.
211      */
212     private void loadFilteredPath(String filter) {
213         if (filter == null) return;
214         filter = getSlashyPath(filter);
215         int starIndex = filter.indexOf(WILDCARD);
216         if (starIndex == -1) {
217             addFile(new File(filter));
218             return;
219         }
220         boolean recursive = filter.indexOf(ALL_WILDCARD) != -1;
221 
222         if (filter.lastIndexOf('/')<starIndex) {
223             starIndex=filter.lastIndexOf('/')+1;
224         }
225         String startDir = filter.substring(0, starIndex - 1);
226         File root = new File(startDir);
227 
228         filter = Pattern.quote(filter);
229         filter = filter.replaceAll("\\" + WILDCARD + "\\" + WILDCARD, MATCH_ALL);
230         filter = filter.replaceAll("\\" + WILDCARD, MATCH_FILE_NAME);
231         Pattern pattern = Pattern.compile(filter);
232 
233         final File[] files = root.listFiles();
234         if (files != null) {
235             findMatchingFiles(files, pattern, recursive);
236         }
237     }
238 
239     private void findMatchingFiles(File[] files, Pattern pattern, boolean recursive) {
240         for (int i = 0; i < files.length; i++) {
241             File file = files[i];
242             String fileString = getSlashyPath(file.getPath());
243             Matcher m = pattern.matcher(fileString);
244             if (m.matches() && file.isFile()) {
245                 addFile(file);
246             }
247             if (file.isDirectory() && recursive) {
248                 final File[] dirFiles = file.listFiles();
249                 if (dirFiles != null) {
250                     findMatchingFiles(dirFiles, pattern, true);
251                 }
252             }
253         }
254     }

256     // change path representation to something more system independent.
257     // This solution is based on an absolute path
258     private String getSlashyPath(final String path) {
259         String changedPath = path;
260         if (File.separatorChar != '/')
261             changedPath = changedPath.replace(File.separatorChar, '/');
262 
263         return changedPath;
264     }

278     private String getParentPath(String filter) {

285      * Adds a file to the classpath if it exists.
286      *
287      * @param file the file to add
288      */
289     public void addFile(File file) {
290         if (file != null && file.exists()) {
291             try {
292                 classPath.add(file.toURI().toURL());
293             } catch (MalformedURLException e) {
294                 throw new AssertionError("converting an existing file to an url should have never thrown an exception!");
295             }
296         }
297     }

308     /**
309      * Adds a classpath to this configuration. It expects a string with
310      * multiple paths, separated by the system dependent path separator.
311      * Expands wildcards, e.g. dir/* into all the jars in dir.
312      *
313      * @param path the path as a path separator delimited string
314      * @see java.io.File#pathSeparator
315      */
316     public void addClassPath(String path) {
317        String[] paths = path.split(File.pathSeparator);
318         for (String cpPath : paths) {
319             // Check to support wild card classpath
320             if (cpPath.endsWith("*")) {
321                 File dir = new File(cpPath.substring(0, cpPath.length() - 1));
322                 File[] files = dir.listFiles();
323                 if (files != null) {
324                     for (File file : files) {
325                         if (file.isFile() && file.getName().endsWith(".jar")) addFile(file);
326                     }
327                 }
328             } else {
329                 addFile(new File(cpPath));
330             }
331         }
332     }
