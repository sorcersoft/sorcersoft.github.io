52  /**
53   * A wrapper for activatable objects, providing separation of the import
54   * codebase (where the server classes are loaded from by the activation group)
55   * from the export codebase (where clients should load classes from for stubs,
56   * etc.) as well as providing an independent security policy file for each
57   * activatable object. This functionality allows multiple activatable objects to
58   * be placed in the same activation group, with each object maintaining a
59   * distinct codebase and policy.
60   * <p>
61   * This wrapper class is assumed to be available directly in the activation
62   * group VM; that is, it is assumed to be in the application classloader, the
63   * extension classloader, or the boot classloader, rather than being downloaded.
64   * Since this class also needs considerable permissions, the easiest thing to do
65   * is to make it an installed extension.
66   * <p>
67   * This wrapper class performs a security check to control what policy files can
68   * be used with a given codebase. It does this by querying the VM's (global)
69   * policy for {@link com.sun.jini.start.SharedActivationPolicyPermission}
70   * grants. The service's associated
71   * {@link com.sun.jini.start.ActivateWrapper.ActivateDesc#importLocation
72   * ActivateDesc.importLocation} is used as the {@link java.security.CodeSource}
73   * for selecting the appropriate permission set to check against. If multiple
74   * codebases are used, then all the codebases must have the necessary
75   * <code>SharedActivationPolicyPermission</code> grants.
76   *
77   * @author Dennis Reedy
78   */

187     /**
188      * Activatable constructor. This constructor:
189      * <UL>
190      * <LI>Retrieves an <code>ActivateDesc</code> from the provided
191      * <code>data</code> parameter.
192      * <LI>creates a {@link ServiceClassLoader} using the import and
193      * export codebases obtained from the provided <code>ActivateDesc</code>,
194      * <LI>checks the import codebase(s) for the required
195      * <code>SharedActivationPolicyPermission</code>
196      * <LI>associates the newly created {@link ServiceClassLoader} and
197      * the corresponding policy file obtained from the <code>ActivateDesc</code>
198      * with the <code>AggregatePolicyProvider</code>
199      * <LI>loads the "wrapped" activatable object's class and calls its
200      * activation constructor with the context classloader set to the newly
201      * created {@link ServiceClassLoader}.
202      * <LI>resets the context class loader to the original context classloader
203      * </UL>
204      * The first instance of this class will also replace the VM's existing
205      * <code>Policy</code> object, if any, with a
206      * <code>AggregatePolicyProvider</code>.
207      * 
208      * @param id The <code>ActivationID</code> of this object
209      * @param data The activation data for this object
210      *
211      * @throws Exception of any errors occur
212      *
213      * @see ServiceClassLoader
214      * @see com.sun.jini.start.AggregatePolicyProvider
215      * @see com.sun.jini.start.SharedActivationPolicyPermission
216      * @see java.security.Policy
217      */
218     public ActivateWrapper(ActivationID id, MarshalledObject data)
219     throws Exception {
220         logger.entering(ActivateWrapper.class.getName(),
221                         "ActivateWrapper",
222                         new Object[]{id, data});
223         ActivateDesc desc = (ActivateDesc)data.get();
224         logger.log(Level.FINEST, "ActivateDesc: {0}", desc);
225 
226         CommonClassLoader commonCL = CommonClassLoader.getInstance();
227         commonCL.addCommonJARs(desc.commonJARs);
228         if(logger.isLoggable(Level.FINEST))
229             logger.log(Level.FINEST,
230                        "Created CommonClassLoader: {0}",
231                        commonCL);
232 
233         ServiceClassLoader cl;
234         try {
235             
236             cl = new ServiceClassLoader(ServiceClassLoader.getURIs(
237                                                            desc.importLocation),
238                                         new ClassAnnotator(desc.exportLocation,
239                                                            new Properties()),
240                                         commonCL);
241 
242             /*
243             cl = new ServiceClassLoader(desc.importLocation,
244                                         new ClassAnnotator(desc.exportLocation,
245                                                            new Properties()),
246                                         commonCL);
247                                         */
248             if(logger.isLoggable(Level.FINEST))
249                 logger.log(Level.FINEST,
250                            "Created ServiceClassLoader: {0}",
251                            cl);
252         } catch(Exception e) {
253             logger.throwing(ActivateWrapper.class.getName(),
254                             "ActivateWrapper",
255                             e);
256             throw e;
257         }
258         checkPolicyPermission(desc.policy, desc.importLocation);
259         synchronized(ActivateWrapper.class) {
260             // supplant global policy 1st time through
261             if(globalPolicy == null) {
262                 initialGlobalPolicy = Policy.getPolicy();
263                 globalPolicy = new AggregatePolicyProvider(initialGlobalPolicy);
264                 Policy.setPolicy(globalPolicy);
265                 logger.log(Level.FINEST,
266                            "Global policy set: {0}",
267                            globalPolicy);
268             }
269             DynamicPolicyProvider service_policy =
270                 new DynamicPolicyProvider(new PolicyFileProvider(desc.policy));
271             LoaderSplitPolicyProvider split_service_policy =
272                 new LoaderSplitPolicyProvider(
273                     cl,
274                     service_policy,
275                     new DynamicPolicyProvider(initialGlobalPolicy));
276             split_service_policy.grant(this.getClass(),
277                                        null, /* Principal[] */
278                                        new Permission[]{new AllPermission()});
279             globalPolicy.setPolicy(cl, split_service_policy);
280             logger.log(Level.FINEST,
281                        "Added policy to set: {0}",
282                        desc.policy);
283         }
284         Thread t = Thread.currentThread();
285         ClassLoader ccl = t.getContextClassLoader();
286         logger.log(Level.FINEST,
287                    "Saved current context class loader: {0}",
288                    ccl);
289         t.setContextClassLoader(cl);
290         logger.log(Level.FINEST, "Set new context class loader: {0}", cl);
291         try {
292             boolean initialize = false;
293             Class ac = Class.forName(desc.className, initialize, cl);
294             logger.log(Level.FINEST,
295                        "Obtained implementation class: {0}",
296                        ac);
297             Constructor constructor = ac.getDeclaredConstructor(actTypes);
298             logger.log(Level.FINEST,
299                        "Obtained implementation constructor: {0}",
300                        constructor);
301             constructor.setAccessible(true);
302             impl = constructor.newInstance(id, desc.data);
303             logger.log(Level.FINEST,
304                        "Obtained implementation instance: {0}",
305                        impl);
306         } finally {
307             t.setContextClassLoader(ccl);
308             logger.log(Level.FINEST,
309                        "Context class loader reset to: {0}",
310                        ccl);
311 
312         }
313         logger.exiting(ActivateWrapper.class.getName(), "ActivateWrapper");
314     }

316     /*
317      * Return a reference to service being wrapped in place of this object.
318      */
319     private Object writeReplace() throws ObjectStreamException {
320         Object impl_proxy = impl;
321         if(impl instanceof ProxyAccessor) {
322             impl_proxy = ((ProxyAccessor)impl).getProxy();
323             logger.log(Level.FINEST,
324                        "Obtained implementation proxy: {0}",
325                        impl_proxy);
326             if(impl_proxy == null) {
327                 throw new InvalidObjectException("Implementation's getProxy() returned null");
328             }
329         }
330         return impl_proxy;
331     }

333     /**
334      * Analog to
335      * {@link java.rmi.activation.Activatable#register(java.rmi.activation.ActivationDesc)
336      * Activatable.register()} for activatable objects that want to use this
337      * wrapper mechanism.
338      * 
339      * @return activation ID of the registered service
340      * @throws ActivationException if there was a problem registering the
341      * activatable class with the activation system
342      * @throws RemoteException if there was a problem communicating with the
343      * activation system
344      */
345     public static ActivationID register(ActivationGroupID gid,
346                                         ActivateDesc desc,
347                                         boolean restart,
348                                         ActivationSystem sys)
349         throws ActivationException, RemoteException {
350         logger.entering(ActivateWrapper.class.getName(),
351                         "register",
352                         new Object[]{gid, desc, restart, sys});
353         MarshalledObject data;
354         try {
355             data = new MarshalledObject(desc);
356         } catch(Exception e) {
357             MarshalException me = new MarshalException("marshalling ActivateDesc",
358                                                        e);
359             logger.throwing(ActivateWrapper.class.getName(), "register", me);
360             throw me;
361         }
362         ActivationDesc adesc = new ActivationDesc(gid,
363                                                   ActivateWrapper.class.getName(),
364                                                   null,
365                                                   data,
366                                                   restart);
367         logger.log(Level.FINEST,
368                    "Registering descriptor with activation: {0}",
369                    adesc);
370         ActivationID aid = sys.registerObject(adesc);
371         logger.exiting(ActivateWrapper.class.getName(), "register", aid);
372         return aid;
373     }

375     /*
376      * Checks that all the provided <code>URL</code> s have permission to use
377      * the given policy.
378      */
379     private static void checkPolicyPermission(String policy, URL[] urls) {
380         logger.entering(ActivateWrapper.class.getName(),
381                         "checkPolicyPermission",
382                         new Object[]{policy, urlsToPath(urls)});
383         // Create desired permission object
384         Permission perm = new SharedActivationPolicyPermission(policy);
385         Certificate[] certs = null;
386         CodeSource cs;
387         ProtectionDomain pd;
388         // Loop over all codebases
389         for (URL url : urls) {
390             // Create ProtectionDomain for given codesource
391             cs = new CodeSource(url, certs);
392             pd = new ProtectionDomain(cs, null, null, null);
393             logger.log(Level.FINEST, "Checking protection domain: {0}", pd);
394             // Check if current domain allows desired permission
395             if (!pd.implies(perm)) {
396                 SecurityException se = new SecurityException("ProtectionDomain "
397                                                              + pd
398                                                              +
399                                                              " does not have required permission: "
400                                                              + perm);
401                 logger.throwing(ActivateWrapper.class.getName(),
402                                 "checkPolicyPermission",
403                                 se);
404                 throw se;
405             }
406         }
407         logger.exiting(ActivateWrapper.class.getName(), "checkPolicyPermission");
408     }

410     /*
411      * Utility method that converts a <code>URL[]</code> into a corresponding,
412      * space-separated string with the same array elements. Note that if the
413      * array has zero elements, the return value is null, not the empty string.
414      */
415     private static String urlsToPath(URL[] urls) {
416         //TODO - check if spaces in file paths are properly escaped (i.e.%
417         // chars)
418         if(urls.length == 0) {
419             return null;
420         } else if(urls.length == 1) {
421             return urls[0].toExternalForm();
422         } else {
423             StringBuffer path = new StringBuffer(urls[0].toExternalForm());
424             for(int i = 1; i < urls.length; i++) {
425                 path.append(' ');
426                 path.append(urls[i].toExternalForm());
427             }
428             return path.toString();
429         }
