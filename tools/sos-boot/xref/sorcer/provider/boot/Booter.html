258     /**
259      * Will return an array of URLs based on the input String array. If the array 
260      * element is a file, the fully qualified file path must be provided. If the
261      * array element is a directory, a fully qualified directory path must be 
262      * provided, and the directory will be searched for all .jar and .zip files
263      * 
264      * @param elements A String array of fully qualified directory path
265      * 
266      * @return An URL[] elements
267      * 
268      * @throws MalformedURLException if any of the elements cannot be converted
269      */
270     public static URL[] toURLs(String[] elements)throws MalformedURLException {
271         ArrayList<URL> list = new ArrayList<URL>();
272         if(elements!=null) {
273             for (String el : elements) {
274                 File element = new File(el);
275                 if (element.isDirectory()) {
276                     URL[] urls = scanDirectory(el);
277                     list.addAll(Arrays.asList(urls));
278                 } else {
279                     list.add(element.toURI().toURL());
280                 }
281             }
282         }
283         return(list.toArray(new URL[list.size()]));
284     }

286     /**
287      * Will return an array of URLs for all .jar and .zip files in the directory,
288      * including the directory itself
289      * 
290      * @param dirPath A fully qualified directory path
291      * 
292      * @return A URL[] for all .jar and .zip files in the directory,
293      *  including the directory itself
294      * 
295      * @throws MalformedURLException If elements in the directory cannot be
296      * created into a URL
297      */
298     public static URL[] scanDirectory(String dirPath) throws MalformedURLException {
299         File dir = new File(dirPath);
300         if(!dir.isDirectory())
301             throw new IllegalArgumentException(dirPath+" is not a directory");
302         if(!dir.canRead())
303             throw new IllegalArgumentException("No read permissions for "+dirPath);
304         File[] files = dir.listFiles();
305         ArrayList<URL> list = new ArrayList<URL>();
306         
307         list.add(dir.toURI().toURL());
308 
309         for (File file : files) {
310             if (file.getName().endsWith(".jar") ||
311                 file.getName().endsWith(".JAR") ||
312                 file.getName().endsWith(".zip") ||
313                 file.getName().endsWith(".ZIP")) {
314 
315                 if (file.isFile())
316                     list.add(file.toURI().toURL());
317             }
318         }
319         return(list.toArray(new URL[list.size()]));
320     }

327      *

331         StringTokenizer tok = new StringTokenizer(arg, " ,"+File.pathSeparator);
332         String[] array = new String[tok.countTokens()];
333         int i=0;

336             i++;
337         }

358     /**
359      * Read the properties file and return an array of String instances which
360      * can be used as Jini configuration overrides
361      *
362      * @param propertiesFile A Properties file containing override declarations
363      * @param components An array of components to match, maty be null. If not
364      * null, then if a key starts with an element in the provided components
365      * array it will be part of the array of overrides returned. If this
366      * parameter is null or zero-length, all property components are returned
367      *
368      * @return An array of String values suitable for use as Jini configuration
369      * overrides. If there are no matching components or the properties file is
370      * empty return a zero-length array
371      *
372      * @throws NullPointerException if propertiesFile is null
373      * @throws IOException if the Properties file cannot be read
374      */
375     public static String[] parseOverrides(String propertiesFile,
376                                           String[] components) throws
377                                                                IOException {
378         if(propertiesFile==null)
379             throw new NullPointerException("propertiesFile is null");
380         ArrayList<String> list = new ArrayList<String>();
381         Properties props = new Properties();
382         props.load(new FileInputStream(propertiesFile));
383         if(components!=null && components.length>0) {
384             for(Enumeration e=props.propertyNames(); e.hasMoreElements();) {
385                 String key = (String)e.nextElement();
386                 for (String component : components) {
387                     if (key.startsWith(component)) {
388                         list.add(key + "=" + props.getProperty(key));
389                     }
390                 }
391             }
392         } else {
393             for(Enumeration e=props.propertyNames(); e.hasMoreElements();) {
394                 String key = (String)e.nextElement();
395                 list.add(key+"="+props.getProperty(key));
396             }
397         }
398         return(list.toArray(new String[list.size()]));
399     }

401     /**
402      * Assemble a String array suitable for use as Jini configuration options.
403      * The String array whose first element is the configuration source and
404      * remaining elements specify override values for entries
405      *
406      * @param configFile The location of the configuration file, must not be
407      * null
408      * @param overrides Array of elements specifying override values for
409      * entries, mat be null
410      *
411      * @return A String array suitable for use as a Jini configuration
412      *
413      * @throws NullPointerException if configFile is null
414      */
415     public static String[] createConfigArgs(String configFile, String[] overrides) {
416         String[] configArgs;
417         if(overrides!=null && overrides.length>0) {
418             configArgs = new String[overrides.length+1];
419             configArgs[0] = configFile;
420             System.arraycopy(overrides, 0, configArgs, 1, overrides.length);
421         } else {
422             configArgs = new String[]{configFile};
423         }
424         return(configArgs);
425     }

428     /**
429 	 * Tries to load properties from a <code>filename</code>, first in a local
430 	 * directory. If there is no file in the local directory then load the file
431 	 * from the classpath at sorcer/util/sorcer.env.
432      * @throws IOException 
433 	 * 
434 	 * @throws IOException
435 	 */
436 	private static void loadEnvironment() throws IOException {
437 		String envFile = System.getProperty("sorcer.env.file");
438 		try {
439 			// check if the properties file is specified by the system property
440 			if (envFile != null) {
441 				props.load(new FileInputStream(envFile));
442 			} else {
443 				envFile = "sorcer.env";
444 
445 				// Try in user home directory first
446 				props.load((new FileInputStream(new File(envFile))));
447 				logger.info("got Env properties in user home: "
448 						+ System.getProperty("user.home"));
449 			}
450 		} catch (Throwable t1) {
451 			try {
452 				// No file at home, give a try as resource
453 				// sorcer/util/sorcer.env
454 				//
455 				InputStream stream = SorcerConstants.class.getResourceAsStream(envFile);
456 				if (stream != null)
457 					props.load(stream);
458 				else
459 					logger.severe("could not load the SORCER Env properties");
460 			} catch (Throwable t2) {
461 				throw new IOException();
462 			}
463 		}
464 		
465 		// use the local host address if requested
466 		String pattern = "${localhost}";
467         String value = props.getProperty("provider.webster.interface");
468         if (value!=null && value.indexOf(pattern) >= 0) {
469 			try {
470 				String val = value.replace(pattern, getHostAddress());
471 				props.put("provider.webster.interface", val);
472 			} catch (UnknownHostException e1) {
473                 logger.log(Level.WARNING, "Could not resolve hostname", e1);
474             }
475 		}
476 	}
