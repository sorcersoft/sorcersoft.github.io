34  /**
35   * @author Rafał Krupiński
36   */
37  public class OpstringServiceDescriptor extends ResolvingServiceDescriptor {
38      private String name;
39  
40      private ServiceElement serviceElement;
41  
42      public OpstringServiceDescriptor(ServiceElement serviceElement, URL policyFile) {
43          this.serviceElement = serviceElement;
44          try {
45              setCodebase(getCodebase(serviceElement));
46          } catch (MalformedURLException e) {
47              throw new IllegalStateException("Error while processing codebase of " + serviceElement.getName(), e);
48          }
49          //TODO Policy files cannot be read from jar
50          setPolicyFile(SorcerEnv.getHomeDir() + File.separator + "configs" + File.separator + "sorcer.policy");
51          String[] configArgs = serviceElement.getServiceBeanConfig().getConfigArgs();
52          setServiceConfigArgs(Arrays.asList(configArgs));
53          setImplClassName(serviceElement.getComponentBundle().getClassName());
54          name = serviceElement.getOperationalStringName() + "/" + serviceElement.getName();
55      }
56  
57      protected Set<URI> getClasspath() {
58          String artifact = serviceElement.getComponentBundle().getArtifact();
59          try {
60              Set<URI> result = new HashSet<URI>();
61              Collections.addAll(result, SorcerResolverHelper.toURIs(resolver.getClassPathFor(artifact, serviceElement.getRemoteRepositories())));
62              return result;
63          } catch (URISyntaxException e) {
64              throw new IllegalStateException("Malformed URI", e);
65          } catch (ResolverException e) {
66              throw new IllegalStateException("Could not resolve artifact " + artifact);
67          }
68      }
69  
70      private static Set<URL> getCodebase(ServiceElement serviceElement) throws MalformedURLException {
71          try {
72              URL codebaseRoot = SorcerEnv.getCodebaseRoot();
73              OpStringUtil.checkCodebase(serviceElement, codebaseRoot.toExternalForm());
74          } catch (IOException e) {
75              throw new IllegalStateException("Malformed URL", e);
76          }
77  
78          Set<URL> result = new HashSet<URL>();
79          URL[] exportURLs = serviceElement.getExportURLs();
80          if (exportURLs.length != 0) {
81              Collections.addAll(result, exportURLs);
82          } else {
83              ClassBundle[] exportBundles = serviceElement.getExportBundles();
84  
85              for (ClassBundle exportBundle : exportBundles) {
86                  URL codebaseRoot = new URL(exportBundle.getCodebase());
87                  String artifact = exportBundle.getArtifact();
88                  if (artifact != null)
89                      result.add(GenericUtil.toArtifactUrl(codebaseRoot, artifact));
90              }
91          }
92          return result;
93      }
94  
95      @Override
96      public String toString() {
97          return "OpstringServiceDescriptor " + name;
98      }
99  }
