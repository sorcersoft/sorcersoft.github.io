95  	@Override
96  	public void execute() throws MojoExecutionException, MojoFailureException {
97  		if (!project.getPackaging().equals("jar")) {
98  			getLog().warn("Plugin misconfigured: runnig on a project with packaging other than jar");
99  		}
100 
101 		String sorcerHome = SorcerEnv.getHomeDir().getPath();
102 		List<String> websterRoots = new LinkedList<String>();
103 
104 		Map<String, String> sysProps = new HashMap<String, String>();
105 		sysProps.put("java.util.logging.config.file", new File(sorcerHome, "configs/sorcer.logging").getPath());
106 		sysProps.put("java.security.policy", new File(testOutputDir, "sorcer.policy").getPath());
107 		sysProps.put("sorcer.env.file", sorcerEnvFile.getPath());
108 		sysProps.put("java.rmi.server.codebase", buildCodeBase(websterRoots));
109 		sysProps.put("java.rmi.server.useCodebaseOnly", "false");
110 		if (!websterRoots.isEmpty()) {
111 			websterRoots.add(repositorySystemSession.getLocalRepository().getBasedir().getPath());
112 			sysProps.put("webster.roots", StringUtils.join(websterRoots, ";"));
113 		}
114 
115 		JavaProcessBuilder builder = new JavaProcessBuilder(getLog());
116 		builder.setProperties(sysProps);
117 		builder.setMainClass(mainClass);
118 		builder.setClassPath(buildClasspath());
119 		builder.setDebugger(debug);
120 		builder.setOutput(logFile);
121 
122 		try {
123 			if (waitBeforeRun > 0) {
124 				Thread.sleep(waitBeforeRun);
125 			}
126 			getLog().info("Starting requestor process");
127 			Process2 process = builder.startProcess();
128 			if (debug) {
129 				process.waitFor();
130 			} else {
131 				process.waitFor(10000, true);
132 			}
133 			getLog().info("Requestor process has finished");
134 		} catch (InterruptedException e) {
135 			throw new MojoExecutionException(e.getMessage(), e);
136 		}
137 	}

139 	private Collection<String> buildClasspath() throws MojoExecutionException {
140 		Collection<Artifact> artifacts = resolveDependencies(KEY_REQUESTOR, requestorClasspath, JavaScopes.TEST);
141 		Collection<String> classPathList = ArtifactUtil.toString(artifacts);
142 		classPathList.add(project.getBuild().getTestOutputDirectory());
143 		return classPathList;
144 	}

146 	private String buildCodeBase(List<String> websterRoots) throws MojoExecutionException {
147 		String host = "http://" + Booter.getWebsterHostName() + ":" + TestCycleHelper.getInstance().getWebsterPort();
148 		Collection<Artifact> artifacts = resolveDependencies(KEY_REQUESTOR, requestorCodebase, JavaScopes.TEST);
149 
150 		try {
151 			String repositoryPath = repositorySystemSession.getLocalRepository().getBasedir().getCanonicalPath();
152 			List<String> codeBaseList = new LinkedList<String>();
153 			for (Artifact artifact : artifacts) {
154 				File artifactFile = artifact.getFile();
155 				String artifactPath = artifactFile.getCanonicalPath();
156 				if (artifactPath.startsWith(repositoryPath)) {
157 					// add jar from repository
158 					codeBaseList.add(host + artifactPath.substring(repositoryPath.length()));
159 				} else {
160 					// add jar from target
161 					codeBaseList.add(host + '/' + artifactFile.getName());
162 					websterRoots.add(artifactFile.getParent());
163 				}
164 			}
165 			return StringUtils.join(codeBaseList, " ");
166 		} catch (IOException e) {
167 			throw new MojoExecutionException(e.getMessage(), e);
168 		}
169 	}

171 	protected Collection<Artifact> resolveDependencies(String propertyKey, Collection<String> userEntries, String scope)
172 			throws MojoExecutionException {
173 		List<String> coordinates = new LinkedList<String>();
174 		if (propertyKey != null && project.getProperties().containsKey(propertyKey)) {
175 			coordinates.add(project.getProperties().getProperty(propertyKey));
176 		}
177 		if (userEntries != null) {
178 			coordinates.addAll(userEntries);
179 		}
180 		try {
181 			Set<Artifact> artifacts = new HashSet<Artifact>(coordinates.size());
182 			for (String coords : coordinates) {
183 				artifacts.addAll(resolveDependencies(new DefaultArtifact(coords), scope));
184 			}
185 			return artifacts;
186 		} catch (DependencyResolutionException e) {
187 			throw new MojoExecutionException(e.getMessage(), e);
188 		}
189 	}
