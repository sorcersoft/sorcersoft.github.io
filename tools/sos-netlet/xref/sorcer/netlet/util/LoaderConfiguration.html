29  /**
30   * Class used to configure a RootLoader from a stream or by using
31   * it's methods.
32   * <p/>
33   * The stream can be for example a FileInputStream from a file with
34   * the following format:
35   * <p/>
36   * <pre>
37   * # comment
38   * main is classname
39   * load path
40   * load file
41   * load pathWith${property}
42   * load pathWith!{required.property}
43   * load path/*.jar
44   * load path/&#42;&#42;/&#42;.jar
45   * </pre>
46   * <ul>
47   * <li>All lines starting with "#" are ignored.</li>
48   * <li>The "main is" part may only be once in the file. The String
49   * afterwards is the name of a class with a main method. </li>
50   * <li>The "load" command will add the given file or path to the
51   * classpath in this configuration object. If the path does not
52   * exist, the path will be ignored.
53   * </li>
54   * <li>properties referenced using !{x} are required.</li>
55   * <li>properties referenced using ${x} are not required. If the
56   * property does not exist the whole load instruction line will
57   * be ignored.</li>
58   * <li>* is used to match zero or more characters in a file.</li>
59   * <li>** is used to match zero or more directories.</li>
60   * <li>Loading paths with <code>load ./*.jar</code> or <code>load *.jar</code> are not supported.</li>
61   * </ul>
62   * <p/>
63   * Defining the main class is required unless setRequireMain(boolean) is
64   * called with false, before reading the configuration.
65   * You can use the wildcard "*" to filter the path, but only for files, not
66   * directories. To match directories use "**". The ${propertyname} is replaced by the value of the system's
67   * property name. You can use user.home here for example. If the property does
68   * not exist, an empty string will be used. If the path or file after the load
69   * command does not exist, the path will be ignored.
70   *
71   * @author Jochen Theodorou
72   * @version $Revision$
73   * @see RootLoader
74   */
75  public class LoaderConfiguration {

90      /**
91       * configures this loader with a stream
92       *
93       * @param is stream used to read the configuration
94       * @throws java.io.IOException if reading or parsing the contents of the stream fails
95       */
96      public void configure(InputStream is) throws IOException {
97          BufferedReader reader = new BufferedReader(new InputStreamReader(is));
98          int lineNumber = 0;
99  
100         while (true) {
101             String line = reader.readLine();
102             if (line == null) break;
103 
104             line = line.trim();
105             lineNumber++;
106 
107             if (line.startsWith("#") || line.length() == 0) continue;
108 
109             if (line.startsWith(LoaderConfigurationHelper.LOAD_PREFIX)) {
110                 String loadPath = line.substring(LoaderConfigurationHelper.LOAD_PREFIX.length()).trim();
111                 loadUrls(LoaderConfigurationHelper.load(loadPath));
112             } else if (line.startsWith(LoaderConfigurationHelper.CODEBASE_PREFIX)) {
113                 // Webster is not yet started so this will work only if we use jars from external websters
114                 // specified using: http://.... or mvn://... @address:port
115                 // or if the environment points to a running system webster
116                 List<String> codebasePaths = new ArrayList<String>();
117                 codebasePaths.add(line);
118                 loadUrls(LoaderConfigurationHelper.setCodebase(codebasePaths, SorcerEnv.getWebsterUrl().toString(), System.out));
119             } else if (line.startsWith(MAIN_PREFIX)) {
120                 if (main != null)
121                     throw new IOException("duplicate definition of main in line " + lineNumber + " : " + line);
122                 main = line.substring(MAIN_PREFIX.length()).trim();
123             } else if (line.startsWith(PROP_PREFIX)) {
124                 String params = line.substring(PROP_PREFIX.length()).trim();
125                 int index = params.indexOf('=');
126                 if (index == -1) {
127                     throw new IOException("unexpected property format - expecting name=value" + lineNumber + " : " + line);
128                 }
129                 String propName = params.substring(0, index);
130                 String propValue= LoaderConfigurationHelper.assignProperties(params.substring(index + 1));
131                 System.setProperty(propName, propValue);
132             } else {
133                 throw new IOException("unexpected line in " + lineNumber + " : " + line);
134             }
135         }
136 
137         if (requireMain && main == null) throw new IOException("missing main class definition in config file");
138     }

140     /**
141      * Load a possibly filtered path. Filters are defined
142      * by using the * wildcard like in any shell.
143      */
144     private void loadUrls(List<URL> urls) {
145         for (URL url : urls)
146             addUrl(url);
147     }

170     /*
171      * separates the given path at the last '/'
172      */
173     private String getParentPath(String filter) {
174         int index = filter.lastIndexOf('/');
175         if (index == -1) return "";
176         return filter.substring(index + 1);

179     /**
180      * Adds a url to the classpath if it exists.
181      *
182      * @param url the url to add
183      */
184     public void addUrl(URL url) {
185         if (url!= null) {
186             if (url.getProtocol().equalsIgnoreCase("http")) {
187                 try {
188                     HttpURLConnection huc =  ( HttpURLConnection ) url.openConnection();
189                     huc.setRequestMethod("HEAD");
190                     if (huc.getResponseCode() == HttpURLConnection.HTTP_OK)
191                     classPath.add(url);
192                 } catch (ProtocolException e) {
193                     logger.severe("Problem with protocol while loading URL to classpath: " + url.toString() + "\n" + e.getMessage());
194                 } catch (IOException e) {
195                     logger.severe("Problem adding remote file to classpath, file does not exist: " + url.toString() + "\n" + e.getMessage());
196                 }
197             } else if (url.getProtocol().equalsIgnoreCase("file") && new File(url.getFile()).exists()) {
198                 classPath.add(url);
199             } else {
200                 logger.severe("Unsupported protocol to be added to classpath: " + url.toString());
201             }
202         }
203     }

205     /**
206      * Adds a file to the classpath if it exists.
207      *
208      * @param file the file to add
209      */
210     public void addFile(File file) {
211         if (file != null && file.exists()) {
212             try {
213                 classPath.add(file.toURI().toURL());
214             } catch (MalformedURLException e) {
215                 throw new AssertionError("converting an existing file to an url should have never thrown an exception!");
216             }
217         }
218     }

229     /**
230      * Adds a classpath to this configuration. It expects a string with
231      * multiple paths, separated by the system dependent path separator.
232      * Expands wildcards, e.g. dir/* into all the jars in dir.
233      *
234      * @param path the path as a path separator delimited string
235      * @see java.io.File#pathSeparator
236      */
237     public void addClassPath(String path) {
238        String[] paths = path.split(File.pathSeparator);
239         for (String cpPath : paths) {
240             // Check to support wild card classpath
241             if (cpPath.endsWith("*")) {
242                 File dir = new File(cpPath.substring(0, cpPath.length() - 1));
243                 File[] files = dir.listFiles();
244                 if (files != null) {
245                     for (File file : files) {
246                         if (file.isFile() && file.getName().endsWith(".jar")) addFile(file);
247                     }
248                 }
249             } else {
250                 addFile(new File(cpPath));
251             }
252         }
253     }
