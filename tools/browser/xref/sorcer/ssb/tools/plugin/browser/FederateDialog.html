55      public void actionPerformed(ActionEvent evt){
56      	boolean isBidirectional=bidirectioal.isSelected();
57      	ServiceNode srcNode=(ServiceNode)src.getSelectedItem();
58      	ServiceNode destNode=(ServiceNode)dest.getSelectedItem();
59      	
60      	if(srcNode==destNode){
61      		Toolkit.getDefaultToolkit().beep();
62      		return;
63      	}
64      	
65      	ServiceRegistrar srcProxy=(ServiceRegistrar)srcNode.getProxy();
66      	ServiceRegistrar destProxy=(ServiceRegistrar)destNode.getProxy();
67      	try{
68  	    	JoinAdmin srcJa=(JoinAdmin)((Administrable)srcProxy).getAdmin();
69  	    	JoinAdmin destJa=(JoinAdmin)((Administrable)destProxy).getAdmin();
70  	    	destJa.addLookupLocators(new LookupLocator[]{srcProxy.getLocator()});
71  	    	if(isBidirectional){
72  	    		srcJa.addLookupLocators(new LookupLocator[]{destProxy.getLocator()});
73  	    	}
74  	    	
75      	}catch(Exception ex){
76      		JOptionPane.showMessageDialog(frame,ex,"SSB: exception",JOptionPane.WARNING_MESSAGE);
77      	}
78  
79      	
80      	dispose();
81      }
