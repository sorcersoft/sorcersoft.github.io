62  /**
63   * Distributed lock manager is responsible for maintaining the lock information
64   * consistent on all participating nodes.
65   * 
66   * <p>A modified version of the original which provides support for leased 
67   * locks, the holding of a resource against a given lock and state
68   * replication.</p>
69   *
70   * @author Roman Rokytskyy (rrokytskyy@acm.org)
71   * @author Dan Creswell (dan@dancres.org)
72   */

74                                                 LockManager,
75                                                 MessageListener {

149     /**
150        Inserts a resource locally.
151      */
152     private boolean localInsert(LockDecree aDecree) {
153         removeExpired(aDecree);
154 
155         if (heldResources.get(aDecree.getKey()) != null)
156             return false;
157 
158         if (heldLocks.get(aDecree.getKey()) != null)
159             return false;
160 
161         heldResources.put(aDecree.getKey(), aDecree);
162         return true;
163     }
164 
165     private boolean localDelete(LockDecree aDecree) {
166         removeExpired(aDecree);
167 
168         return ((heldResources.remove(aDecree.getKey())) != null);
169     }

171     /**
172      * Performs local lock. This method also performs the clean-up of the lock
173      * table, all expired locks are removed.
174      */
175     private boolean localLock(LockDecree lockDecree) {
176         // remove expired locks
177         removeExpired(lockDecree);
178 
179         LockDecree localLock =
180             (LockDecree) heldLocks.get(lockDecree.getKey());
181 
182         if (localLock == null) {
183 
184             // promote lock into commited state
185             lockDecree.commit();
186 
187             // Overridden: DCC - hold lock regardless, we want to replicate
188             //
189             // no lock exist, perform local lock, note:
190             // we do not store locks that were requested by other manager.
191             // if (lockDecree.managerId.equals(id))
192             // System.out.println("LocalLock: " + lockDecree.getKey() + 
193             //                    ", " + lockDecree);
194 
195             heldLocks.put(lockDecree.getKey(), lockDecree);
196 
197             // everything is fine :)
198             return true;
199         } else
200         if (localLock.requester.equals(lockDecree.requester))
201             // requester already owns the lock
202             return true;
203         else
204             // lock does not belong to requester
205             return false;
206 
207     }

209     /**
210      * Releases lock locally.
211      * 
212      * @param lockDecree instance of {@link LockDecree} describing the lock.
213      */
214     private boolean localRelease(LockDecree lockDecree) {
215         // remove expired locks
216         removeExpired(lockDecree);
217 
218         LockDecree localLock=
219                 (LockDecree) heldLocks.get(lockDecree.getKey());
220 
221         if(localLock == null) {
222             // no lock exist
223             return true;
224         }
225         else if(localLock.requester.equals(lockDecree.requester)) {
226             // requester owns the lock, release the lock
227             heldLocks.remove(lockDecree.getKey());
228             return true;
229         }
230         else
231         // lock does not belong to requester
232             return false;
233     }

291      * 
292      * @param decree instance of {@link LockDecree} containing information
293      * about the lock.

295     private boolean canRelease(LockDecree decree) {
296         // clean expired locks
297         removeExpired(decree);
298 
299         // we need to check only hold locks, because
300         // prepared locks cannot contain the lock
301         LockDecree lock = (LockDecree)heldLocks.get(decree.getKey());

303             // check if this holds...
304             return true;
305         else
306             return lock.requester.equals(decree.requester);

313     /**
314      * Removes expired locks.
315      * 
316      * @param decree instance of {@link LockDecree} describing the lock.
317      */
318     private void removeExpired(LockDecree decree) {
319         // remove the invalid (expired) lock
320         LockDecree localLock = (LockDecree)heldLocks.get(decree.getKey());
321         if (localLock != null && !localLock.isValid()) {
322             // System.out.println("Clear out dead lock");
323             heldLocks.remove(localLock.getKey());
324         }
325     }

409     /**
410      * Unlocks an object with <code>lockId</code> on behalf of the specified

412      * @param lockId <code>long</code> representing the object to be unlocked.
413      * @param owner object that releases the lock.
414      *
415      * @throws LockNotReleasedException when the lock cannot be released.
416      * @throws IllegalArgumentException if lockId or owner are not serializable.
417      */
418     public void unlock(Object lockId, Object owner, Transaction aTxn)
419         throws LockNotReleasedException, ChannelException
420     {
421 
422         // System.err.println("Performing voting");
423 
424         boolean released = false;
425 
426         try {
427             released =
428                 votingAdapter.vote(new ReleaseLockDecree(lockId,
429                                                          owner, id, aTxn),
430                                    VOTE_TIMEOUT);
431         } catch (Exception anE) {
432             System.err.println("Server failed");
433             anE.printStackTrace(System.err);
434             throw new ChannelException("Failed", anE);
435         }
436         
437         if (!released)
438             throw new LockNotReleasedException("Lock cannot be unlocked.");
439     }

445     /**
446      * Checks the list of prepared locks/unlocks to determine if we are in the
447      * middle of the two-phase commit process for the lock acqusition/release.
448      * Here we do not tolerate if the request comes from the same node on behalf
449      * of the same owner.
450      * 
451      * @param preparedContainer either <code>preparedLocks</code> or
452      * <code>preparedReleases</code> depending on the situation.
453      * 
454      * @param requestedDecree instance of <code>LockDecree</code> representing
455      * the lock.
456      */
457     private boolean checkPrepared(Map preparedContainer, 
458         LockDecree requestedDecree)
459     {
460         LockDecree preparedDecree =
461             (LockDecree)preparedContainer.get(requestedDecree.getKey());
462 
463         // if prepared lock is not valid, remove it from the list
464         if ((preparedDecree != null) && !preparedDecree.isValid()) {
465             preparedContainer.remove(preparedDecree.getKey());
466 
467             preparedDecree = null;
468         }
469 
470         if (preparedDecree != null) {
471             if (requestedDecree.requester.equals(preparedDecree.requester))
472                 return true;
473             else
474                 return false;
475         } else
476             // it was not prepared... sorry...
477             return true;
478     }

480     /**
481      * Prepare phase for the lock acquisition or release.
482      * 
483      * @param decree should be an instance <code>LockDecree</code>, if not,
484      * we throw <code>VoteException</code> to be ignored by the
485      * <code>VoteChannel</code>.
486      * 
487      * @return <code>true</code> when preparing the lock operation succeeds.
488      * 
489      * @throws VoteException if we should be ignored during voting.
490      */
491     public synchronized boolean prepare(Object decree) throws VoteException {
492         // System.err.println("Processing decree: " + decree.getClass().getName());
493 
494         if (!(decree instanceof LockDecree))
495             throw new VoteException("Uknown decree type. Ignore me.");
496             
497         if (decree instanceof AcquireLockDecree) {
498             // System.err.println("In acquire");
499             AcquireLockDecree acquireDecree = (AcquireLockDecree)decree;
500 
501 			if (log.isLoggable(Level.FINER))
502 				log.fine("Preparing to acquire decree " + acquireDecree.lockId);
503             
504             // System.err.println("Checking prepared");
505 
506             if (!checkPrepared(preparedLocks, acquireDecree))
507                 // there is a prepared lock owned by third party
508                 return false;
509 
510             // System.err.println("Checking can lock");
511 
512             if (canLock(acquireDecree)) {
513                 preparedLocks.put(acquireDecree.getKey(), acquireDecree);
514                 // System.out.println("canLock said: " + canLock(acquireDecree));
515                 return true;
516             } else
517                 // we are unable to aquire local lock
518                 return false;
519         } else if (decree instanceof ReleaseLockDecree) {
520 
521             // System.out.println("In release");
522 
523             ReleaseLockDecree releaseDecree = (ReleaseLockDecree)decree;
524             
525 			if (log.isLoggable(Level.FINER))
526 				log.fine("Preparing to release decree " + releaseDecree.lockId);
527 
528             // System.out.println("Check prepared");
529 
530             if (!checkPrepared(preparedReleases, releaseDecree))
531                 // there is a prepared release owned by third party
532                 return false;
533 
534             // System.out.println("Validating release");
535 
536             if (canRelease(releaseDecree)) {
537                 preparedReleases.put(releaseDecree.getKey(), releaseDecree);
538                 // we have local lock and the prepared lock
539                 // System.out.println("Passed");
540                 return true;
541             } else {
542                 // System.out.println("Failed");
543                 // we were unable to aquire local lock
544                 return false;
545             }
546         } else if (decree instanceof NewDecree) {
547             NewDecree resourceDecree = (NewDecree) decree;
548 
549             if(log.isLoggable(Level.FINER))
550                 log.fine("Preparing to newResource decree " +
551                           resourceDecree.lockId);
552 
553             if (!checkPrepared(preparedLocks, resourceDecree)) {
554                 // System.out.println("Blocking lockprepare");
555                 return false;
556             }
557 
558             if (!checkPrepared(preparedNews, resourceDecree)) {
559                 // System.out.println("Blocking resourceprepare");
560                 return false;
561             }
562 
563             if (canInsert(resourceDecree)) {
564                 preparedNews.put(resourceDecree.getKey(), resourceDecree);
565                 // System.out.println("canLock said: " + canLock(resourceDecree));
566                 return true;
567             } else {
568                 // System.out.println("cantInsert");
569                 // we are blocked by a lock
570                 return false;
571             }
572         } else if (decree instanceof DeleteDecree) {
573             DeleteDecree resourceDecree = (DeleteDecree) decree;
574 
575             if(log.isLoggable(Level.FINER))
576                 log.fine("Preparing to deleteResource decree " +
577                           resourceDecree.lockId);
578 
579             if (!checkPrepared(preparedLocks, resourceDecree))
580                 return false;
581 
582             if (!checkPrepared(preparedDeletes, resourceDecree))
583                 return false;
584 
585             if (canDelete(resourceDecree)) {
586                 preparedDeletes.put(resourceDecree.getKey(), resourceDecree);
587 
588                 return true;
589             } else
590                 return false;
591         }
592 
593         // we should not be here
594         return false;
595     }

597     /**
598      * Commit phase for the lock acquisition or release.
599      * 
600      * @param decree should be an instance <code>LockDecree</code>, if not,
601      * we throw <code>VoteException</code> to be ignored by the
602      * <code>VoteChannel</code>.
603      * 
604      * @return <code>true</code> when commiting the lock operation succeeds.
605      * 
606      * @throws VoteException if we should be ignored during voting.
607      */
608     public synchronized boolean commit(Object decree) throws VoteException {
609         if (!(decree instanceof LockDecree))
610             throw new VoteException("Uknown decree type. Ignore me.");
611 
612         if (decree instanceof AcquireLockDecree) {
613 
614             // System.err.println("Commiting acquire");
615 
616 			if (log.isLoggable(Level.FINER))
617 				log.fine("Committing decree acquisition "
618 						+ ((LockDecree) decree).lockId);
619 
620             // System.err.println("Checking prepared");
621 
622             if (!checkPrepared(preparedLocks, (LockDecree)decree))
623                 // there is a prepared lock owned by third party
624                 return false;
625 
626             if (localLock((LockDecree)decree)) {
627                 preparedLocks.remove(((LockDecree)decree).getKey());
628 
629                 // System.err.println("Local lock succeeded");
630                 return true;
631             } else
632                 return false;
633         } else if (decree instanceof ReleaseLockDecree) {
634             
635 			if (log.isLoggable(Level.FINER))
636 				log.fine("Committing decree release "
637 						+ ((LockDecree) decree).lockId);
638             
639             if (!checkPrepared(preparedReleases, (LockDecree)decree))
640                 // there is a prepared release owned by third party
641                 return false;
642 
643             // System.err.println("Looking to release");
644             if (localRelease((LockDecree)decree)) {
645                 preparedReleases.remove(((LockDecree)decree).getKey());
646                 return true;
647             } else
648                 return false;
649         } else if (decree instanceof NewDecree) {
650 
651 			if (log.isLoggable(Level.FINER))
652 				log.fine("Committing decree newResource "
653 						+ ((LockDecree) decree).lockId);
654             
655             if (!checkPrepared(preparedLocks, (LockDecree)decree))
656                 // there is a prepared lock owned by third party
657                 return false;
658 
659             if (!checkPrepared(preparedNews, (LockDecree)decree))
660                 // there is a prepared resource owned by third party
661                 return false;
662 
663             if (localInsert((LockDecree)decree)) {
664                 preparedNews.remove(((LockDecree)decree).getKey());
665                 return true;
666             } else
667                 return false;
668         } else if (decree instanceof DeleteDecree) {
669         	if (log.isLoggable(Level.FINER))
670                 log.fine("Committing decree deleteResource " +
671                           ((LockDecree)decree).lockId);
672 
673             if (!checkPrepared(preparedLocks, (LockDecree)decree))
674                 // there is a prepared lock owned by third party
675                 return false;
676 
677             if (!checkPrepared(preparedDeletes, (LockDecree)decree))
678                 // there is a prepared delete owned by third party
679                 return false;
680 
681             if (localDelete((LockDecree)decree)) {
682                 preparedDeletes.remove(((LockDecree)decree).getKey());
683                 return true;
684             } else
685                 return false;
686         }
687 
688         // we should not be here
689         return false;
690     }

692     /**
693      * Abort phase for the lock acquisition or release.
694      * 
695      * @param decree should be an instance <code>LockDecree</code>, if not,
696      * we throw <code>VoteException</code> to be ignored by the
697      * <code>VoteChannel</code>.
698      * 
699      * @throws VoteException if we should be ignored during voting.
700      */
701     public synchronized void abort(Object decree) throws VoteException {
702         if (!(decree instanceof LockDecree))
703             throw new VoteException("Uknown decree type. Ignore me.");
704 
705         if (decree instanceof AcquireLockDecree) {
706             
707         	if (log.isLoggable(Level.FINER))
708                 log.fine("Aborting decree acquisition " +
709                           ((LockDecree)decree).lockId);
710             
711             if (!checkPrepared(preparedLocks, (LockDecree)decree))
712                 // there is a prepared lock owned by third party
713                 return;
714 
715             preparedLocks.remove(((LockDecree)decree).getKey());
716         } else if (decree instanceof ReleaseLockDecree) {
717             
718         	if (log.isLoggable(Level.FINER))
719                 log.fine("Aborting decree release " +
720                           ((LockDecree)decree).lockId);
721             
722             if (!checkPrepared(preparedReleases, (LockDecree)decree))
723                 // there is a prepared release owned by third party
724                 return;
725 
726             preparedReleases.remove(((LockDecree)decree).getKey());
727         } else if (decree instanceof NewDecree) {
728             
729         	if (log.isLoggable(Level.FINER))
730                 log.fine("Aborting decree newResource " +
731                           ((LockDecree)decree).lockId);
732             
733             if (!checkPrepared(preparedNews, (LockDecree)decree))
734                 // there is a prepared resource owned by third party
735                 return;
736 
737             preparedNews.remove(((LockDecree)decree).getKey());
