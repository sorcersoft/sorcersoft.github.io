590 	public static void copyFile(String src, String dest) throws IOException {
591 
592 		FileInputStream fis = new FileInputStream(src);
593 		FileOutputStream fos = new FileOutputStream(dest);
594 		byte[] b = new byte[2048];
595 		int nBytes = fis.read(b);
596 		while (nBytes != -1) {
597 			fos.write(b, 0, nBytes);
598 			nBytes = fis.read(b);
599 		}
600 		fis.close();
601 		fos.close();
602 	}

604 	static void registerOSXApp(sorcer.ssb.osx.OSXApplication theApp) {
605 		try {
606 			Class osxAdapter = Class.forName("sorcer.ssb.osx.OSXAdapter");
607 
608 			Class[] defArgs = { sorcer.ssb.osx.OSXApplication.class };
609 			Method registerMethod = osxAdapter.getDeclaredMethod(
610 					"registerMacOSXApplication", defArgs);
611 			if (registerMethod != null) {
612 				Object[] args = { theApp };
613 				registerMethod.invoke(osxAdapter, args);
614 			}
615 			// This is slightly gross. to reflectively access methods with
616 			// boolean args,
617 			// use "boolean.class", then pass a Boolean object in as the arg,
618 			// which apparently
619 			// gets converted for you by the reflection system.
620 			defArgs[0] = boolean.class;
621 			Method prefsEnableMethod = osxAdapter.getDeclaredMethod(
622 					"enablePrefs", defArgs);
623 			if (prefsEnableMethod != null) {
624 				Object args[] = { Boolean.TRUE };
625 				prefsEnableMethod.invoke(osxAdapter, args);
626 			}
627 		} catch (NoClassDefFoundError e) {
628 			// This will be thrown first if the OSXAdapter is loaded on a system
629 			// without the EAWT
630 			// because OSXAdapter extends ApplicationAdapter in its def
631 			System.err
632 					.println("This version of Mac OS X does not support the Apple EAWT.  Application Menu handling has been disabled ("
633 							+ e + ")");
634 		} catch (ClassNotFoundException e) {
635 			// This shouldn't be reached; if there's a problem with the
636 			// OSXAdapter we should get the
637 			// above NoClassDefFoundError first.
638 			System.err
639 					.println("This version of Mac OS X does not support the Apple EAWT.  Application Menu handling has been disabled ("
640 							+ e + ")");
641 		} catch (Exception e) {
642 			System.err.println("Exception while loading the OSXAdapter:");
643 			e.printStackTrace();
