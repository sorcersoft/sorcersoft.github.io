165 	private String [] parse(JEditorPane ep){
166 		ArrayList list=new ArrayList();
167 		String text=ep.getText();
168 		StringTokenizer tok=new StringTokenizer(text,"\n");
169 		while(tok.hasMoreTokens()){
170 			String ln=tok.nextToken().trim();
171 			if(ln.startsWith("//")==false && ln.length()>0){
172 				list.add(ln);
173 			}
174 		}
175 		String [] ret=new String[list.size()];
176 		list.toArray(ret);
177 		return ret;
178 	}
