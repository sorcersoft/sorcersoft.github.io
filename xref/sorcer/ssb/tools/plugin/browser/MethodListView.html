154 								}

156 										selectedRows[i], 1);

172 	private String parseArray(Object[] data) {
173 		StringBuffer buf = new StringBuffer();
174 		for (int i = 0; i < data.length - 1; i++) {
175 			if (data[i] != null) {
176 				buf.append(data[i].toString());
177 				buf.append(" ");
178 			}
179 		}
180 		if (data.length > 0 && data[data.length - 1] != null) {
181 			buf.append(data[data.length - 1].toString());
182 		}
183 		return buf.toString();
184 	}

186 	private class MethodCellRenderer extends DefaultTableCellRenderer {
187 
188 		public Component getTableCellRendererComponent(JTable table,
189 				Object value, boolean isSelected, boolean hasFocus, int row,
190 				int column) {
191 
192 			JLabel lab = (JLabel) super.getTableCellRendererComponent(table,
193 					value, isSelected, hasFocus, row, column);
194 			lab.setIcon(null);
195 			lab.setForeground(Color.black);
196 			if (column == NODE_COLUMN) {
197 				PropertiesNode pn = (PropertiesNode) table.getValueAt(row,
198 						NODE_COLUMN);
199 				Method method = (Method) pn.getAdditionalData();
200 				lab.setText(LusTree.parseMethod(method));
201 				Class[] params = method.getParameterTypes();
202 				if (!(params.length == 0 && !method.getReturnType().getName()
203 						.equals("void"))) {
204 					lab.setForeground(Color.lightGray);
205 				}
206 
207 			}
208 			return lab;
209 		}
