142 	/*
143 	 * Poll the JavaSpaceAdmin proxy every <code>refreshRate</code> millis If we
144 	 * get an Exception, drop out of the loop and report it The most likely
145 	 * cause of an Exception occuring will be down to a retrieved entry not
146 	 * having a valid codebase set for it
147 	 * 
148 	 * If the client application that is writing the entries to the space is
149 	 * being run from within the SSB environment, makes sure you have set a
150 	 * codebase alias, but right clicking on the project icon and selecting
151 	 * "Add alias to webserver" and then selecting "Yes" when prompted.
152 	 */
153 	public void run() {
154 		System.out.println("Starting JavaSpace05 reader thread");
155 
156 		while (!_updater.isInterrupted()) {
157 			try {
158 
159 				doUpdates();
160 
161 				Thread.sleep(refreshRate);
162 
163 			} catch (InterruptedException ex) {
164 				System.out.println("Stopping JavaSpace05 reader thread");
165 				_updater = null;
166 				return;
167 			} catch (Exception ex) {
168 				// drop out here if we get an exception,
169 				// likely causes are UnusableEntryExceptions
170 				// because the client program that wrote the entry
171 				// didn't set a codebase (or set a wrong one) or the HTTPD of
172 				// the actual
173 				// codebase isn't working
174 				// JOptionPane.showMessageDialog(this,ex);
175 				ex.printStackTrace();
176 				_updater = null;
177 				return;
178 			}
179 		}
180 
181 	}

184 	 * Request all the entries currently in the space.
185 	 * 
186 	 * @throws TransactionException
187 	 *             ,RemoteException,UnusableEntryException
188 	 */
189 	private void doUpdates() throws TransactionException,
190 			UnusableEntryException, RemoteException {
191 
192 		if (firstTime) {
193 			firstTime = false;
194 			JTable table = new JTable(entryCountTableModel);
195 			JScrollPane scrollPane = new JScrollPane(table);
196 			remove(introView);
197 			add(scrollPane, BorderLayout.CENTER);
198 			invalidate();
199 			getParent().validate();
200 			addMouseListener(table);
201 
202 		}
203 		// here we're asking for all the entries in the space
204 		// you could modify this sample GUI to allow user input of an entry
205 		// class name
206 		// AdminIterator iter=javaSpaceAdmin.contents(null/*Entry
207 		// template*/,null/*Transaction*/);
208 		ArrayList tmplist = new ArrayList();
209 		tmplist.add(null);
210 		MatchSet iter = javaSpace.contents(tmplist, null, 5L * 60000L,
211 				Integer.MAX_VALUE);
212 
213 		List list = new ArrayList();
214 		Map entryMap = new HashMap();
215 		int ueCount = 0;
216 		int got = 0;
217 		while (iter != null && true) {
218 			try {
219 				Entry e = iter.next();
220 				if (e == null)/* || got>=maxEntries) */{
221 					break;
222 				}
223 				got++;
224 				Class entryClass = e.getClass();
225 
226 				String entryClassName = entryClass.getName();
227 
228 				Object[] data = (Object[]) entryMap.get(entryClassName);
229 				if (data == null) {
230 					Object template = null;
231 					try {
232 						template = entryClass.newInstance();
233 
234 					} catch (Exception ex) {
235 						System.out.println(ex);
236 					}
237 					entryMap.put(entryClassName, new Object[] { entryClassName,
238 							new Integer(1), template });
239 				} else {
240 					Integer count = (Integer) data[1];
241 					data[1] = new Integer(count.intValue() + 1);
242 				}
243 
244 			} catch (UnusableEntryException uee) {
245 				// add as UnusableEnrty
246 				uee.printStackTrace();
247 				ueCount++;
248 				// throw uee;
249 			}
250 		}
251 
252 		try {
253 			if (iter != null) {
254 				Lease lease = iter.getLease();
255 				if (lease != null) {
256 					lease.cancel();
257 				}
258 			}
259 
260 		} catch (Exception ex) {
261 			ex.printStackTrace();
262 		}
263 
264 		System.out.println("JavaSpace05 contents viewer retreived " + got
265 				+ " entries");
266 
267 		if (ueCount > 0) {
268 			entryMap.put(UNUSABLE_ENTRY, new Object[] { UNUSABLE_ENTRY,
269 					new Integer(ueCount), null });
270 		}
271 		// get the object counts
272 		Collection col = entryMap.values();
273 		List counterList = new ArrayList();
274 		counterList.addAll(col);
275 
276 		entryCountTableModel.update(counterList);
277 	}

480 		private void parseEntry(Object entry) {
481 			try {
482 				Class ec = entry.getClass();
483 				Field[] f = ec.getFields();
484 				for (int i = 0; i < f.length; i++) {
485 					Object[] fData = { f[i].getType().getName(),
486 							f[i].getName(), f[i].get(entry) };
487 					_data.add(fData);
488 				}
489 			} catch (Exception ex) {
490 				ex.printStackTrace();
491 			}
492 		}
493 
494 		public void clear() {
495 			_data = new ArrayList();
496 
497 			fireTableDataChanged();
498 		}
