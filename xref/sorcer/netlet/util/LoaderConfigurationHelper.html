55      public static List<URL> load(String str) {
56          List<URL> urlsList = new ArrayList<URL>();
57          URI uri;
58          str = assignProperties(str);
59          try {
60              uri = new URI(str);
61          } catch (URISyntaxException e) {
62              logger.log(Level.SEVERE, "Error while parsing URL " + str, e);
63              return urlsList;
64          }
65          String scheme = uri.getScheme();
66          if ("file".equals(scheme)) {
67              return getFilesFromFilteredPath(str);
68          } else if ("mvn".equals(scheme)) {
69              String url = str.substring(6);
70              // Check if URL specifies ana artifact on a remote webster
71              String[] urlEntries = url.split("@");
72              String finalUrl = null;
73              try {
74                  if (urlEntries.length > 1) {
75                      finalUrl = "http://" + urlEntries[1] + "/" +
76                              Resolver.resolveRelative(urlEntries[0]);
77                      if (existRemoteFile(new URL(finalUrl))) {
78                          urlsList.add(new URL(finalUrl));
79                      }
80                  } else {
81                      finalUrl = Resolver.resolveAbsolute(urlEntries[0]);
82                      if (finalUrl!=null && new File(finalUrl).exists())
83                          urlsList.add(new File(finalUrl).toURI().toURL());
84                      else
85                          logger.severe("Problem adding library to codebase, file not found for: " + urlEntries[0]);
86                  }
87              } catch (MalformedURLException e) {
88                      logger.severe("Problem creating URL: " + finalUrl);
89              }
90          } else if ("http".equals(scheme)) {
91              try {
92                  urlsList.add(new URL(str));
93              } catch (MalformedURLException e) {
94                  logger.severe("Problem creating URL: " + str);
95              }
96          } else {
97              if ("artifact".equals(scheme)) {
98                  ArtifactURLConfiguration artifactConf = new ArtifactURLConfiguration(uri.getSchemeSpecificPart());
99                  try {
100                     org.rioproject.resolver.Resolver resolver = getResolver();
101                     RemoteRepository[] repos = artifactConf.getRepositories();
102 
103                     URL[] classpath = SorcerResolverHelper.toURLs(resolver.getClassPathFor(artifactConf.getArtifact(), repos));
104                     Collections.addAll(urlsList, classpath);
105                 } catch (ResolverException e) {
106                     logger.log(Level.SEVERE, "Could not resolve " + str, e);
107                 } catch (MalformedURLException e) {
108                     throw new RuntimeException(e);
109                 } catch (URISyntaxException e) {
110                     throw new RuntimeException(e);
111                 }
112             }
113         }
114         return urlsList;
115     }

123     public static String parseCodebase(URL websterUrl, String str) {
124         if ((!str.startsWith("mvn://")) &&  (!str.startsWith("http://")) && (!str.startsWith("artifact:"))) {
125             logger.severe("Codebase can only be specified using mvn://, http:// or artifact:");
126             return null;
127         }
128         if (str.startsWith("mvn://")) {
129             String url = str.substring(6);
130             // Check if URL specifies as artifact on a remote webster
131             String[] urlEntries = url.split("@");
132             String finalUrl = null;
133             try {
134                 if (urlEntries.length > 1) {
135                     finalUrl = "http://" + urlEntries[1] + "/" +
136                             Resolver.resolveRelative(urlEntries[0]);
137                     if (existRemoteFile(new URL(finalUrl))) {
138                         return new URL(finalUrl).toString();
139                     }
140                 } else if (websterUrl!=null) {
141                     finalUrl = Resolver.resolveAbsolute(websterUrl, urlEntries[0]);
142                     if (existRemoteFile(new URL(finalUrl)))
143                         return new URL(finalUrl).toString();
144                 }
145             } catch (MalformedURLException e) {
146                 logger.severe("Problem creating URL: " + finalUrl);
147             }
148         }
149         return str;
150     }

152     public static List<URL> setCodebase(List<String> codebaseLines, String websterStrUrl, PrintStream out) {
153         String curCodebase = System.getProperty(JavaSystemProperties.RMI_SERVER_CODEBASE);
154         StringBuilder codebaseSb = new StringBuilder();
155         if (curCodebase!=null) codebaseSb.append(curCodebase);
156         List<URL> codebaseUrls = new ArrayList<URL>();
157         URL websterUrl = null;
158         try {
159             if (websterStrUrl!=null)
160                 websterUrl = new URL(websterStrUrl);
161         } catch (MalformedURLException me) {
162         }
163         for (String codebaseStr : codebaseLines) {
164             if (codebaseStr.startsWith(LoaderConfigurationHelper.CODEBASE_PREFIX))
165                 codebaseStr = codebaseStr.substring(LoaderConfigurationHelper.CODEBASE_PREFIX.length()).trim();
166             if ((!codebaseStr.startsWith("mvn://")) &&  (!codebaseStr.startsWith("http://")) && (!codebaseStr.startsWith("artifact:"))) {
167                 if (out!=null) out.println("Codebase can only be specified using mvn://, http:// or artifact:");
168                 else logger.severe("Codebase can only be specified using mvn://, http:// or artifact:");
169                 return null;
170             }
171 
172             String parsedCodebase = LoaderConfigurationHelper.parseCodebase(websterUrl, codebaseStr);
173             try {
174                 codebaseUrls.add(new URL(parsedCodebase));
175             } catch (MalformedURLException me) {
176                 if (out!=null) out.println("Codebase url is malformed: " + me.getMessage());
177                 else logger.severe("Codebase url is malformed: " + me.getMessage());
178             }
179 
180             if (parsedCodebase!=null)
181                 codebaseSb.append(" ").append(parsedCodebase);
182         }
183         System.setProperty(JavaSystemProperties.RMI_SERVER_CODEBASE, codebaseSb.toString());
184         return codebaseUrls;
185     }

190     public static String assignProperties(String str) {
191         int propertyIndexStart = 0, propertyIndexEnd = 0;
192         boolean requireProperty = false;
193         String result = "";
194 
195         while (propertyIndexStart < str.length()) {
196             {
197                 int i1 = str.indexOf("${", propertyIndexStart);
198                 int i2 = str.indexOf("!{", propertyIndexStart);
199                 if (i1 == -1) {
200                     propertyIndexStart = i2;
201                 } else if (i2 == -1) {
202                     propertyIndexStart = i1;
203                 } else {
204                     propertyIndexStart = Math.min(i1, i2);
205                 }
206                 requireProperty = propertyIndexStart == i2;
207             }
208             if (propertyIndexStart == -1) break;
209             result += str.substring(propertyIndexEnd, propertyIndexStart);
210 
211             propertyIndexEnd = str.indexOf("}", propertyIndexStart);
212             if (propertyIndexEnd == -1) break;
213 
214             String propertyKey = str.substring(propertyIndexStart + 2, propertyIndexEnd);
215             String propertyValue = null;
216 
217             propertyValue = System.getProperty(propertyKey);
218 
219 
220             // assume properties contain paths
221             if (propertyValue == null) {
222                 if (requireProperty) {
223                     throw new IllegalArgumentException("Variable " + propertyKey + " in nsh.config references a non-existent System property! Try passing the property to the VM using -D" + propertyKey + "=myValue in JAVA_OPTS");
224                 } else {
225                     return null;
226                 }
227             }
228             propertyValue = getSlashyPath(propertyValue);
229             propertyValue = correctDoubleSlash(propertyValue,propertyIndexEnd,str);
230             result += propertyValue;
231 
232             propertyIndexEnd++;
233             propertyIndexStart = propertyIndexEnd;
234         }
235 
236         if (propertyIndexStart == -1 || propertyIndexStart >= str.length()) {
237             result += str.substring(propertyIndexEnd);
238         } else if (propertyIndexEnd == -1) {
239             result += str.substring(propertyIndexStart);
240         }
241 
242         return result;
243     }

246     public static boolean existRemoteFile(URL url) {
247         try {
248             HttpURLConnection huc =  ( HttpURLConnection ) url.openConnection();
249             huc.setRequestMethod("HEAD");
250             if (huc.getResponseCode() == HttpURLConnection.HTTP_OK)
251                 return true;
252         } catch (ProtocolException e) {
253             logger.severe("Problem with protocol while loading URL to classpath: " + url.toString() + "\n" + e.getMessage());
254         } catch (IOException e) {
255             logger.severe("Problem adding remote file to classpath, file does not exist: " + url.toString() + "\n" + e.getMessage());
256         }
257         return false;
258     }

261      * Get files to load a possibly filtered path. Filters are defined
262      * by using the * wildcard like in any shell.

264     public static List<URL> getFilesFromFilteredPath(String filter) {
265         List<URL> filesToLoad = new ArrayList<URL>();
266         if (filter == null) return null;
267         filter = getSlashyPath(filter);
268         int starIndex = filter.indexOf(WILDCARD);
269         if (starIndex == -1) {
270             try {
271                 filesToLoad.add(new File(filter).toURI().toURL());
272             } catch (MalformedURLException e) {
273                 logger.severe("Problem converting file to URL: " + e.getMessage());
274             }
275             //addFile(new File(filter));
276             return filesToLoad;
277         }
278         boolean recursive = filter.indexOf(ALL_WILDCARD) != -1;
279 
280         if (filter.lastIndexOf('/')<starIndex) {
281             starIndex=filter.lastIndexOf('/')+1;
282         }
283         String startDir = filter.substring(0, starIndex - 1);
284         File root = new File(startDir);
285 
286         filter = Pattern.quote(filter);
287         filter = filter.replaceAll("\\" + WILDCARD + "\\" + WILDCARD, MATCH_ALL);
288         filter = filter.replaceAll("\\" + WILDCARD, MATCH_FILE_NAME);
289         Pattern pattern = Pattern.compile(filter);
290 
291         final File[] files = root.listFiles();
292         if (files != null) {
293           //  filesToLoad.addAll(findMatchingFiles(files, pattern, recursive));
294         }
295         return filesToLoad;
296     }

298     private static List<File> findMatchingFiles(File[] files, Pattern pattern, boolean recursive) {
299         List<File> filesToLoad = new ArrayList<File>();
300         for (int i = 0; i < files.length; i++) {
301             File file = files[i];
302             String fileString = getSlashyPath(file.getPath());
303             Matcher m = pattern.matcher(fileString);
304             if (m.matches() && file.isFile()) {
305                 filesToLoad.add(file);
306                 //addFile(file);
307             }
308             if (file.isDirectory() && recursive) {
309                 final File[] dirFiles = file.listFiles();
310                 if (dirFiles != null) {
311                     filesToLoad.addAll(findMatchingFiles(dirFiles, pattern, true));
312                 }
313             }
314         }
315         return filesToLoad;
316     }

318     private static String correctDoubleSlash(String propertyValue, int propertyIndexEnd, String str) {
319         int index = propertyIndexEnd+1;
320         if ( index<str.length() && str.charAt(index)=='/' &&
321                 propertyValue.endsWith("/") &&
322                 propertyValue.length()>0)
323         {
324             propertyValue = propertyValue.substring(0,propertyValue.length()-1);
325         }
326         return propertyValue;
327     }

329     // change path representation to something more system independent.
330     // This solution is based on an absolute path
331     private static String getSlashyPath(final String path) {
332         String changedPath = path;
333         if (File.separatorChar != '/')
334             changedPath = changedPath.replace(File.separatorChar, '/');
335 
336         return changedPath;
337     }
