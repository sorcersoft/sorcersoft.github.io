582 				exertions.add((Exertion) elems[i]);

585 		if (exertions.size() > 0) {
586 			Job j = job(elems);

595 	public static <T> Job job(T... elems) throws ExertionException,
596 			ContextException, SignatureException {
597 		String name = getUnknown();
598 		Signature signature = null;
599 		ControlContext control = null;
600 		Context<?> data = null;
601 		ReturnPath rp = null;
602 		List<Exertion> exertions = new ArrayList<Exertion>();
603 		List<Pipe> pipes = new ArrayList<Pipe>();
604 
605 		for (int i = 0; i < elems.length; i++) {
606 			if (elems[i] instanceof String) {
607 				name = (String) elems[i];
608 			} else if (elems[i] instanceof Exertion) {
609 				exertions.add((Exertion) elems[i]);
610 			} else if (elems[i] instanceof ControlContext) {
611 				control = (ControlContext) elems[i];
612 			} else if (elems[i] instanceof Context) {
613 				data = (Context<?>) elems[i];
614 			} else if (elems[i] instanceof Pipe) {
615 				pipes.add((Pipe) elems[i]);
616 			} else if (elems[i] instanceof Signature) {
617 				signature = ((Signature) elems[i]);
618 			} else if (elems[i] instanceof ReturnPath) {
619 				rp = ((ReturnPath) elems[i]);
620 			}
621 		}
622 
623 		Job job = null;
624         boolean defaultSig = false;
625         if (signature == null) {
626             signature = sig("service", Jobber.class);
627             defaultSig = true;
628         }
629         if (signature instanceof NetSignature) {
630             job = new NetJob(name);
631         } else if (signature instanceof ObjectSignature) {
632             job = new ObjectJob(name);
633         }
634         if (!defaultSig) {
635             job.getSignatures().clear();
636             job.addSignature(signature);
637         } else {
638             job.addSignature(signature);
639         }
640         if (data != null)
641 			job.setContext(data);
642 
643 		if (rp != null) {
644 			((ServiceContext) job.getDataContext()).setReturnPath(rp);
645 		}
646 
647 		if (job instanceof NetJob && control != null) {
648 			job.setControlContext(control);
649 			if (control.getAccessType().equals(Access.PULL)) {
650 				Signature procSig = job.getProcessSignature();
651 				procSig.setServiceType(Spacer.class);
652 				job.getSignatures().clear();
653 				job.addSignature(procSig);
654 				if (data != null)
655 					job.setContext(data);
656 				else
657 					job.getDataContext().setExertion(job);
658 			}
659 		}
660 		if (exertions.size() > 0) {
661 			for (Exertion ex : exertions) {
662 				job.addExertion(ex);
663 			}
664 			for (Pipe p : pipes) {
665 				logger.finer("from context: "
666 						+ ((Exertion) p.in).getDataContext().getName()
667 						+ " path: " + p.inPath);
668 				logger.finer("to context: "
669 						+ ((Exertion) p.out).getDataContext().getName()
670 						+ " path: " + p.outPath);
671 				((Exertion) p.out).getDataContext().connect(p.outPath,
672 						p.inPath, ((Exertion) p.in).getContext());
673 			}
674 		} else
675 			throw new ExertionException("No component exertion defined for job: " + job.getName());
676 
677 		return job;
678 	}

878         }

881         if (obj == null) {
