58  			String outputMessage = "processed by " + getHostname();
59  			context.putValue(

61  			context.putValue(ServiceAccount.COMMENT, outputMessage);
62  
63  		} catch (Exception ex) {
64  			// ContextException, UnknownHostException
65  			throw new RemoteException(selector + " process execption", ex);
66  		}
