44  	public static Job createArithmeticJob() throws Exception {
45  		Task task1 = getAddTask();
46  		Task task2 = getMultiplyTask();
47  		Task task3 = getSubtractTask();
48  
49  		Job internal = new NetJob("2tasks");
50  		internal.addExertion(task2);
51  		internal.addExertion(task1);
52  		
53  		Job job = new NetJob("1job1task");
54  		job.addExertion(internal);
55  		job.addExertion(task3);
56  		
57  		// make the result of second task as the first argument of task
58  		// three
59  		task2.getContext().connect("out/value", "arg1/value", task3.getContext());
60  		// make the result of the first task as the second argument of task
61  		// three
62  		task1.getContext().connect("out/value", "arg2/value", task3.getContext());
63  		
64  		return job;
65  	}

67  	public static Job createFlatAritheticJob() throws Exception {
68  		Task task1 = getAddTask();
69  		Task task2 = getMultiplyTask();
70  		Task task3 = getSubtractTask();
71  
72  		Job job = new NetJob("3tasks");
73  		job.addExertion(task1);
74  		job.addExertion(task2);
75  		job.addExertion(task3);
76  		
77  		// make the result of second task as the first argument of task
78  		// three
79  		task2.getContext().connect("out/value", "arg1/value", task3.getContext());
80  		// make the result of the first task as the second argument of task
81  		// three
82  		task1.getContext().connect("out/value", "arg2/value", task3.getContext());
83  		
84  		return job;
85  	}

87  	private static Task getAddTask() throws Exception {
88  		Context context = new PositionalContext("add");
89  		context.putInValue("arg1/value", 20.0);
90  		context.putInValue("arg2/value", 80.0);
91  		// We know that the output is gonna be placed in this path
92  		context.putOutValue("out/value", 0);
93  		NetSignature method = new NetSignature("add", Adder.class);
94  		Task task = new NetTask("add", method);
95  		task.setContext(context);
96  		return task;
97  	}
98  
99  	private static Task getMultiplyTask() throws Exception {
100 		Context context = new PositionalContext("multiply");
101 		context.putInValue("arg1/value", 10.0);
102 		context.putInValue("arg2/value", 50.0);
103 		// We know that the output is gonna be placed in this path
104 		context.putOutValue("out/value", 0);
105 		NetSignature method = new NetSignature("multiply", Multiplier.class);
106 		Task task = new NetTask("multiply", method);
107 		task.setContext(context);
108 		return task;
109 	}
110 
111 	private static Task getSubtractTask() throws Exception {
112 		PositionalContext context = new PositionalContext("subtract");
113 		// We want to stick in the result of multiply in here
114 		context.putInValueAt("arg1/value", 0.0, 1);
115 		// We want to stick in the result of add in here
116 		context.putInValueAt("arg2/value", 0.0, 2);
117 		NetSignature method = new NetSignature("subtract", Subtractor.class);
118 		Task task = new NetTask("subtract",
119 				"processing results from two previous tasks", method);
120 		task.setContext(context);
121 		return task;
122 	}
