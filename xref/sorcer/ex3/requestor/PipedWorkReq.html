18  package sorcer.ex3.requestor;

20  import sorcer.core.context.ServiceContext;
21  import sorcer.core.exertion.NetJob;
22  import sorcer.core.exertion.NetTask;
23  import sorcer.core.requestor.ServiceRequestor;
24  import sorcer.core.signature.NetSignature;
25  import sorcer.ex2.provider.InvalidWork;
26  import sorcer.ex2.provider.Work;
27  import sorcer.service.*;
28  import sorcer.service.Strategy.Access;
29  import sorcer.service.Strategy.Flow;
30  
31  public class PipedWorkReq extends ServiceRequestor {
32  
33      public Exertion getExertion(String... args) throws ExertionException {
34          String requestorName = getProperty("requestor.name");
35  
36          // define requestor data
37          try {
38              Context context1 = new ServiceContext("work1");
39              context1.putValue("requestor/name", requestorName);
40              context1.putValue("requestor/operand/1", 20);
41              context1.putValue("requestor/operand/2", 80);
42              context1.putValue("requestor/work", Works.work1);
43              context1.putOutValue("provider/result", 0);
44  
45              Context context2 = new ServiceContext("work2");
46              context2.putValue("requestor/name", requestorName);
47              context2.putValue("requestor/operand/1", 10);
48              context2.putValue("requestor/operand/2", 50);
49              context1.putValue("requestor/work", Works.work2);
50              context2.putOutValue("provider/result", 0);
51  
52              Context context3 = new ServiceContext("work3");
53              context3.putValue("requestor/name", requestorName);
54              context3.putInValue("requestor/operand/1", 0);
55              context3.putInValue("requestor/operand/2", 0);
56              context1.putValue("requestor/work", Works.work3);
57  
58  
59              // pass the parameters from one dataContext to the next dataContext
60              // piping parameters should be annotated via in, out, or inout paths
61              context1.connect("provider/result", "requestor/operand/1", context3);
62              context2.connect("provider/result", "requestor/operand/2", context3);
63  
64              // define required signatures
65              NetSignature signature1 = new NetSignature("doWork",
66                      sorcer.ex2.provider.Worker.class);
67              NetSignature signature2 = new NetSignature("doWork",
68                      sorcer.ex2.provider.Worker.class);
69              NetSignature signature3 = new NetSignature("doWork",
70                      sorcer.ex2.provider.Worker.class);
71  
72              // define required services
73              Task task1 = new NetTask("work1", signature1, context1);
74              Task task2 = new NetTask("work2", signature2, context2);
75              Task task3 = new NetTask("work3", signature3, context3);
76  
77              // define a job
78              Job job = new NetJob("piped");
79              job.addExertion(task1);
80              job.addExertion(task2);
81              job.addExertion(task3);
82  
83  
84              // define a job control strategy
85              // use the catalog to delegate the tasks
86              job.setAccessType(Access.PUSH);
87              // either parallel or sequential
88              job.setFlowType(Flow.SEQ);
89              // time the job execution
90              job.setExecTimeRequested(true);
91  
92              return job;
93          } catch (Exception e) {
94              throw new ExertionException("Failed to create exertion", e);
95          }
96      }
