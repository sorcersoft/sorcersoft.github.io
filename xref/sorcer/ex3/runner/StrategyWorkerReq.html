18  package sorcer.ex3.runner;
19  
20  import sorcer.core.context.ServiceContext;
21  import sorcer.core.exertion.NetJob;
22  import sorcer.core.exertion.NetTask;
23  import sorcer.core.requestor.ServiceRequestor;
24  import sorcer.core.signature.NetSignature;
25  import sorcer.ex2.provider.InvalidWork;
26  import sorcer.ex2.provider.Work;
27  import sorcer.service.*;
28  import sorcer.service.Strategy.Access;
29  import sorcer.service.Strategy.Flow;
30  import sorcer.util.Sorcer;
31  
32  public class StrategyWorkerReq extends ServiceRequestor {
33  
34      public Exertion getExertion(String... args) throws ExertionException {
35          String requestorName = getProperty("requestor.name");
36          String pn1, pn2, pn3;
37          pn1 = Sorcer.getSuffixedName(getProperty("provider.name.1"));
38          pn2 = Sorcer.getSuffixedName(getProperty("provider.name.2"));
39          pn3 = Sorcer.getSuffixedName(getProperty("provider.name.3"));
40  
41          // define requestor data
42          Work work1 = new Work() {
43              public Context exec(Context cxt) throws InvalidWork, ContextException {
44                  int arg1 = (Integer)cxt.getValue("requestor/operand/1");
45                  int arg2 = (Integer)cxt.getValue("requestor/operand/2");
46                  cxt.putOutValue("provider/result", arg1 + arg2);
47                  return cxt;
48              }
49          };
50  
51          Work work2 = new Work() {
52              public Context exec(Context cxt) throws InvalidWork, ContextException {
53                  int arg1 = (Integer)cxt.getValue("requestor/operand/1");
54                  int arg2 = (Integer)cxt.getValue("requestor/operand/2");
55                  cxt.putOutValue("provider/result", arg1 * arg2);
56                  return cxt;
57              }
58          };
59  
60          Work work3 = new Work() {
61              public Context exec(Context cxt) throws InvalidWork, ContextException {
62                  int arg1 = (Integer)cxt.getValue("requestor/operand/1");
63                  int arg2 = (Integer)cxt.getValue("requestor/operand/2");
64                  cxt.putOutValue("provider/result", arg1 - arg2);
65                  return cxt;
66              }
67          };
68  
69          try {
70              Context context1 = new ServiceContext("work1");
71              context1.putValue("requestor/name", requestorName);
72              context1.putValue("requestor/operand/1", 1);
73              context1.putValue("requestor/operand/2", 1);
74              context1.putValue("requestor/work", work1);
75              context1.putValue("to/provider/name", pn1);
76  
77              Context context2 = new ServiceContext("work2");
78              context2.putValue("requestor/name", requestorName);
79              context2.putValue("requestor/operand/1", 2);
80              context2.putValue("requestor/operand/2", 2);
81              context1.putValue("requestor/work", work2);
82              context2.putValue("to/provider/name", pn2);
83  
84              Context context3 = new ServiceContext("work3");
85              context3.putValue("requestor/name", requestorName);
86              context3.putValue("requestor/operand/1", 3);
87              context3.putValue("requestor/operand/2", 3);
88              context1.putValue("requestor/work", work3);
89              context3.putValue("to/provider/name", pn3);
90  
91              // define required signatures
92              NetSignature signature1 = new NetSignature("doWork",
93                      sorcer.ex2.provider.Worker.class, pn1);
94              NetSignature signature2 = new NetSignature("doWork",
95                      sorcer.ex2.provider.Worker.class, pn2);
96              NetSignature signature3 = new NetSignature("doWork",
97                      sorcer.ex2.provider.Worker.class, pn3);
98  
99              // define required services
100             Task task1 = new NetTask("work1", signature1, context1);
101             task1.setExecTimeRequested(true);
102             Task task2 = new NetTask("work2", signature2, context2);
103             Task task3 = new NetTask("work3", signature3, context3);
104 
105             // define a job
106             Job job = new NetJob("flow");
107             job.setExecTimeRequested(true);
108             job.addExertion(task1);
109             job.addExertion(task2);
110             job.addExertion(task3);
111 
112             // PUSH or PULL provider access
113             boolean isPushAccess = getProperty("provider.access.type", "PUSH").equals("PUSH");
114             if (isPushAccess)
115                 job.setAccessType(Access.PUSH);
116             else
117                 job.setAccessType(Access.PULL);
118 
119             // Exertion control flow PARALLEL or SEQUENTIAL
120             boolean iSequential = getProperty("provider.control.flow", "SEQUENTIAL").equals("SEQUENTIAL");
121             if (iSequential)
122                 job.setFlowType(Flow.SEQ);
123             else
124                 job.setFlowType(Flow.PAR);
125 
126             logger.info("isPushAccess: " + isPushAccess + " iSequential: " + iSequential);
127 
128             return job;
129         } catch (Exception e) {
130             throw new ExertionException("Failed to create exertion", e);
131         }
132     }
133
