106 	@Override
107 	public void execute() throws MojoExecutionException, MojoFailureException {
108 		if (!project.getPackaging().equals("jar")) {
109 			getLog().warn("Plugin misconfigured: running on a project of packaging other than jar");
110 		}
111 
112 		Map<String, String> defaultSystemProps = new HashMap<String, String>();
113 		defaultSystemProps.put(JAVA_UTIL_LOGGING_CONFIG_FILE, new File(sorcerHome, "configs/sorcer.logging").getPath());
114 		defaultSystemProps.put(JAVA_SECURITY_POLICY, new File(testOutputDir, "sorcer.policy").getPath());
115 		defaultSystemProps.put(S_KEY_SORCER_ENV, sorcerEnvFile.getPath());
116 		defaultSystemProps.put(JAVA_RMI_SERVER_USE_CODEBASE_ONLY, "false");
117 
118 		List<ClientRuntimeConfiguration> configurations = buildClientsList();
119 		for (int i = 0; i < configurations.size(); i++) {
120 			ClientRuntimeConfiguration config = configurations.get(i);
121 			JavaProcessBuilder builder = config.preconfigureProcess();
122 
123 			Map<String, String> properties = new HashMap<String, String>();
124 			properties.putAll(defaultSystemProps);
125 			properties.putAll(config.getSystemProperties());
126 			builder.setProperties(properties);
127 
128 			builder.setWorkingDir(basedir);
129 			builder.setDebugger(debug);
130 			builder.setOutput(getLogFile(configurations, i));
131 			if (config.arguments != null) {
132 				builder.setParameters(Arrays.asList(config.arguments));
133 			}
134 
135 			Process2 process = null;
136 			try {
137 				getLog().info("Starting requestor process");
138 				process = builder.startProcess();
139 				if (debug) {
140 					process.waitFor();
141 					getLog().info("Requestor process has finished");
142 				} else {
143 					if (process.waitFor(timeout) == null)
144 						getLog().warn("Requestor process has been destroyed");
145 				}
146 			} catch (InterruptedException e) {
147 				process.destroy();
148 				throw new MojoExecutionException(e.getMessage(), e);
149 			} catch (IllegalStateException x) {
150 				//fail in DestroyMojo, don't log stack trace, there is nothing interesting
151 				getLog().warn(x.getMessage());
152 				TestCycleHelper.getInstance().setFail();
153 			}
154 		}
155 	}

157 	private List<ClientRuntimeConfiguration> buildClientsList() throws MojoExecutionException {
158 		String host = "http://" + Booter.getWebsterHostName() + ":" + TestCycleHelper.getInstance().getWebsterPort();
159 		if (requestors == null || requestors.length == 0) {
160 			ClientRuntimeConfiguration config = new ClientRuntimeConfiguration(mainClass, buildClasspath(requestorClasspath));
161 			updateCodebaseAndRoots(config, host, requestorCodebase);
162 			return Arrays.asList(config);
163 		} else {
164 			List<ClientRuntimeConfiguration> result = new ArrayList<ClientRuntimeConfiguration>(requestors.length);
165 			for (ClientConfiguration requestor : requestors) {
166 				String configMainClass = mainClass != null ? mainClass : requestor.mainClass;
167 				String[] userClasspath = requestor.classpath != null ? requestor.classpath : requestorClasspath;
168 				ClientRuntimeConfiguration config = new ClientRuntimeConfiguration(configMainClass, buildClasspath(userClasspath));
169 
170 				String[] userCodebase = requestor.codebase != null ? requestor.codebase : requestorCodebase;
171 				updateCodebaseAndRoots(config, host, userCodebase);
172 				config.arguments = requestor.arguments;
173 				result.add(config);
174 			}
175 			return result;
176 		}
177 	}

194 	private void updateCodebaseAndRoots(ClientRuntimeConfiguration config, String websterUrl, String[] userCodebase) throws MojoExecutionException {
195 		Collection<Artifact> artifacts = resolveDependencies(KEY_REQUESTOR, userCodebase, scope);
196 		List<String> codebase = new LinkedList<String>();
197 		try {
198 			String repositoryPath = repositorySystemSession.getLocalRepository().getBasedir().getCanonicalPath();
199 			List<String> websterRoots = new LinkedList<String>();
200 			for (Artifact artifact : artifacts) {
201 				File artifactFile = artifact.getFile();
202 				String artifactPath = artifactFile.getCanonicalPath();
203 				if (artifactPath.startsWith(repositoryPath)) {
204 					// add jar from repository
205 					codebase.add(websterUrl + artifactPath.substring(repositoryPath.length()));
206 				} else {
207 					// add jar from target dir
208 					codebase.add(websterUrl + '/' + artifactFile.getName());
209 					websterRoots.add(artifactFile.getParent());
210 				}
211 			}
212 			// if list of root directories for webster is not empty, we must add the default directory
213 			if (!websterRoots.isEmpty()) {
214 				websterRoots.add(repositorySystemSession.getLocalRepository().getBasedir().getPath());
215 			}
216 			config.setWebsterRoots(websterRoots);
217 			config.codebase = codebase.toArray(new String[codebase.size()]);
218 		} catch (IOException e) {
219 			throw new MojoExecutionException(e.getMessage(), e);
220 		}
221 	}

223 	protected Collection<Artifact> resolveDependencies(String propertyKey, String[] userEntries, String scope)
224 			throws MojoExecutionException {
225 		List<String> coordinates = new LinkedList<String>();
226 		if (propertyKey != null && project.getProperties().containsKey(propertyKey)) {
227 			coordinates.add(project.getProperties().getProperty(propertyKey));
228 		}
229 		if (userEntries != null) {
230 			coordinates.addAll(Arrays.asList(userEntries));
231 		}
232 		try {
233 			Set<Artifact> artifacts = new HashSet<Artifact>(coordinates.size());
234 			for (String coords : coordinates) {
235 				artifacts.addAll(resolveDependencies(new DefaultArtifact(coords), scope));
236 			}
237 			return artifacts;
238 		} catch (DependencyResolutionException e) {
239 			throw new MojoExecutionException(e.getMessage(), e);
240 		}
