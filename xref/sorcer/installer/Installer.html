125     public void install() {
126         for (String group : groupDirMap.keySet()) {
127             // Ignore Sigar
128             if (group.equals("org.sorcersoft.sigar")) continue;
129 
130             String dir = groupDirMap.get(group);
131             String version = versionsMap.get(group);
132 
133             if (dir == null || version == null || !new File(Resolver.getRootDir() + "/" + dir).exists()) {
134                 logger.severe("Problem installing jars for groupId: " + group + " directory or version not specified: " + dir + " " + version);
135                 errorCount++;
136                 continue;
137             }
138             File[] jars = new File(Resolver.getRootDir() + "/" + dir).listFiles(new FileFilter() {
139                 @Override
140                 public boolean accept(File pathname) {
141                     if (pathname.getName().endsWith("jar"))
142                         return true;
143                     return false;
144                 }
145             });
146 
147             for (File jar : jars) {
148                 String fileNoExt = jar.getName().replace(".jar", "");
149                 String artifactDir = Resolver.getRepoDir() + "/" + group.replace(".", "/") + "/" + fileNoExt + "/" + version;
150                 try {
151                     FileUtils.forceMkdir(new File(artifactDir));
152                     extractZipFile(jar, "META-INF/maven/" + group + "/" + fileNoExt + "/pom.xml",
153                             artifactDir + "/" + fileNoExt + "-" + version + ".pom");
154                     FileUtils.copyFile(jar, new File(artifactDir, fileNoExt + "-" + version + ".jar"));
155                 } catch (IOException io) {
156                     errorCount++;
157                     logger.severe("Problem installing jar: " + fileNoExt + " to: " + artifactDir);
158                 }
159             }
160         }
161         // install commons
162         File[] jars = new File(Resolver.getRootDir() + "/" + COMMONS_LIBS).listFiles(new FileFilter() {
163             @Override
164             public boolean accept(File pathname) {
165                 if (pathname.getName().endsWith("jar"))
166                     return true;
167                 return false;
168             }
169         });
170 
171         for (File jar : jars) {
172             ArtifactCoordinates ac = getArtifactCoordinatesFromJar(jar);
173             if (ac!=null && ac.getGroupId()!=null && ac.getArtifactId()!=null && ac.getVersion()!=null) {
174                 String artifactDir = Resolver.getRepoDir() + "/" + ac.getGroupId().replace(".", "/") + "/" + ac.getArtifactId() + "/" + ac.getVersion();
175                 try {
176                     FileUtils.forceMkdir(new File(artifactDir));
177                     extractZipFile(jar, "META-INF/maven/" + ac.getGroupId() + "/" + ac.getArtifactId() + "/pom.xml",
178                             artifactDir + "/" + ac.getArtifactId() + "-" + ac.getVersion() + ".pom");
179                     FileUtils.copyFile(jar, new File(artifactDir, ac.getArtifactId() + "-" + ac.getVersion() + ".jar"));
180                     logger.info("Installed jar and pom file: " + artifactDir + File.separator + ac.getArtifactId() + "-" + ac.getVersion() + ".jar");
181                 } catch (IOException io) {
182                     errorCount++;
183                     logger.severe("Problem installing jar: " + ac.getArtifactId() + " to: " + artifactDir + "\n" + io.getMessage());
184                 }
185             }
186         }
187     }

190     public ArtifactCoordinates getArtifactCoordinatesFromJar(File jar) {
191         File tempDir = new File(System.getProperty("java.io.tmpdir"));
192         String fileName = tempDir.getAbsolutePath() + File.separator+ "pom-" + System.currentTimeMillis() + ".properties";
193         extractPomPropFile(jar, fileName);
194         Properties properties = new Properties();
195         ArtifactCoordinates ac = null;
196         try {
197             properties.load(new FileInputStream(fileName));
198             ac = new ArtifactCoordinates(properties.getProperty("groupId"), properties.getProperty("artifactId"), "jar", properties.getProperty("version"), null);
199         } catch (FileNotFoundException e) {
200             logger.fine("Could not find pom.properties in file: " + jar.toString() + "\n" + e.getMessage());
201         } catch (IOException e) {
202             logger.fine("Could not find pom.properties in file: " + jar.toString() + "\n" + e.getMessage());
203         }
204         return ac;
205     }

244     public ArtifactCoordinates getArtifactCoordsFromPom(String fileName) {
245         DocumentBuilderFactory domFactory =
246                 DocumentBuilderFactory.newInstance();
247         domFactory.setNamespaceAware(true);
248         DocumentBuilder builder = null;
249         String groupId = null;
250         String artifactId = null;
251         String version = null;
252         String packaging = null;
253         try {
254             builder = domFactory.newDocumentBuilder();
255             Document doc = builder.parse(fileName);
256             XPath xpath = XPathFactory.newInstance().newXPath();
257             Map<String, String> namespaces = new HashMap<String, String>();
258             namespaces.put("pom", "http://maven.apache.org/POM/4.0.0");
259             xpath.setNamespaceContext(
260                     new NamespaceContextImpl("http://maven.apache.org/POM/4.0.0",
261                             namespaces));
262             XPathExpression expr = xpath.compile("/pom:project/pom:groupId");
263             Object result = expr.evaluate(doc, XPathConstants.NODESET);
264             NodeList nodes = (NodeList) result;
265             if (nodes.getLength()>0)
266                 groupId = nodes.item(0).getTextContent();
267             else {
268                 expr = xpath.compile("/pom:project/pom:parent/pom:groupId");
269                 result = expr.evaluate(doc, XPathConstants.NODESET);
270                 nodes = (NodeList) result;
271                 if (nodes.getLength()>0)
272                     groupId = nodes.item(0).getTextContent();
273             }
274 
275             expr = xpath.compile("/pom:project/pom:artifactId");
276             result = expr.evaluate(doc, XPathConstants.NODESET);
277             nodes = (NodeList) result;
278             artifactId = nodes.item(0).getTextContent();
279 
280             expr = xpath.compile("/pom:project/pom:version");
281             result = expr.evaluate(doc, XPathConstants.NODESET);
282             nodes = (NodeList) result;
283             if (nodes.getLength()>0)
284                 version = nodes.item(0).getTextContent();
285             else {
286                 expr = xpath.compile("/pom:project/pom:parent/pom:version");
287                 result = expr.evaluate(doc, XPathConstants.NODESET);
288                 nodes = (NodeList) result;
289                 if (nodes.getLength()>0)
290                     version = nodes.item(0).getTextContent();
291             }
292             expr = xpath.compile("/pom:project/pom:packaging");
293             result = expr.evaluate(doc, XPathConstants.NODESET);
294             nodes = (NodeList) result;
295             if (nodes.getLength()>0)
296                 packaging = nodes.item(0).getTextContent();
297 
298         } catch (ParserConfigurationException e) {
299             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
300             return null;
301         } catch (XPathExpressionException e) {
302             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
303             return null;
304         } catch (SAXException e) {
305             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
306             return null;
307         } catch (IOException e) {
308             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
309             return null;
310         }
311 
312         return new ArtifactCoordinates(groupId, artifactId, packaging, version, null);
313     }

338     public static void extractZipFile(File zipFileSrc, String relativeFilePath, String targetFilePath) {
339         try {
340             ZipFile zipFile = new ZipFile(zipFileSrc);
341             Enumeration<? extends ZipEntry> e = zipFile.entries();
342 
343             while (e.hasMoreElements()) {
344                 ZipEntry entry = (ZipEntry) e.nextElement();
345                 // if the entry is not directory and matches relative file then extract it
346                 if (!entry.isDirectory() && entry.getName().equals(relativeFilePath)) {
347                     InputStream bis = new BufferedInputStream(
348                             zipFile.getInputStream(entry));
349 
350                     // write the inputStream to a FileOutputStream
351                     OutputStream outputStream =
352                             new FileOutputStream(new File(targetFilePath));
353 
354                     int read = 0;
355                     byte[] bytes = new byte[1024];
356 
357                     while ((read = bis.read(bytes)) != -1) {
358                         outputStream.write(bytes, 0, read);
359                     }
360                     bis.close();
361                     outputStream.close();
362                 } else {
363                     continue;
364                 }
365             }
366         } catch (IOException e) {
367             logger.severe("IOError :" + e);
368             e.printStackTrace();
369         }
370     }
