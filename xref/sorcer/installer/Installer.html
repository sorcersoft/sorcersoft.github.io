78      public boolean isInstallRequired() throws IOException {
79          File home = SorcerEnv.getHomeDir();
80          IOUtils.ensureFile(home, IOUtils.FileCheck.directory);
81          File logDir = new File(home, "logs");
82          IOUtils.ensureFile(logDir, IOUtils.FileCheck.directory, IOUtils.FileCheck.writable);
83          File sorcerApi = new File(libResolver.resolveAbsolute("org.sorcersoft.sorcer:sorcer-api"));
84          return sorcerApi.exists() && !markerFile.exists();

110     protected static void loadProperties(Properties properties, String resourceName) throws IOException {
111         URL resourceVersions = Thread.currentThread().getContextClassLoader().getResource(resourceName);
112         if (resourceVersions == null) {
113             throw new IOException("Could not find versions.properties");
114         }
115         InputStream inputStream = null;
116         try {
117             inputStream = resourceVersions.openStream();
118             properties.load(inputStream);
119         } finally {
120             closeQuietly(inputStream);
121         }
122     }

127         try {

131             closeQuietly(inputStream);

135     /**
136      * Install poms for artifacts of packaging pom
137      * put other poms in the artifactPom map
138      */
139     public void installPoms() throws IOException {
140         File pomDir = new File(SorcerEnv.getHomeDir(), "configs/poms");
141 
142         File[] poms = pomDir.listFiles(new FileFilter() {
143             @Override
144             public boolean accept(File pathname) {
145                 return pathname.getName().endsWith("pom");
146             }
147         });
148         for (File pom : poms) {
149             ArtifactCoordinates ac = getArtifactCoordsFromPom(pom);
150             if (ac == null) {
151                 log.error("Problem while reading pom", pom);
152                 errorCount++;
153                 return;
154             }
155 
156             String artifactId = ac.getArtifactId();
157             if (!ArtifactCoordinates.PCKG_POM.equals(ac.getType())) {
158                 artifactsPoms.put(artifactId, pom);
159             } else {
160                 try {
161                     installPom(pom, ac);
162                 } catch (Exception io) {
163                     errorCount++;
164                     log.error("Problem installing pom file: {}", pom.getAbsolutePath(), io);
165                 }
166             }
167         }
168     }

170     public void installJars() throws IOException {
171         File libDir = new File(SorcerEnv.getLibPath());
172 
173         Collection<String> paths = new LinkedList<String>();
174         paths.addAll(groupDirMap.values());
175         paths.add(COMMONS_LIBS);
176 
177         for (String dir : paths) {
178             installJars(new File(libDir, dir));
179         }
180     }

182     protected void installJars(File fromDir) {
183         log.info("Installing files from {}", fromDir);
184         for (File file : fromDir.listFiles(new FileFilter() {
185             @Override
186             public boolean accept(File file) {
187                 String name = file.getName();
188                 return !file.isDirectory() && (name.endsWith("jar") || name.endsWith("zip"));
189             }
190         })) {
191             try {
192                 install(file);
193             } catch (Exception x) {
194                 ++errorCount;
195                 log.error("Error while installing {}", file, x);
196             }
197         }
198     }

200     protected File getPomFromJar(File jarFile, String artifactId) throws IOException {
201         ZipFile jarZip = null;
202         try {
203             jarZip = new ZipFile(jarFile);
204             ZipEntry pomEntry = get(jarZip, new ZipEntryFilter() {
205                 @Override
206                 public boolean accept(ZipEntry entry) {
207                     String name = entry.getName();
208                     return (!entry.isDirectory() && name.startsWith("META-INF/maven/") && name.endsWith("pom.xml"));
209                 }
210             });
211             if (pomEntry == null)
212                 throw new FileNotFoundException("pom.xml entry not found in " + jarFile);
213             File result = File.createTempFile(artifactId, ".pom");
214             FileUtils.copyInputStreamToFile(jarZip.getInputStream(pomEntry), result);
215             return result;
216         } finally {
217             IOUtils.closeQuietly(jarZip);
218         }
219     }

221     public void install(File file) throws IOException, InstallationException, SettingsBuildingException {
222         File fileNoExt = replaceExt(file, null);
223         String artifactId = fileNoExt.getName();
224 
225         ArtifactCoordinates ac;
226         File pom;
227         boolean pomInTmp = false;
228 
229         if (artifactsPoms.containsKey(artifactId)) {
230             pom = artifactsPoms.get(artifactId);
231         } else {
232             pom = getPomFromJar(file, artifactId);
233             pomInTmp = true;
234         }
235         ac = getArtifactCoordsFromPom(pom);
236 
237         aetherService.install(ac.getGroupId(), ac.getArtifactId(), ac.getVersion(), ac.getClassifier(), pom, file);
238         if (pomInTmp)
239             FileUtils.forceDelete(pom);
240     }

256     protected static ArtifactCoordinates getArtifactCoordsFromPom(InputStream file) {
257         DocumentBuilderFactory domFactory =
258                 DocumentBuilderFactory.newInstance();
259         domFactory.setNamespaceAware(true);
260         DocumentBuilder builder;
261         String groupId = null;
262         String artifactId;
263         String version = null;
264         String packaging;
265         try {
266             builder = domFactory.newDocumentBuilder();
267             Document doc = builder.parse(file);
268             XPath xpath = XPathFactory.newInstance().newXPath();
269             Map<String, String> namespaces = new HashMap<String, String>();
270             namespaces.put("pom", "http://maven.apache.org/POM/4.0.0");
271             xpath.setNamespaceContext(
272                     new NamespaceContextImpl("http://maven.apache.org/POM/4.0.0",
273                             namespaces));
274             XPathExpression expr = xpath.compile("/*[local-name() = 'project']/*[local-name() = 'groupId']");
275             Object result = expr.evaluate(doc, XPathConstants.NODESET);
276             NodeList nodes = (NodeList) result;
277             if (nodes.getLength() > 0)
278                 groupId = nodes.item(0).getTextContent();
279             else {
280                 expr = xpath.compile("/*[local-name() = 'project']/*[local-name() = 'parent']/*[local-name() = 'groupId']");
281                 result = expr.evaluate(doc, XPathConstants.NODESET);
282                 nodes = (NodeList) result;
283                 if (nodes.getLength() > 0)
284                     groupId = nodes.item(0).getTextContent();
285             }
286 
287             expr = xpath.compile("/*[local-name() = 'project']/*[local-name() = 'artifactId']");
288             result = expr.evaluate(doc, XPathConstants.NODESET);
289             nodes = (NodeList) result;
290             if (nodes.getLength() == 0) {
291                 log.error("Problem installing file: {}: could not read artifactId", file);
292                 return null;
293             }
294             artifactId = nodes.item(0).getTextContent();
295 
296             expr = xpath.compile("/*[local-name() = 'project']/*[local-name() = 'version']");
297             result = expr.evaluate(doc, XPathConstants.NODESET);
298             nodes = (NodeList) result;
299             if (nodes.getLength() > 0)
300                 version = nodes.item(0).getTextContent();
301             else {
302                 expr = xpath.compile("/*[local-name() = 'project']/*[local-name() = 'parent']/*[local-name() = 'version']");
303                 result = expr.evaluate(doc, XPathConstants.NODESET);
304                 nodes = (NodeList) result;
305                 if (nodes.getLength() > 0)
306                     version = nodes.item(0).getTextContent();
307             }
308             expr = xpath.compile("/*[local-name() = 'project']/*[local-name() = 'packaging']");
309             result = expr.evaluate(doc, XPathConstants.NODESET);
310             nodes = (NodeList) result;
311             if (nodes.getLength() > 0) {
312                 packaging = nodes.item(0).getTextContent();
313             } else
314                 packaging = ArtifactCoordinates.DEFAULT_PACKAGING;
315         } catch (Exception e) {
316             log.error("Problem reading file: {}", file, e);
317             return null;
318         }
319 
320         return new ArtifactCoordinates(groupId, artifactId, packaging, version, null);
321     }

323     private void createMarker() {
324         if (errorCount == 0) {
325             try {
326                 FileUtils.touch(markerFile);
327             } catch (IOException e) {
328                 ++errorCount;
329                 log.warn("Error while creating marker file {}", markerFile, e);
330             }
331         }
332         log.info("Installer finished with {} errors", errorCount);
333     }

335     public static ZipEntry get(ZipFile zipFile, ZipEntryFilter filter) {
336         Enumeration<? extends ZipEntry> e = zipFile.entries();
337         while (e.hasMoreElements()) {
338             ZipEntry entry = e.nextElement();
339             // if the entry is not directory and matches relative file then extract it
340             if (filter.accept(entry))
341                 return entry;
342         }
343         return null;
344     }

346     /**
347      * replaces filename extension to a provided one. Special case: if ext is null, remove '.' as well
348      */
349     private static File replaceExt(File file, String ext) {
350         String path = file.getPath();
351         int cut = path.lastIndexOf('.');
352         if (ext != null)
353             ++cut;
354         String result = path.substring(0, cut);
355         if (ext != null)
356             result += ext;
357         return new File(result);
358     }
