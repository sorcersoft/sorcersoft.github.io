93      public void install() {
94          for (String group : groupDirMap.keySet()) {
95              // Ignore Sigar
96              if (group.equals("org.sorcersoft.sigar")) continue;
97  
98              String dir = groupDirMap.get(group);
99              String version = versionsMap.get(group);
100 
101             if (dir == null || version == null || !new File(Resolver.getRootDir() + "/" + dir).exists()) {
102                 logger.severe("Problem installing jars for groupId: " + group + " directory or version not specified: " + dir + " " + version);
103                 continue;
104             }
105             File[] jars = new File(Resolver.getRootDir() + "/" + dir).listFiles(new FileFilter() {
106                 @Override
107                 public boolean accept(File pathname) {
108                     if (pathname.getName().endsWith("jar"))
109                         return true;
110                     return false;
111                 }
112             });
113 
114             for (File jar : jars) {
115                 String fileNoExt = jar.getName().replace(".jar", "");
116                 String artifactDir = Resolver.getRepoDir() + "/" + group.replace(".", "/") + "/" + fileNoExt + "/" + version;
117                 try {
118                     FileUtils.forceMkdir(new File(artifactDir));
119                     extractZipFile(jar, "META-INF/maven/" + group + "/" + fileNoExt + "/pom.xml",
120                             artifactDir + "/" + fileNoExt + "-" + version + ".pom");
121                     FileUtils.copyFile(jar, new File(artifactDir, fileNoExt + "-" + version + ".jar"));
122                 } catch (IOException io) {
123                     logger.severe("Problem installing jar: " + fileNoExt + " to: " + artifactDir);
124                 }
125             }
126         }
127     }

130     public void installPoms() {
131 
132         String pomDir = SorcerEnv.getHomeDir() + "/configs/poms/";
133 
134         File[] jars = new File(pomDir).listFiles(new FileFilter() {
135             @Override

137                 if (pathname.getName().endsWith("pom"))
138                     return true;
139                 return false;  //To change body of implemented methods use File | Settings | File Templates.
140             }
141         });
142         String group = "org.sorcersoft.sorcer";
143         for (File jar : jars) {
144                 String fileNoExt = jar.getName().replace("-" + SorcerEnv.getSorcerVersion() + ".pom", "");
145                 String artifactDir = Resolver.getRepoDir() + "/" + group.replace(".", "/") + "/" + fileNoExt + "/" + SorcerEnv.getSorcerVersion();
146                 try {
147                     FileUtils.forceMkdir(new File(artifactDir));
148                     FileUtils.copyFile(jar, new File(artifactDir, jar.getName()));
149                 } catch (IOException io) {
150                     logger.severe("Problem installing jar: " + fileNoExt + " to: " + artifactDir);
151                 }
152         }

171     public static void extractZipFile(File zipFileSrc, String relativeFilePath, String targetFilePath) {
172         try {
173             ZipFile zipFile = new ZipFile(zipFileSrc);
174             Enumeration<? extends ZipEntry> e = zipFile.entries();
175 
176             while (e.hasMoreElements()) {
177                 ZipEntry entry = (ZipEntry) e.nextElement();
178                 // if the entry is not directory and matches relative file then extract it
179                 if (!entry.isDirectory() && entry.getName().equals(relativeFilePath)) {
180                     InputStream bis = new BufferedInputStream(
181                             zipFile.getInputStream(entry));
182 
183                     // write the inputStream to a FileOutputStream
184                     OutputStream outputStream =
185                             new FileOutputStream(new File(targetFilePath));
186 
187                     int read = 0;
188                     byte[] bytes = new byte[1024];
189 
190                     while ((read = bis.read(bytes)) != -1) {
191                         outputStream.write(bytes, 0, read);
192                     }
193                     bis.close();
194                     outputStream.close();
195                 } else {
196                     continue;
197                 }
198             }
199         } catch (IOException e) {
200             logger.severe("IOError :" + e);
201             e.printStackTrace();
202         }
