23  /**
24   * A service <code>Signature</code> is an indirect behavioral feature of
25   * {@link Exertion}s that declares a service that can be performed by instances
26   * of {@link Service}s. It contains a service type and a selector of operation
27   * of that service type (interface). Its implicit parameter and return value is
28   * a service {@link Context}. Thus, the explicit signature of service-oriented
29   * operations is defined by the same {@link Context} type for any exertion
30   * parameter and return value . A signature may include a collection of optional
31   * attributes describing a preferred {@link Service} with a given service type.
32   * Also a signature can carry own implementation when its type is implemented
33   * with the provided codebase.
34   * <p>
35   * In other words, a service signature is a specification of a service that can
36   * be requested dynamically at the boundary of a service provider. Operations
37   * include modifying a service {@link Context} or disclosing information about
38   * the service dataContext.
39   * 
40   * @author Mike Sobolewski
41   */

140 	public static class ReturnPath<T> implements Serializable, Parameter {
141 		static final long serialVersionUID = 6158097800741638834L;
142 		public String path;

149 		public ReturnPath(String path, String... argPaths) {
150 			this.path = path;
151 			if (argPaths != null && argPaths.length > 0) {
152 				this.argPaths = argPaths;
153 				direction = Direction.OUT;
154 			}
155 		}
156 		
157 		public ReturnPath(String path, Direction direction, String...  argPaths) {
158 			this.path = path;
159 			this.argPaths = argPaths;
160 			this.direction = direction;
161 		}
162 		
163 		public ReturnPath(String path, Direction direction, Class<T> returnType, String...  argPaths) {
164 			this.path = path;
165 			this.direction = direction;
166 			this.argPaths = argPaths;
167 			type = returnType;
168 		}
169 		
170 		public String toString() {
171 			String params = "";
172 			if (argPaths != null) 
173 				 params = " argPaths: " + Arrays.toString(argPaths);
174 			return path + (direction != null ? " direction: " + direction : "") 
175 				+ params;
176 		}
