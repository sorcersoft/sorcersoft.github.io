19  package sorcer.service;
20  
21  import sorcer.security.util.SorcerPrincipal;
22  
23  import java.io.Serializable;
24  
25  /**
26   * The <code>Link</code> interface allows path values of service contexts to
27   * be linked to other service contexts using instances implementing this
28   * interface. The interface provides for service dataContext linking. Context links
29   * are references to an offset (path) in a dataContext, which allows the reuse of
30   * existing service dataContext objects.
31   * 
32   * @author Mike Sobolewski
33   */
34  @SuppressWarnings("rawtypes")
35  public interface Link extends Serializable {
36  
37  	/**
38  	 * Returns a name of this link.
39  	 * 
40  	 * @return a link name
41  	 */
42  	public String getName();
43  
44  	/**
45  	 * Assign a link name for its linked dataContext.
46  	 * 
47  	 * @param name
48  	 *            a name for this dataContext link
49  	 */
50  	public void setName(String name);
51  
52  	/**
53  	 * Returns the path offset into this linked dataContext.
54  	 * 
55  	 * @return an offset of this link
56  	 */
57  	public String getOffset();
58  
59  	/**
60  	 * Sets the offset in this linked dataContext. If the offset itself is obtained
61  	 * by traversing a link (meaning there is a redundant link), the offset is
62  	 * recalculated and the link object is reset to point to the owning dataContext
63  	 * (removing the redundancy).
64  	 * 
65  	 * @param offset
66  	 * @throws ContextException
67  	 */
68  	public void setOffset(String offset) throws ContextException;
69  
70  	/**
71  	 * Assignes a link service dataContext associated with this link.
72  	 * 
73  	 * @param ctxt
74  	 */
75  	public void setContext(Context ctxt);
76  
77  	/**
78  	 * Returns the context associated with this link.
79  	 */
80  	public Context getContext() throws ContextException;
81  	
82  	/**
83  	 * Returns true if the linked dataContext associated with this link is enclosed
84  	 * into the link, otherwise false.
85  	 * 
86  	 * @return true if the linked dataContext is enclosed in this link
87  	 */
88  	public boolean isLocal();
89  
90  	/**
91  	 * Returns a boolean indicating whether the linked contexed is retrived from
92  	 * a persitent dataContext storage into this link - <code>true</code>
93  	 * indicating if the linked contexed is already fetched; <code>false</code>
94  	 * otherwise.
95  	 * 
96  	 * @return a boolean indicating if the linked contexed is already fetched
97  	 */
98  	public boolean isFetched();
99  
100 	/**
101 	 * Assignes a boolean indicating whether the linked contexed is retrived
102 	 * from a persitent dataContext storage into this link.
103 	 * 
104 	 * @param state
105 	 *            <code>true</code> indicating if the linked contexed is
106 	 *            already fetched; <code>false</code> otherwise
107 	 */
108 	public void isFetched(boolean state);
109 
110     Context getContext(SorcerPrincipal principal)
111             throws ContextException;
112 }
