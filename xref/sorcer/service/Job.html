45  /**
46   * A job is a composite service-oriented message comprised of {@link sorcer.service.Exertion}
47   * instances with its own service {@link sorcer.service.Context} and a collection of service
48   * {@link sorcer.service.Signature}s. The job's signature is usually referring to a
49   * {@link Jobber} and the job's context describes the composition
50   * of component exertions as defined by the Interpreter programming pattern.
51   * 
52   * @see Exertion
53   * @see Task
54   * 
55   * @author Mike Sobolewski
56   */

185 	public Exertion getMasterExertion() {
186 		Uuid contextId = null;
187 		try {
188 			//contextName = (String) controlContext.getValue(ControlContext.MASTER_EXERTION);
189             contextId = (Uuid) controlContext.getValue(ControlContext.MASTER_EXERTION);
190 		} catch (ContextException ex) {
191 			ex.printStackTrace();
192 		}
193 		if (contextId == null
194 				&& controlContext.getFlowType().equals(ControlContext.SEQUENTIAL)) {
195 			return (size() > 0) ? exertionAt(size() - 1) : null;
196 		} else {
197 			Exertion master = null;
198 			for (int i = 0; i < size(); i++) {
199 				if (exertionAt(i).getId().equals(contextId)) {
200 					master = exertionAt(i);
201 					break;
202 				}
203 			}
204 			return master;
205 		}
206 	}

479 	public Object getJobValue(String path) throws ContextException {
480         String[] attributes = StringUtils.tokenizerSplit(path, CPS);
481 		// remove the leading attribute of the current exertion
482 		if (attributes[0].equals(getName())) {
483 			String[] attributes1 = new String[attributes.length - 1];
484 			System.arraycopy(attributes, 1, attributes1, 0,
485 					attributes.length - 1);
486 			attributes = attributes1;
487 		}
488 		String last = attributes[0];
489 		Exertion exti = this;
490 		for (String attribute : attributes) {
491 			if (((ServiceExertion) exti).hasChild(attribute)) {
492 				exti = ((Job) exti).getChild(attribute);
493 				if (exti instanceof Task) {
494 					last = attribute;
495 					break;
496 				}
497 			} else {
498 				break;
499 			}
500 		}
501 		int index = path.indexOf(last);
502 		String contextPath = path.substring(index + last.length() + 1);
503 
504 		return exti.getContext().getValue(contextPath);
505 	}
