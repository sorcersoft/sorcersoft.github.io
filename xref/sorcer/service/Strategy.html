25  public interface Strategy extends Arg {
26  
27  	/**
28  	 * There are three flow types of {@link Flow}s that can be associated
29  	 * with exertions control dataContext. Flow type specify the flow of control (
30  	 * <code>SEQ</code>, <code>PAR</code>, or
31  	 * <code>CON</code>) for all component exertions at the same level.
32  	 */
33  	public enum Flow {
34  		SEQ, PAR, CON
35  	}
36  	
37  	/**
38  	 * When the access type of a control dataContext is set to <code>PULL</code>
39  	 * then the associated exertion is passed onto a {@link Spacer}, otherwise
40  	 * (access type is PUSH) the exertion is passed directly on to the provider
41  	 * specified by the <code>PROCESS</code>signature	 * 
42  	 */
43  	public enum Access {
44  		PUSH, PULL, SWIF, QOS_PUSH, QOS_PULL, CATALOG, DIRECT
45  	}
46  		
47  	/**
48  	 * Indicates if on-demand provisioning is used for not existing currently
49  	 * service but required in the overlay network of service providers..
50  	 */
51  	public enum Provision {
52  		YES, TRUE, NO, FALSE
53  	}
54  
55  	/**
56  	 * The <code>CONCURRENT</code> flow is controlled by mutex state WAIT,
57  	 * NOTIFY, and NOTIFY_ALL of the control dataContext. Only one exertion can own
58  	 * the mutex at a time. If a second provider tries to acquire the exertion,
59  	 * it will block (be suspended) until the owning provider releases the
60  	 * mutex. An exertion access to which is guarded by a mutual-exclusion is
61  	 * called monitor.
62  	 */
63  	public enum Monitor {
64  		YES, TRUE, NO, FALSE, WAIT, NOTIFY, NOTIFY_ALL
65  	}
66  	
67  	/**
68  	 * Indicates whether the service requestor waits for results or not.
69  	 */
70  	public enum Wait {
71  		YES, TRUE, NO, FALSE
72  	}
73
