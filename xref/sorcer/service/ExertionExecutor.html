32  public class ExertionExecutor {
33      public static Object exec(Exertion exertion, Arg... entries)
34              throws ExertionException, ContextException {
35          Exertion xrt;
36          try {
37              if (exertion.getClass() == Task.class) {
38                  if (((Task) exertion).getInnerTask() != null)
39                      xrt = exert(((Task) exertion).getInnerTask(), null, entries);
40                  else
41                      xrt = exertOpenTask(exertion, entries);
42              } else {
43                  xrt = exert(exertion, null, entries);
44              }
45          } catch (Exception e) {
46              e.printStackTrace();
47              throw new ExertionException(e);
48          }
49          ReturnPath returnPath = xrt.getDataContext().getReturnPath();
50          if (returnPath != null) {
51              if (xrt instanceof Task) {
52                  return xrt.getDataContext().getValue(returnPath.path);
53              } else if (xrt instanceof Job) {
54                  return ((Job) xrt).getValue(returnPath.path);
55              }
56          } else {
57              if (xrt instanceof Task) {
58                  return xrt.getDataContext();
59              } else if (xrt instanceof Job) {
60                  return ((Job) xrt).getJobContext();
61              }
62          }
63          throw new ExertionException("No return path in the exertion: "
64                  + xrt.getName());
65      }

67      public static <T extends Exertion> T exert(T input,
68                                                 Transaction transaction, Arg... entries)
69              throws ExertionException {
70          try {
71              ServiceExerter esh = new ServiceExerter(input);
72              Exertion result = null;
73              try {
74                  result = esh.exert(transaction, null, entries);
75              } catch (Exception e) {
76                  e.printStackTrace();
77              }
78              return (T) result;
79          } catch (Exception e) {
80              throw new ExertionException(e);
81          }
82      }

84      public static Exertion exertOpenTask(Exertion exertion,
85                                           Arg... entries) throws ContextException, ExertionException {
86          Exertion closedTask = null;
87          List<Arg> params = Arrays.asList(entries);
88          List<Object> items = new ArrayList<Object>();
89          for (Arg param : params) {
90              if (param instanceof ControlContext
91                      && ((ControlContext) param).getSignatures().size() > 0) {
92                  List<Signature> sigs = ((ControlContext) param).getSignatures();
93                  ControlContext cc = (ControlContext) param;
94                  cc.setSignatures(null);
95                  Context tc = exertion.getDataContext();
96                  items.add(tc);
97                  items.add(cc);
98                  items.addAll(sigs);
99                  closedTask = TaskFactory.task(exertion.getName(), items.toArray());
100             }
101         }
102         try {
103             closedTask = closedTask.exert(entries);
104         } catch (Exception e) {
105             e.printStackTrace();
106             throw new ExertionException(e);
107         }
108         return closedTask;
109     }
