36  /**
37   * A utility class that provides access to SORCER services and some
38   * infrastructure services. It extends the <code>ServiceAccessor</code>
39   * functionality.
40   * 
41   * The <code>getService</code> methods directly use ServiceAccessor calls while
42   * the <code>getProvider</code> methods use a SORCER Cataloger's cached services
43   * with round robin load balancing.
44   * 
45   * The individual SORCER services should be accessed using this utility since it
46   * uses local cached proxies for frequently used SORCER infrastructure services,
47   * for example: Cataloger, JavaSpace, Jobber. ProviderAccessor normally uses
48   * Cataloger if available, otherwise it uses Jini lookup services as implemented
49   * by <code>ServiceAccessor</code>.
50   * 
51   * @see ServiceAccessor
52   */

167 	/**
168 	 * Returns a SORCER service provider with the specified name and service
169 	 * type, using a Cataloger if available, otherwise using Jini lookup
170 	 * services.
171 	 * 
172 	 * @param providerName
173 	 *            the name of service provider
174 	 * @param serviceType
175 	 *            a provider service type (interface)
176 	 * @return a SORCER provider service
177 	 */
178 	public static Provider getProvider(String providerName, Class serviceType) {
179 		Provider servicer = null;
180 		if (providerName != null && providerName.equals(ANY))
181 			providerName = null;
182 		try {
183 			//servicer = (Service)ProviderLookup.getService(providerName, serviceType);
184 			cataloger = getCataloger();
185 			if (cataloger != null) {
186 				int tryNo = 0;
187 				while (tryNo < LUS_REAPEAT) {
188 					servicer = cataloger.lookup(providerName, serviceType);
189 					//servicer = (Service)cataloger.lookupItem(providerName, serviceType).service;
190 					if (servicer != null)
191 						break;
192 						
193 					Thread.sleep(WAIT_FOR);
194 					tryNo++;
195 				}
196 			}
197 			// fall back on Jini LUS
198 			if (servicer == null) {
199 				servicer = (Provider) getService(providerName, serviceType);
200 			}
201 		} catch (Throwable ex) {
202 			logger.throwing(ProviderAccessor.class.getName(), "getProvider", ex);
203 			ex.printStackTrace();
204 		}
205 		return servicer;
206 	}

581 	/**
582 	 * Returns a SORCER service using a cached Cataloger instance by this
583 	 * ProviderAccessor. However if it not possible uses a ServiceAccessor to
584 	 * get a requested service form Jini lookup services directly. This approach
585 	 * allows for SORCER requestors and providers to avoid continuous usage of
586 	 * lookup discovery for each needed service that is delegated to a SORCER
587 	 * Cataloger service.
588 	 * 
589 	 * @param providerName
590 	 *            - a name of requested service
591 	 * @param primaryInterface
592 	 *            - service type of requested provider
593 	 * @return a requested service or null if a Cataloger is not available
594 	 * @throws AccessorException
595 	 */
596 	public static Provider lookup(String providerName, Class primaryInterface)
597 			throws AccessorException {
598 		try {
599 			// check if the cataloger is alive then return a reqested service
600 			// provider
601 			if (ProviderAccessor.isAlive((Provider) cataloger))
602 				return cataloger.lookup(providerName, primaryInterface);
603 			else {
604 				// try to get a new cataloger and lookup again
605 				cataloger = (Cataloger) getService(
606 						Sorcer.getProperty(S_CATALOGER_NAME), Cataloger.class);
607 				if (cataloger != null) {
608 					logger.info("Got service provider from Cataloger");
609 					return cataloger.lookup(providerName, primaryInterface);
610 				} else {
611 					// just get a provider without a Cataloger, use directly
612 					// LUSs
613 					logger.severe("No SORCER cataloger available");
614 					return (Provider) getService(providerName, primaryInterface);
615 				}
616 			}
617 		} catch (RemoteException ex) {
618 			logger.throwing(ProviderAccessor.class.getName(), "lookup", ex);
619 			return null;
620 		}
621 	}
