134 		for (amt--; amt >= 0; amt--)

138 	private void do_sblocks(int[] in, int[] out) {
139 		int i, j, val;
140 		for (i = 0; i < 8; i++) {
141 			val = sblocks[i][in[i * 6] << 1 | in[i * 6 + 5]][in[i * 6 + 1] << 3
142 					| in[i * 6 + 2] << 2 | in[i * 6 + 3] << 1 | in[i * 6 + 4]];
143 			out[i * 4] = val >> 3 & 1;
144 			out[i * 4 + 1] = val >> 2 & 1;
145 			out[i * 4 + 2] = val >> 1 & 1;
146 			out[i * 4 + 3] = val & 1;
147 		}
148 
149 		// Debug information
150 		// System.out.println("Before SB: ");
151 		// pr_bits(in,48);
152 		// System.out.println("After SB : ");

219 		int rots, i, tmp0l, tmp1l, tmp0r, tmp1r;
220 
221 		rots = keyrots[iter];
222 
223 		tmp0l = ikey[0];
224 		tmp1l = ikey[1];
225 		tmp0r = ikey[28];
226 		tmp1r = ikey[29];
227 		for (i = 0; i < 28 - rots; i++) {
228 			ikey[i] = ikey[i + rots];
229 			ikey[28 + i] = ikey[28 + i + rots];
230 		}
231 		if (rots == 2) {
232 			ikey[26] = tmp0l;
233 			ikey[27] = tmp1l;
234 			ikey[54] = tmp0r;
235 			ikey[55] = tmp1r;
236 		} else {
237 			ikey[27] = tmp0l;
238 			ikey[55] = tmp0r;
239 		}

284 		int bits[] = new int[64];
285 		int outl[] = new int[32];
286 		int outr[] = new int[32];
287 		int done[] = new int[66]; // In c-code array-size was 64 !?!
288 		int i, j, dess, iters;
289 		int ikey[] = new int[56];
290 		int saltmask[] = new int[12];
291 		char[] answer = new char[14];
292 
293 		for (i = 0; i < 64; i++) {
294 			bits[i] = 0;
295 		}
296 		load_key(ikey, password);
297 		load_salt(saltmask, salt);
298 
299 		// Debug information
300 		// System.out.print("Key : ");
301 		// pr_bits(ikey,56);
302 		// System.out.print("Salt: ");
303 		// pr_bits(saltmask,12);
304 
305 		for (dess = 0; dess < 25; dess++) {
306 			for (iters = 0; iters < 16; iters += 2) { // Debug information
307 				// System.out.print("Mycrypt "+iters+" : ");
308 				// pr_bits(bits,64);
309 				do_f(bits, 32, outl, iters, ikey, saltmask);
310 				xor(outl, bits, 0, 32);
311 				do_f(outl, 0, outr, iters + 1, ikey, saltmask);
312 				xor(outr, bits, 32, 32);
313 
314 				if (iters != 14) {
315 					for (i = 0; i < 32; i++) {
316 						bits[i] = outl[i];
317 						bits[i + 32] = outr[i];
318 					}
319 				} else {
320 					for (i = 0; i < 32; i++) {
321 						bits[i] = outr[i];
322 						bits[i + 32] = outl[i];
323 					}
324 				}
325 				// Debug information
326 				// if(iters==14)
327 				// { System.out.print("End :    ");
328 				// pr_bits(bits,64);
329 				// }
330 
331 			}
332 		}
333 		permute(IPinv_p, 64, bits, done);
334 
335 		// Debug information
336 		// System.out.print("OUT BITS: ");
337 		// pr_bits(done,64);
338 
339 		answer[0] = salt[0];
340 		answer[1] = salt[1];
341 		for (i = 0; i < 11; i++) {
342 			char c = 0;
343 			for (j = 0; j < 6; j++) {
344 				c |= done[6 * i + j] << (5 - j);
345 			}
346 			answer[i + 2] = bin_to_ascii(c);
347 		}
348 		// anwser[i+2]='\0';
349 		return answer;
350 	};
