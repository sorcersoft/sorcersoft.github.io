38      public static boolean isSosURL(Object url) {
39          if (url instanceof URL && ((URL)url).getProtocol().equals("sos"))
40              return true;
41          else
42              return false;
43      }
44  
45      public static boolean isSosIdURL(Object url) {
46          if (isSosURL(url) && ((URL) url).getRef() != null)
47              return true;
48          else
49              return false;
50      }
51  
52      public static Uuid getUuid(URL url) {
53          String urlRef = url.getRef();
54          int index = urlRef.indexOf('=');
55          // storeType = SorcerDatabaseViews.getStoreType(reference.substring(0,
56          // index));
57          return UuidFactory.create(urlRef.substring(index + 1));
58      }
59  
60  	public static Store getStoreType(URL url) {
61  		String urlRef = url.getRef();
62  		int index = urlRef.indexOf('=');
63  		return Store.getStoreType(urlRef.substring(0, index));
64  	}
65  
66      public static String getProviderName(URL url) {
67          if (url == null)
68              return null;
69          else
70              return url.getPath().substring(1);
71      }
72  
73      public static String getServiceType(URL url) {
74          return url.getHost();
75      }
76  
77      /**
78       * Returns a context to be used with
79       * {@link StorageManagement#contextStore(Context)}
80       *
81       * @param uuid
82       *            {@link Uuid}
83       * @param type
84       *            one of: exertion, context, var, table, varModel, object
85       * @return retrieval {@link Context}
86       * @throws ContextException
87       */
88      static public Context getRetrieveContext(Uuid uuid, Store type)
89              throws ContextException {
90  		ServiceContext cxt = new ServiceContext("retrieve dataContext");
91          cxt.putInValue(StorageManagement.object_type, type);
92          cxt.putInValue(StorageManagement.object_uuid, uuid);
93          cxt.setReturnPath(StorageManagement.object_retrieved);
94          return cxt;
95      }
96  
97      static public Context getUpdateContext(Object object, URL url)
98              throws ContextException {
99          return getUpdateContext(object, getUuid(url));
100     }
101 
102     /**
103 	 * Returns a dataContext to be used with
104      * {@link StorageManagement#contextUpdate(Context)}
105      *
106      * @param object
107      *            to be updated
108      * @param uuid
109      *            {@link Uuid} og the updated object
110      * @return update {@link Context}
111      * @throws ContextException
112      */
113     static public Context getUpdateContext(Object object, Uuid uuid)
114             throws ContextException {
115         ServiceContext cxt = new ServiceContext("update context");
116         cxt.putInValue(StorageManagement.object_uuid, uuid);
117         cxt.putInValue(StorageManagement.object_updated, object);
118         cxt.setReturnPath(StorageManagement.object_url);
119         return cxt;
120     }
121 
122     static public Context getListContext(Store storeType)
123             throws ContextException {
124         ServiceContext cxt = new ServiceContext("storage list context");
125         cxt.putInValue(StorageManagement.store_type, storeType);
126         cxt.setReturnPath(StorageManagement.store_content_list);
127         return cxt;
128     }
