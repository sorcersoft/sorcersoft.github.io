48  		URL url = null;

75  		currentLine = 0;

139 		BOF = false;

142

144 			EOF = true;

154 	private void readRowSet() {
155 		int setCount = 0;
156 		String s = null;
157 
158 		if (notLimitedBuffer) {
159 			try {
160 				while (!EOFReached) {
161 					if ((s = inStream.readLine()) != null) {
162 						lineBuffer.addElement(s);
163 						lineCount++;
164 						setCount++;
165 					} else
166 						EOFReached = true;
167 				}
168 			} catch (IOException e) {
169 				System.out.println("File Read Error");
170 			}
171 		} else {
172 			try {
173 				while ((setCount < rowSetSize) & (!EOFReached)) {
174 					if ((s = inStream.readLine()) != null) {
175 						lineBuffer.addElement(s);
176 						lineCount++;
177 						setCount++;
178 					} else
179 						EOFReached = true;
180 				}
181 			} catch (IOException e) {
182 				System.out.println("File Read Error");
183 			}
184 		}

219 	public String getRow(int rowNumber) {
220 		BOF = false;
221 		EOF = false;
222 
223 		while ((rowNumber > lineCount) & (!EOFReached)) {
224 			readRowSet();
225 		}
226 		if ((rowNumber > lineCount) & (EOFReached)) {
227 			EOF = true;
228 			return null;
229 		} else {
230 			return (String) lineBuffer.elementAt(rowNumber - 1);
231 		}
232 	}

234 	public String[] getRowArray(int rowNumber) {
235 
236 		String row = getRow(rowNumber);
237 		if (row == null)
238 			return null;
239 		else {
240 			StringTokenizer tokenString = new StringTokenizer(row, delimiter);
241 			int numberOfTokens = tokenString.countTokens();
242 			String[] tokens = new String[numberOfTokens];
243 			for (int i = 0; i < numberOfTokens; i++)
244 				tokens[i] = tokenString.nextToken();
245 			return tokens;
246 		}

353 	public Hashtable getSQLQueries() {
354 		Hashtable sqlList = new Hashtable();
355 		String lineString, keyString = null;
356 		StringBuffer sqlString = new StringBuffer(512);
357 		int sep;
358 
359 		for (int i = 0; i < lineBuffer.size(); i++) {
360 			lineString = (String) lineBuffer.elementAt(i);
361 			if (lineString.indexOf(":") == -1) {
362 				lineString.trim();
363 				sqlString.append(lineString).append(" ");
364 			} else {
365 				if (keyString != null) {
366 					sqlList.put(keyString, sqlString.toString());
367 					sqlString.setLength(0);
368 				}
369 				sep = lineString.indexOf(":");
370 				keyString = lineString.substring(0, sep);
371 				sqlString.append(lineString.substring(sep + 1)).append(" ");
372 			}
373 		}
374 		sqlList.put(keyString, sqlString.toString());
375 		return sqlList;
