18  package sorcer.util;

20  import java.io.File;
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.OutputStream;
24  
25  /**
26   * @author Rafał Krupiński
27   */
28  public class IOUtil {
29  	/**
30  	 * Deletes a direcory and all its files.
31  	 *
32  	 * @param dir
33  	 *            to be deleted
34  	 * @return true if the directory is deleted
35  	 * @throws Exception
36  	 */
37  	public static boolean deleteDir(File dir) {
38  		if (dir.isDirectory()) {
39  			String[] children = dir.list();
40  			for (int i = 0; i < children.length; i++) {
41  				boolean success = deleteDir(new File(dir, children[i]));
42  				if (!success) {
43  					return false;
44  				}
45  			}
46  		}
47  
48  		// The directory is now empty so delete it
49  		return dir.delete();
50  	}
51  
52  	/**
53  	 * Copy in to out stream. Do not allow other threads to read from the input
54  	 * or write to the output while copying is taking place.
55  	 */
56  	synchronized public static void copy(InputStream in, OutputStream out)
57  			throws IOException {
58  		byte[] buffer = new byte[256];
59  		int bytesRead;
60  		while (true) {
61  			bytesRead = in.read(buffer);
62  			if (bytesRead == -1)
63  				break;
64  			out.write(buffer, 0, bytesRead);
65  		}
66  	}
67  }
