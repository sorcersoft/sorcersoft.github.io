28  	 * Execute specified command and return its results. Waits for the command
29  	 * to complete and returns its completion status and data written to
30  	 * standard output and error streams. The process' standard input is set to
31  	 * EOF. Example:
32  	 * 
33  	 * <pre>
34  	 * System.out.println(ExecUtils.execCommand(&quot;/bin/ls&quot;).getOut());
35  	 * </pre>
36  	 * 
37  	 * @param cmd
38  	 *            the command to execute
39  	 * @return the results of the command execution
40  	 * @throws IOException
41  	 *             if an I/O error occurs
42  	 * @throws InterruptedException
43  	 *             if thread is interrupted before command completes
44  	 */
45  	public static CmdResult execCommand(String cmd) throws IOException,
46  			InterruptedException {
47  
48  		String[] cmdarray = null;
49  		if (GenericUtil.isWindows()) {
50  			cmdarray = new String[3];
51  			cmdarray[0] = "cmd";
52  			cmdarray[1] = "/C";
53  			cmdarray[2] = cmd;
54  			return execCommand(Runtime.getRuntime().exec(cmdarray));
55  		} else {
56  			return execCommand(Runtime.getRuntime().exec(cmd));
57  		}		
58  	}

61  	 * Added by E. D. Thompson AFRL/RZTT 20100827 Execute specified command and
62  	 * it arguments and return its results. Waits for the command to complete
63  	 * and returns its completion status and data written to standard output and
64  	 * error streams. The process' standard input is set to EOF. Example:
65  	 * 
66  	 * <pre>
67  	 * String[] cmd = { &quot;/bin/ls&quot;, &quot;-lah&quot; };
68  	 * System.out.println(ExecUtils.execCommand(cmd).getOut());
69  	 * </pre>
70  	 * 
71  	 * @param cmdarray
72  	 *            the command and arguments to execute
73  	 * @return the results of the command execution
74  	 * @throws IOException
75  	 *             if an I/O error occurs
76  	 * @throws InterruptedException
77  	 *             if thread is interrupted before command completes
78  	 */
79  	public static CmdResult execCommand(String[] cmdarray) throws IOException,
80  			InterruptedException {
81  		
82  		if (GenericUtil.isWindows()) {
83  			String[] ncmdarray = new String[cmdarray.length + 2];
84  			ncmdarray[0] = "cmd";
85  			ncmdarray[1] = "/C";
86  			int ctr = 2;
87  			for (int i = 0; i < cmdarray.length; i++) {
88  				ncmdarray[ctr] = cmdarray[i];
89  				ctr++;
90  			}
91  			cmdarray = ncmdarray;
92  		}
93  		return execCommand(Runtime.getRuntime().exec(cmdarray));
94  	}

148 	/**
149 	 * Attach to the specified process, feed specified standard input, and
150 	 * return process' results. Waits for the process to complete and returns
151 	 * completion status and data written to standard output and error streams.
152 	 * 
153 	 * @see #execCommand(Process)
154 	 * 
155 	 * @param process
156 	 *            the process to attach to
157 	 * @param stdin
158 	 *            the data to redirect to process' standard input
159 	 * @return the results of the process
160 	 * @throws IOException
161 	 *             if an I/O error occurs
162 	 * @throws InterruptedException
163 	 *             if thread is interrupted before process ends
164 	 */
165 	public static CmdResult execCommand(final Process process,
166 			final InputStream stdin, boolean outLogged) throws IOException,
167 			InterruptedException {
168 		// concurrency to avoid stdio deadlocks
169 		Redir stdout = null;
170 		String out = null;
171 		if (!outLogged) {
172 			stdout = new Redir(process.getInputStream());
173 			new Thread(stdout, "[" + Thread.currentThread().getName() + "] STDOUT-" + process.toString()).start();
174 		}
175 		Redir stderr = new Redir(process.getErrorStream());
176 		new Thread(stderr, "[" + Thread.currentThread().getName() + "] STDERR-" + process.toString()).start();
177 		// redirect input in the current thread
178 		if (stdin != null) {
179 			OutputStream pout = process.getOutputStream();
180 			new RedirectingInputStream(stdin, true, true).redirectAll(pout);
181 		}
182 		process.waitFor();
183 		int exitValue = process.exitValue();
184 
185 		if (stdout != null) {
186 			stdout.throwIfHadException();
187 			out = new String(stdout.getResult());
188 		}
189 		stderr.throwIfHadException();
190 		String err = new String(stderr.getResult());
191 
192 		return new CmdResult(exitValue, out, err);
193 	}

213 			final InputStream stdin) throws IOException, InterruptedException {
214 		// concurrency to avoid stdio deadlocks
215 		Redir stdout = new Redir(process.getInputStream());
216 		Redir stderr = new Redir(process.getErrorStream());
217 		new Thread(stdout, "[" + Thread.currentThread().getName() + "] STDOUT-" + process.toString()).start();
218 		new Thread(stderr, "[" + Thread.currentThread().getName() + "] STDERR-" + process.toString()).start();
219 		// redirect input in the current thread
220 		if (stdin != null) {
221 			OutputStream pout = process.getOutputStream();
222 			new RedirectingInputStream(stdin, true, true).redirectAll(pout);
223 		}
224 
225 		stdout.throwIfHadException();
226 		stderr.throwIfHadException();
227 		String out = new String(stdout.getResult());
228 		String err = new String(stderr.getResult());
229 
230 		return new CmdResult(-1, out, err);
231 	}

333 	private static class Pipe implements Runnable {
334 		final InputStream src;
335 		final OutputStream sink;
336 		final boolean autoFlush;
337 		final boolean autoClose;
338 		final BrokenPipeHandler brokenPipeHandler;
339 
340 		public Pipe(InputStream src, OutputStream sink,
341 				BrokenPipeHandler brokenPipeHandler) {
342 			this(src, sink, brokenPipeHandler, true, false);
343 		}
344 
345 		public Pipe(InputStream src, OutputStream sink,
346 				BrokenPipeHandler brokenPipeHandler, boolean autoFlush,
347 				boolean autoClose) {
348 			this.src = src;
349 			this.sink = sink;
350 			this.brokenPipeHandler = brokenPipeHandler;
351 			this.autoFlush = autoFlush;
352 			this.autoClose = autoClose;
353 		}
354 
355 		public void run() {
356 			RedirectingInputStream sd = new RedirectingInputStream(src,
357 					autoFlush, autoClose);
358 			try {
359 				sd.redirectAll(sink);
360 			} catch (IOException e) {
361 				if (brokenPipeHandler != null) {
362 					brokenPipeHandler.brokenPipe(e, src, sink);
363 				}
364 			}
365 		}
366 	}

368 	private static class Redir implements Runnable {
369 		final Pipe pipe;
370 		IOException ex;
371 
372 		Redir(InputStream is) {
373 			ByteArrayOutputStream bos = new ByteArrayOutputStream();
374 			BrokenPipeHandler bph = new BrokenPipeHandler() {
375 				public void brokenPipe(IOException ex, InputStream src,
376 						OutputStream sink) {
377 					setException(ex);
378 				}
379 			};
380 			this.pipe = new Pipe(is, bos, bph, true, true);
381 		}
382 
383 		public void run() {
384 			pipe.run();
385 		}
386 
387 		synchronized void setException(IOException e) {
388 			this.ex = e;
389 		}
390 
391 		synchronized void throwIfHadException() throws IOException {
392 			if (ex != null)
393 				throw ex;
394 		}
395 
396 		public byte[] getResult() {
397 			return ((ByteArrayOutputStream) pipe.sink).toByteArray();
398 		}

401 	private static class ProcessHandler {
402 		final Process process;
403 		final Thread tstdin;
404 		final Thread tstdout;
405 		final Thread tstderr;
406 		final Thread texitHandler;
407 
408 		ProcessHandler(final Process process, InputStream stdin,
409 				boolean inAutoFlush, boolean inAutoClose,
410 				BrokenPipeHandler inBrokenHandler, OutputStream stdout,
411 				boolean outAutoFlush, boolean outAutoClose,
412 				BrokenPipeHandler outBrokenHandler, OutputStream stderr,
413 				boolean errAutoFlush, boolean errAutoClose,
414 				BrokenPipeHandler errBrokenHandler,
415 				final ProcessExitHandler exitHandler) throws IOException {
416 			this.process = process;
417 			this.tstdin = createPipe(stdin, process.getOutputStream(),
418 					inBrokenHandler, inAutoFlush, inAutoClose);
419 			this.tstdout = createPipe(process.getInputStream(), stdout,
420 					outBrokenHandler, outAutoFlush, outAutoClose);
421 			this.tstderr = createPipe(process.getErrorStream(), stderr,
422 					errBrokenHandler, errAutoFlush, errAutoClose);
423 			if (exitHandler != null) {
424 				this.texitHandler = new Thread(new ExitHandler(process,
425 						exitHandler));
426 			} else {
427 				texitHandler = null;
428 			}
429 		}
430 
431 		void start() {
432 			if (tstdin != null)
433 				tstdin.start();
434 			if (tstdout != null)
435 				tstdout.start();
436 			if (tstderr != null)
437 				tstderr.start();
438 			if (texitHandler != null)
439 				texitHandler.start();
440 		}

442 		private static class ExitHandler implements Runnable {
443 			final Process process;
444 			final ProcessExitHandler exitHandler;
445 
446 			ExitHandler(Process process, ProcessExitHandler exitHandler) {
447 				this.process = process;
448 				this.exitHandler = exitHandler;
449 			}
450 
451 			public void run() {
452 				try {
453 					process.waitFor();
454 				} catch (InterruptedException e) {
455 					// silently ignore and destroy all in finally
456 				} finally {
457 					// just in case, or if interrupted
458 					process.destroy();
459 					exitHandler.processExited(process);
460 				}
461 			}
462 		}

464 		private static Thread createPipe(InputStream src, OutputStream sink,
465 				BrokenPipeHandler bph, boolean autoFlush, boolean autoClose)
466 				throws IOException {
467 			if (src == null) {
468 				if (sink != null && autoClose)
469 					sink.close();
470 				return null;
471 			} else if (sink == null) {
472 				if (autoClose)
473 					src.close();
474 				return null;
475 			} else {
476 				return new Thread(
477 						new Pipe(src, sink, bph, autoFlush, autoClose));
478 			}
479 		}
