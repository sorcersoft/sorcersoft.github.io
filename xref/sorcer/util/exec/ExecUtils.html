100     /**
101      * Attach to the specified process, feed specified standard input, and
102      * return process' results. Waits for the process to complete and returns
103      * completion status and data written to standard output and error streams.
104      *
105      * @see #execCommand(Process)
106      *
107      * @param process
108      *            the process to attach to
109      * @param stdin
110      *            the data to redirect to process' standard input
111      * @return the results of the process
112      * @throws IOException
113      *             if an I/O error occurs
114      * @throws InterruptedException
115      *             if thread is interrupted before process ends
116      */
117     public static CmdResult execCommand(final Process process,
118                                         final InputStream stdin, boolean outLogged) throws IOException,
119             InterruptedException {
120         // concurrency to avoid stdio deadlocks
121         Redir stdout = null;
122         String out = null;
123         if (!outLogged) {
124             stdout = new Redir(process.getInputStream());
125             new Thread(stdout, tName("STDOUT-" + process.toString())).start();
126         }
127         Redir stderr = new Redir(process.getErrorStream());
128         new Thread(stderr, tName("STDERR-" + process.toString())).start();
129         // redirect input in the current thread
130         if (stdin != null) {
131             OutputStream pout = process.getOutputStream();
132             new RedirectingInputStream(stdin, true, true).redirectAll(pout);
133         }
134         process.waitFor();
135         int exitValue = process.exitValue();
136         Thread.sleep(50);
137         if (stdout != null) {
138             stdout.throwIfHadException();
139             out = new String(stdout.getResult());
140         }
141         stderr.throwIfHadException();
142         String err = new String(stderr.getResult());
143 
144         return new CmdResult(exitValue, out, err);
145     }

244     private static class Pipe implements Runnable {
245         final InputStream src;
246         final OutputStream sink;
247         final boolean autoFlush;
248         final boolean autoClose;
249         final BrokenPipeHandler brokenPipeHandler;
250         public Pipe(InputStream src, OutputStream sink,
251                     BrokenPipeHandler brokenPipeHandler) {
252             this(src, sink, brokenPipeHandler, true, false);
253         }
254         public Pipe(InputStream src, OutputStream sink,
255                     BrokenPipeHandler brokenPipeHandler,
256                     boolean autoFlush, boolean autoClose) {
257             this.src = src;
258             this.sink = sink;
259             this.brokenPipeHandler = brokenPipeHandler;
260             this.autoFlush = autoFlush;
261             this.autoClose = autoClose;
262         }
263         public void run() {
264             RedirectingInputStream sd =
265                 new RedirectingInputStream(src, autoFlush, autoClose);
266             try {
267                 sd.redirectAll(sink);
268             }
269             catch (IOException e) {
270                 if (brokenPipeHandler != null) {
271                     brokenPipeHandler.brokenPipe(e, src, sink);
272                 }
273             }
274         }
275     }

278     public static Thread executeCommandWithWorkerNoBlocking(
279             String[] command, final boolean printOutput,
280             final boolean printError, final long timeOut, File dir,
281             File logFile, boolean doSynchronizedLaunch) {
282 
283         Runtime runtime;
284         WorkerNoBlock worker;
285         Process process;
286         StreamGobbler outputGobbler, errorGobbler;
287 
288         // windows platform independent
289         //
290         if (GenericUtil.isWindows()) {
291             String[] ncmdarray = new String[command.length + 2];
292             ncmdarray[0] = "cmd";
293             ncmdarray[1] = "/C";
294             int ctr = 2;
295             for (int i = 0; i < command.length; i++) {
296                 ncmdarray[ctr] = command[i];
297                 ctr++;
298             }
299             command = ncmdarray;
300         }
301 
302         // mkdirs
303         //
304         if (!dir.exists()) dir.mkdirs();
305 
306         try {
307             if (doSynchronizedLaunch) {
308                 synchronized (GenericUtil.class) {
309                     runtime = Runtime.getRuntime();
310                     process = runtime.exec(command, null, dir);
311                     outputGobbler = new StreamGobbler(process.getInputStream(),
312                             "STD OUT", printOutput, logFile, dir);
313                     errorGobbler = new StreamGobbler(process.getErrorStream(),
314                             "STD ERR", printError, logFile, dir);
315                     outputGobbler.start();
316                     errorGobbler.start();
317                     worker = new WorkerNoBlock(process);
318                     worker.start();
319                 }
320             } else {
321                 runtime = Runtime.getRuntime();
322                 process = runtime.exec(command, null, dir);
323                 outputGobbler = new StreamGobbler(process.getInputStream(),
324                         "STD OUT", printOutput, logFile, dir);
325                 errorGobbler = new StreamGobbler(process.getErrorStream(),
326                         "STD ERR", printError, logFile, dir);
327                 outputGobbler.start();
328                 errorGobbler.start();
329                 worker = new WorkerNoBlock(process);
330                 worker.start();
331             }
332 
333         } catch (FileNotFoundException ex) {
334             String errorMessage = "the log file was not found.";
335             throw new RuntimeException(errorMessage, ex);
336 
337         } catch (IOException ex) {
338             String errorMessage = "the command: " + command
339                     + ", did not complete due to an " + "io error.";
340             throw new RuntimeException(errorMessage, ex);
341         }
342         return (Thread) worker;
343     }

385     private static class StreamGobbler extends Thread {
386 
387         private boolean displayStreamOutput;
388         private InputStream is;
389         private PrintWriter logPw;
390         private String type;
391         public boolean keepGoing = true;
392         private File dir = null;
393 
394         public StreamGobbler(InputStream is, String type,
395                              boolean displayStreamOutput, File logFile, File dir)
396                 throws FileNotFoundException {
397             this.is = is;
398             this.type = type;
399             this.displayStreamOutput = displayStreamOutput;
400             logPw = new PrintWriter(new FileOutputStream(logFile));
401             this.dir = dir;
402         }
403 
404         public void closeDown() {
405             GenericUtil.appendFileContents("StreamGobbler.closingDown(): setting flag to stop running "
406                     + " keepGoing = false now; stream gobbler type = " + type, dir);
407             keepGoing = false;
408         }
409 
410         public void run() {
411             try {
412                 InputStreamReader isr = new InputStreamReader(is);
413                 BufferedReader br = new BufferedReader(isr);
414                 String line = null;
415                 while ((line = br.readLine()) != null && keepGoing) {
416 
417                     GenericUtil.appendFileContents("StreamGobbler.run(): while loop; "
418                             + " stream gobbler type = " + type + "; line = " + line, dir);
419 
420                     if (displayStreamOutput) System.out.println(type + ">" + line);
421 
422                     GenericUtil.appendFileContents("StreamGobbler.run(): while loop; "
423                             + " stream gobbler type = " + type + "; here0", dir);
424 
425                     logPw.println(line);
426 
427 
428                     GenericUtil.appendFileContents("StreamGobbler.run(): while loop; "
429                             + " stream gobbler type = " + type + "; here1", dir);
430 
431                     logPw.flush();
432 
433 
434                     GenericUtil.appendFileContents("StreamGobbler.run(): while loop; "
435                             + " stream gobbler type = " + type + "; here2", dir);
436 
437                     while (!br.ready() && keepGoing) {
438                         //System.out.println("gobbler type = " + type + " is not ready.");
439                         GenericUtil.appendFileContents("StreamGobbler.run(): inner while loop; "
440                                 + "br not ready; stream gobbler type = " + type + "keepGoing = " + keepGoing, dir);
441                         try {
442                             Thread.sleep(1000);
443                         } catch (InterruptedException e) {
444                             e.printStackTrace();
445                         }
446                     }
447                     if (!keepGoing) break;
448                 }
449                 GenericUtil.appendFileContents("StreamGobbler.run(): exited run inner loop for type = "
450                         + type + "; keepGoing = " + keepGoing, dir);
451                 logPw.flush();
452                 logPw.close();
453                 br.close();
454                 isr.close();
455                 is.close();
456             } catch (IOException ioe) {
457                 System.out.println("***exception in gobbler type = " + type + ": " + ioe);
458                 GenericUtil.appendFileContents("StreamGobbler.run(): exception = "
459                         + ioe, dir);
460                 ioe.printStackTrace();
461             } finally {
462                 logPw.flush();
463                 logPw.close();
464                 try {
465                     is.close();
466                 } catch (IOException e) {
467                     // TODO Auto-generated catch block
468                     e.printStackTrace();
469                 }
470             }
471         }
472     }

475     private static class Redir implements Runnable {
476         final Pipe pipe;
477         IOException ex;
478         Redir(InputStream is) {
479             ByteArrayOutputStream bos = new ByteArrayOutputStream();
480             BrokenPipeHandler bph = new BrokenPipeHandler() {
481                 public void brokenPipe(IOException ex, InputStream src,
482                                        OutputStream sink) {
483                     setException(ex);
484                 }
485             };
486             this.pipe = new Pipe(is, bos, bph, true, true);
487         }
488         public void run() {
489             pipe.run();
490         }
491         synchronized void setException(IOException e) {
492             this.ex = e;
493         }
494         synchronized void throwIfHadException() throws IOException {
495             if (ex != null) throw ex;
496         }
497         public byte[] getResult() {
498             return ((ByteArrayOutputStream)pipe.sink).toByteArray();
499         }
500     }

502     private static class ProcessHandler {
503         final Process process;
504         final Thread tstdin;
505         final Thread tstdout;
506         final Thread tstderr;
507         final Thread texitHandler;
508         ProcessHandler(final Process process,
509                        InputStream stdin, boolean inAutoFlush, boolean inAutoClose,
510                        BrokenPipeHandler inBrokenHandler,
511                        OutputStream stdout, boolean outAutoFlush, boolean outAutoClose,
512                        BrokenPipeHandler outBrokenHandler,
513                        OutputStream stderr, boolean errAutoFlush, boolean errAutoClose,
514                        BrokenPipeHandler errBrokenHandler,
515                        final ProcessExitHandler exitHandler)
516             throws IOException
517         {
518             this.process = process;
519             this.tstdin = createPipe(stdin, process.getOutputStream(),
520                                      inBrokenHandler, inAutoFlush, inAutoClose);
521             this.tstdout = createPipe(process.getInputStream(), stdout,
522                                       outBrokenHandler, outAutoFlush, outAutoClose);
523             this.tstderr = createPipe(process.getErrorStream(), stderr,
524                                       errBrokenHandler, errAutoFlush, errAutoClose);
525             if (exitHandler != null) {
526                 this.texitHandler = new Thread(new ExitHandler(process, exitHandler), "Process-" + process + "-Exit");
527             }
528             else {
529                 texitHandler = null;
530             }
531         }
532 
533         void start() {
534             if (tstdin != null) tstdin.start();
535             if (tstdout != null) tstdout.start();
536             if (tstderr != null) tstderr.start();
537             if (texitHandler != null) texitHandler.start();
538         }

540         private static class ExitHandler implements Runnable {
541             final Process process;
542             final ProcessExitHandler exitHandler;
543             ExitHandler(Process process, ProcessExitHandler exitHandler) {
544                 this.process = process;
545                 this.exitHandler = exitHandler;
546             }
547             public void run() {
548                 try {
549                     process.waitFor();
550                 }
551                 catch (InterruptedException e) {
552                     // silently ignore and destroy all in finally
553                 }
554                 finally {
555                     // just in case, or if interrupted
556                     process.destroy();
557                     exitHandler.processExited(process);
558                 }
559             }
560         }

562         private static Thread createPipe(InputStream src, OutputStream sink,
563                                          BrokenPipeHandler bph,
564                                          boolean autoFlush, boolean autoClose)
565             throws IOException
566         {
567             if (src == null) {
568                 if (sink != null && autoClose) sink.close();
569                 return null;
570             }
571             else if (sink == null) {
572                 if (autoClose) src.close();
573                 return null;
574             }
575             else {
576                 return new Thread(new Pipe(src, sink, bph, autoFlush, autoClose));
577             }
578         }
