214 	static public void main(String argv[]) {
215 		String curToken = null;
216 		//System.out.println("nsh main args: " + Arrays.toString(argv));	
217 		if (argv.length > 0) {
218             if ((argv.length == 1 && argv[0].startsWith("--"))
219 					|| (argv.length == 2 && (argv[0].equals("-e"))
220 							|| argv[0].equals("-f")
221 							|| argv[0].equals("-version") || argv[0]
222 							.equals("-help")) || argv.length == 1) {
223 				commandLine = false;
224 			} else {
225 				commandLine = true;
226 			}
227 		}
228 
229 		try {
230 			// default shellOutput
231 			shellOutput = System.out;
232 			if (commandLine) {
233 				shellOutput.println("SORCER Network Shell (nsh " + CUR_VERSION
234 						+ ", JVM: " + System.getProperty("java.version"));
235 				shellOutput.println("Type 'quit' to terminate the shell");
236 				shellOutput.println("Type 'help' for command help");
237 			}
238 			
239 			argv = buildInstance(argv);
240 			if (!instance.commandLine) {
241 				// System.out.println("main appMap: " + appMap);
242 				execNoninteractiveCommand(argv);
243 				//Thread.sleep(1000);
244 				System.exit(0);
245 			}
246 			if (argv.length == 1 && argv[0].indexOf("-") == 0) {
247 				shellOutput.println(UNKNOWN_COMMAND_MSG);
248 				shellOutput.flush();
249 				System.exit(1);
250 			}
251 
252 			principal = new SorcerPrincipal(NetworkShell.getUserName());
253 			principal.setId(NetworkShell.getUserName());
254 
255 			shellOutput.print(SYSTEM_PROMPT);
256 			shellOutput.flush();
257 			request = "";
258 			request = shellInput.readLine();
259 		} catch (Throwable e) {
260 			e.printStackTrace();
261 			System.exit(1);
262 		}
263 		ShellCmd cmd = null;
264 		//System.out.println("main request: " + request);
265 		if (request==null) {
266 			// Exit when CTRL+D is pressed
267 			System.exit(0);
268 		}
269 		while ((request.length() > 0 && BUILTIN_QUIT_COMMAND.indexOf(request) < 0)
270 				|| request.length() == 0) {
271 			shellTokenizer = new StringTokenizer(request);
272 			curToken = "";
273 			if (shellTokenizer.hasMoreTokens()) {
274 				curToken = shellTokenizer.nextToken();
275 			}
276 			try {
277 				if (commandTable.containsKey(curToken)) {
278 					cmd = (ShellCmd) commandTable.get(curToken);
279 					cmd.execute();
280 				}
281 				// admissible shortcuts in the 'synonyms' map
282 				else if (aliases.containsKey(curToken)) {
283 					String cmdName = aliases.get(curToken);
284 					int i = cmdName.indexOf(" -");
285 					if (i > 0) {
286 						request = cmdName;
287 						cmdName = cmdName.substring(0, i);
288 						shellTokenizer = new StringTokenizer(request);
289 					} else {
290 						request = cmdName + " " + request;
291 						cmdName = new StringTokenizer(cmdName).nextToken();
292 						shellTokenizer = new StringTokenizer(request);
293 					}
294 					cmd = (ShellCmd) commandTable.get(cmdName);
295 					cmd.execute();
296 				} else if (request.length() > 0) {
297 					if (request.equals("?")) {
298 						instance.listCommands();
299 
300 					} else {
301 						shellOutput.println(UNKNOWN_COMMAND_MSG);
302 					}
303 				}
304 				shellOutput.print(SYSTEM_PROMPT);
305 				shellOutput.flush();
306 				String in = shellInput.readLine();
307 				// Exit if CTRL+D pressed
308 				if (in==null) System.exit(0);
309 				
310 				// fore !! run the previous command
311 				if (!in.equals("!!")) {
312 					instance.request = in;
313 				}
314 			} catch (IOException io) {
315                 shellOutput.println(io.getMessage());
316                 try {
317                     request = shellInput.readLine();
318                 } catch (IOException e) {
319                     e.printStackTrace();
320                     System.exit(1);
321                 }
322             } catch (Throwable ex) {
323 				ex.printStackTrace();
324 				try {
325 					request = shellInput.readLine();
326 				} catch (IOException e) {
327 					e.printStackTrace();
328 					System.exit(1);
329 				}
330 			}
331 		}
332 		// shellOutput.println("Thanks for using the SORCER Network Shell!");
333 	}

343 	static public void startApplication(String appPath)
344 			throws EvaluationException, IOException, InterruptedException {
345 		shellOutput.println("nsh executing: " + appPath);
346 		CmdResult result = null;
347 		if (appPath.endsWith(".xml")) {
348 			result = ExecUtils.execCommand("ant -f " + appPath);
349 			shellOutput.println("cmd result out: " + result.getOut());
350 		} else {
351 			result = ExecUtils.execCommand(appPath);
352 			shellOutput.println("cmd result out: " + result.getOut());
353 		}
354 		if (result.getErr() != null && result.getErr().length() > 0)
355 			shellOutput.println("cmd result err: " + result.getErr());
356 		shellOutput.println("cmd exit value: " + result.getExitValue());
357 
358 		if (result.getExitValue() != 0)
359 			throw new RuntimeException("Failed to start " + appPath);
360 	}

362 	static private void execNoninteractiveCommand(String args[])
363 			throws Throwable {
364 		// check for external commands
365 		System.out.println("noninteractive nsh: " + Arrays.toString(args));
366 
367 		if (args[0].indexOf("--") == 0) {
368 			String path = nishAppMap.get(args[0].substring(2));
369             if (path!=null)
370                 startApplication(path);
371             else
372                 shellOutput.println("No such internal command available: " + args[0].substring(2));
373 			System.exit(0);
374 		}
375 		request = arrayToRequest(args);
376         try {
377             if (args.length == 1) {
378                 if (args[0].equals("-version")) {
379                     shellOutput.println("SORCER Network Shell (nsh " + CUR_VERSION
380                             + ", JVM: " + System.getProperty("java.version"));
381                 } else if (args[0].equals("-help")) {
382                     ShellCmd cmd = (HelpCmd) commandTable.get("help");
383                     cmd.execute();
384                 } else {
385                     // Added reading the file as default first argument
386                     // Check if file exists
387                     ShellCmd cmd = (ShellCmd) commandTable.get("exert");
388                     cmd.execute();
389                 }
390             } else if (args.length == 2) {
391                 if (args[0].equals("-f")) {
392                     // evaluate file
393                     ShellCmd cmd = (ShellCmd) commandTable.get("exert");
394                     cmd.execute();
395                 } else if (args[0].equals("-e")) {
396                     // evaluate command line expression
397                     ExertCmd cmd = (ExertCmd) commandTable.get("exert");
398                     cmd.setScriptFile(huntForTheScriptFile(args[1]));
399                     cmd.execute();
400                 }
401             }
402         } catch (IOException io) {
403             shellOutput.println(io.getMessage());
404             // Do nothing since an error message was already printed out by th huntForTheScriptFile method
405         }

453 	public void enumerateCommands() {
454 		for (int i = 0; i < shellCommands.length; i++) {
455 			addToCommandTable(shellCommands[i], shellCmdClasses[i]);
456 			ShellCmd cmdInstance = commandTable.get(shellCommands[i]);
457 			String[] subCmds = toArray(cmdInstance.getCommandWord(), ", ");
458 			if (subCmds.length > 1) {
459 				for (String subCmd : subCmds) {
460 					commandTable.put(subCmd, cmdInstance);
461 					// System.out.println("subCmd " + subCmd);
462 				}
463 			}
464 		}
465 		// verify that the basic minimal commands are available
466 		for (int i = 0; i < shellCommands.length; i++)
467 			if (!commandTable.containsKey(shellCommands[i])
468 					&& ((ShellCmd) commandTable.get(shellCommands[i]))
469 							.getCommandWord().indexOf(shellCommands[i]) >= 0) {
470 				shellOutput.println("Missing basic command :  "
471 						+ shellCommands[i]);
472 				System.exit(1);
473 			}
474 	}

490 	private void initSettings(String[] groups, LookupLocator[] locators,
491 			long discoTimeout) throws ConfigurationException {
492 		settings.put(GROUPS, groups);
493 		settings.put(LOCATORS, locators);
494 		Properties props = new Properties();
495 		String sorcerHome = System.getenv("SORCER_HOME");
496 		if (sorcerHome == null)
497 			throw new RuntimeException("SORCER_HOME must be set");
498 		props.put("java.protocol.handler.pkgs", "net.jini.url|sorcer.util.bdb.sos");
499 		Properties addedProps = getConfiguredSystemProperties();
500 		props.putAll(addedProps);
501 		Properties sysProps = System.getProperties();
502 		sysProps.putAll(props);
503 		System.setProperties(sysProps);
504 		settings.put(SYS_PROPS, props);
505 		settings.put(DISCOVERY_TIMEOUT, discoTimeout);
506 		if (homeDir == null)
507 			homeDir = System.getProperty("user.dir");
508 		currentDir = new File(homeDir);
509 
510 		if (!commandLine) {
511 			String logDirPath = System.getProperty(COMPONENT + "logDir",
512 					sorcerHome + File.separator + "bin" + File.separator
513 							+ "shell" + File.separator + "logs");
514 		
515 			File logDir = new File(logDirPath);
516 			if (!logDir.exists()) {
517 				if (!logDir.mkdir()) {
518 					System.err.println("Unable to create log directory at "
519 							+ logDir.getAbsolutePath());
520 				}
521 			}
522 
523 			shellLog = new File(logDir, "nsh.log");
524 			System.out.println("Created the nsh shel log file: " + shellLog.getAbsolutePath());
525 			if (shellLog.exists()) {
526 				shellLog.delete();
527 			}
528 			if (logDir.exists()) {
529 				try {
530 					System.setErr(new PrintStream(
531 							new FileOutputStream(shellLog)));
532 				} catch (FileNotFoundException e) {
533 					e.printStackTrace();
534 				}
535 			}
536 			System.err
537 					.println("===============================================");
538 			System.err.println(shellName
539 					+ " non-interactive network shell\n"
540 					+ "Log creation : " + new Date(startTime).toString() + "\n"
541 					+ "Operator : " + System.getProperty("user.name"));
542 			System.err
543 					.println("===============================================");
544 			if (addedProps.size() > 0) {
545 				StringBuilder buff = new StringBuilder();
546 				for (Map.Entry<Object, Object> entry : addedProps.entrySet()) {
547 					String key = (String) entry.getKey();
548 					String value = (String) entry.getValue();
549 					buff.append("\n");
550 					buff.append("    ").append(key).append("=").append(value);
551 				}
552 				System.err.println("Added System Properties {"
553 						+ buff.toString() + "\n}");
554 			}
555 		}
556 
557 		setShellOutput((PrintStream) sysConfig.getEntry(CONFIG_COMPONENT,
558 				"output", PrintStream.class, System.out));
559 	}

562 	 * Set system properties from configuration
563 	 */
564 	private Properties getConfiguredSystemProperties()
565 			throws ConfigurationException {
566 		Configuration config = getConfiguration();
567 		Properties sysProps = new Properties();
568 		String[] systemProperties = (String[]) config.getEntry(
569 				"sorcer.tools.shell", "systemProperties", String[].class,
570 				new String[0]);
571 		if (systemProperties.length > 0) {
572 			if (systemProperties.length % 2 != 0) {
573 				System.err
574 						.println("systemProperties elements has odd length : "
575 								+ systemProperties.length);
576 			} else {
577 				for (int i = 0; i < systemProperties.length; i += 2) {
578 					String name = systemProperties[i];
579 					String value = systemProperties[i + 1];
580 					sysProps.setProperty(name, value);
581 				}
582 			}
583 		}
584 		return sysProps;
585 	}

666 		public void execute() {		
667 			// noninteractive shell
668 			if (shellTokenizer == null) {
669 				instance.listCommands();
670 				return;

886 			int port = 0;
887 			/* The first token is the "http" token */
888 			tok.nextToken();

890 				String value = tok.nextToken();

933 							System.in));

938 								|| response.startsWith("Y")) {

940 							if (createWebster(port, instance.httpRoots, instance.httpJars, shellOutput))

944 				} catch (IOException e) {

947 							+ "Exception :" + e.getClass().getName() + ": "
