76      /**
77       * Create a new Webster. The port is determined by the
78       * webster.port system property. If the
79       * webster.port system property does not exist, an
80       * anonymous port will be allocated.
81       *
82       * @throws BindException if Webster cannot create a socket
83       */
84      public Webster() throws BindException {
85          String s = System.getProperty("webster.port");
86          if (s != null && s != "0") {
87              try {
88                  port = new Integer(s);
89              } catch (NumberFormatException e) {
90                  if (s.equals("${webster.port}")) {
91                      throw new RuntimeException("The required system property for 'webster.port' not set");
92                  }
93              }
94          } else {
95              port = getAvailablePort();
96          }
97          this.isDaemon = false;
98          initialize();
99      }

225     /**
226      * Create a new Webster, compatible with the ServiceStarter mechanism in
227      * Jini 2.0
228      *
229      * @param args      String[] of options. Valid options are [-port port],
230      *                  [-roots list-of-roots], [-bindAddress address], [-minThreads minThreads],
231      *                  [-maxThreads maxThreads] [-soTimeout soTimeout]
232      * @param lifeCycle The LifeCycle object, may be null
233      * @throws BindException if Webster cannot create a socket
234      */
235     public Webster(String[] args, com.sun.jini.start.LifeCycle lifeCycle)
236             throws Exception {
237         if (args == null)
238             throw new NullPointerException("Configuration is null");
239 
240         Webster.webster = this;
241         this.lifeCycle = lifeCycle;
242         String[] configRoots = null;
243         String[] configArgs = null;
244         String roots = null;
245         String[] options = null;
246         String bindAddress = null;
247         if (args.length == 1 && new File(args[0]).isFile()) {
248             final Configuration config = ConfigurationProvider.getInstance(args);
249             try {
250                 configRoots = (String[]) config.getEntry(CODESERVER,
251                                                          "roots", String[].class);
252             } catch (Exception e) {
253                 e.printStackTrace();
254                 configRoots = null;
255             }
256             try {
257                 configArgs = (String[]) config.getEntry(CODESERVER,
258                                                         "options", String[].class);
259             } catch (Exception e) {
260                 e.printStackTrace();
261                 configArgs = null;
262             }
263 //			logger.info("webster roots: " + Arrays.toString(configRoots));
264 //			logger.info("webster options: " + Arrays.toString(configArgs));
265         }
266         if (configRoots != null) {
267             websterRoot = configRoots;
268             options = configArgs;
269         } else {
270             options = args;
271         }
272         //logger.info("roots concat: " + roots + "\noptions: " + Arrays.toString(options));
273         for (int i = 0; i < options.length; i++) {
274             String option = options[i];
275             if (option.equals("-port")) {
276                 i++;
277                 this.port = Integer.parseInt(options[i]);
278             } else if (option.equals("-roots")) {
279                 i++;
280                 roots = options[i];
281             } else if (option.equals("-bindAddress")) {
282                 i++;
283                 bindAddress = options[i];
284             } else if (option.equals("-startPort")) {
285                 i++;
286                 startPort = Integer.parseInt(options[i]);
287             } else if (option.equals("-endPort")) {
288                 i++;
289                 endPort = Integer.parseInt(options[i]);
290             } else if (option.equals("-minThreads")) {
291                 i++;
292                 minThreads = Integer.parseInt(options[i]);
293             } else if (option.equals("-maxThreads")) {
294                 i++;
295                 maxThreads = Integer.parseInt(options[i]);
296             } else if (option.equals("-soTimeout")) {
297                 i++;
298                 soTimeout = Integer.parseInt(options[i]);
299             } else if (option.equals("-isDaemon")) {
300                 i++;
301                 isDaemon = Boolean.parseBoolean(options[i]);
302             } else if (option.equals("-debug")) {
303                 i++;
304                 debug = Boolean.parseBoolean(options[i]);
305             } else {
306                 throw new IllegalArgumentException(option);
307             }
308         }
309         if (websterRoot != null)
310             init(bindAddress);
311         else
312             initialize(roots, bindAddress);
313 
314     }

349     /*
350      * Initialize Webster
351      *
352      * @param roots The root(s) to serve code from. This is a semicolon
353      * delimited list of directories
354      */
355     private void init(String bindAddress)
356             throws BindException {
357         String str = null;
358         if (!debug) {
359             str = System.getProperty("webster.debug");
360             if (str != null && str.equals("true"))
361                 debug = true;
362         }
363         str = System.getProperty("webster.tmp.dir");
364         if (str != null) {
365             tempDir = str;
366             if (debug)
367                 System.out.println("tempDir: " + tempDir);
368         }
369 
370         for (int j = 0; j < websterRoot.length; j++) {
371             if (debug) {
372                 System.out.println("Root " + j + " = " + websterRoot[j]);
373             }
374             if (logger.isDebugEnabled()) {
375                 logger.debug("Root " + j + " = " + websterRoot[j]);
376             }
377         }
378 
379         InetAddress addr;
380         try {
381             if (bindAddress == null) {
382                 bindAddress = System.getProperty("webster.interface");
383             }
384             if (bindAddress == null)
385                 bindAddress = InetAddress.getLocalHost().getHostAddress();
386 
387             addr = InetAddress.getByName(bindAddress);
388         } catch (UnknownHostException e) {
389             logger.error( "Bind address server socket failure", e);
390             return;
391         }
392 
393         start(addr);
394     }

396     private void start(InetAddress addr) throws BindException {
397         if (port == 0) {
398             try {
399                 port = getPortAvailable();
400             } catch (IOException e) {
401                 e.printStackTrace();
402                 logger.error( "Cannot determine a server socket port", e);
403                 System.exit(1);
404             }
405             startPort = port;
406             endPort = port;
407         } else {
408             if (startPort == endPort) {
409                 startPort = port;
410                 endPort = port;
411             }
412         }
413 
414         for (int i = startPort; i <= endPort; i++) {
415             start(i, addr);
416             return;
417         }
418     }

420     // start with the first available port in the range STARTPORT-ENDPORT
421     private void start(int websterPort, InetAddress address) throws BindException {
422         try {
423             port = websterPort;
424             // check if the port is not required by the JVM system property
425             String s = System.getProperty("webster.port");
426             if (s != null && s.length() > 0) {
427                 port = new Integer(s);
428             }
429             ss = new ServerSocket(port, 0, address);
430         } catch (IOException ioe) {
431             if (startPort == endPort) {
432                 //logger.error( "Port bind server socket failure: " + endPort, ioe);
433                 throw new BindException(ioe.getMessage());
434                 //System.exit(1);
435             } else {
436                 //System.err.println("Port bind server socket failure: " + port);
437                 throw new BindException(ioe.getMessage());
438             }
439         }
440         port = ss.getLocalPort();
441 
442         if (debug)
443             System.out.println("Webster serving on : "
444                                        + ss.getInetAddress().getHostAddress() + ":" + port);
445         if (logger.isInfoEnabled())
446             logger.info("Webster serving on: "
447                                 + ss.getInetAddress().getHostAddress() + ":" + port);
448         if (debug)
449             System.out.println("Webster listening on port: " + port);
450         if (logger.isDebugEnabled()) {
451             logger.debug("Webster listening on port: " + port);
452         }
453         try {
454             pool = (ThreadPoolExecutor) Executors.newFixedThreadPool(maxThreads);
455             if (debug)
456                 System.out.println("Webster minThreads [" + minThreads + "], "
457                                            + "maxThreads [" + maxThreads + "]");
458             if (logger.isDebugEnabled())
459                 logger.debug("Webster minThreads [" + minThreads + "], "
460                                     + "maxThreads [" + maxThreads + "]");
461         } catch (Exception e) {
462             logger.error( "Could not create ThreadPool", e);
463             throw new RuntimeException("Could not create Thread Pool");
464         }
465         if (soTimeout > 0) {
466             if (debug)
467                 System.out.println("Webster Socket SO_TIMEOUT set to ["
468                                            + soTimeout + "] millis");
469             if (logger.isDebugEnabled())
470                 logger.debug("Webster Socket SO_TIMEOUT set to [" + soTimeout
471                                     + "] millis");
472         }
473         /* Set system property */
474         System.setProperty(CODESERVER, "http://" + getAddress() + ":"
475                 + getPort());
476 
477         if (logger.isDebugEnabled())
478             logger.debug("Webster isDaemon: " + isDaemon);
479 
480         Thread runner = new Thread(this, "Webster");
481         if (isDaemon) {
482             runner.setDaemon(true);
483         }
484         runner.start();
485     }

515      * Setup the websterRoot property
516      */
517     private void setupRoots(String roots) {
518         if (roots == null)
519             throw new NullPointerException("roots is null");
520         StringTokenizer tok = new StringTokenizer(roots, ";");
521         websterRoot = new String[tok.countTokens()];
522         if (websterRoot.length > 1) {
523             for (int j = 0; j < websterRoot.length; j++) {
524                 websterRoot[j] = tok.nextToken();
525                 if (debug)
526                     System.out.println("Root " + j + " = " + websterRoot[j]);
527                 if (logger.isDebugEnabled())
528                     logger.debug("Root " + j + " = " + websterRoot[j]);
529             }
530         } else {
531             websterRoot[0] = roots;
532             if (debug)
533                 System.out.println("Root  = " + websterRoot[0]);
534             if (logger.isDebugEnabled())
535                 logger.debug("Root  = " + websterRoot[0]);
536         }

567     private String readRequest(InputStream inputStream) throws IOException {
568         StringBuilder sb = new StringBuilder();
569         int read;
570         int prev = -1;
571         while ((read = inputStream.read()) != -1) {
572             if (read != '\n' && read != '\r')
573                 sb.append((char) read);
574             if (read == '\n' && prev == '\r') {
575                 break;
576             }
577             if (read == '\r' && prev == '0') {
578                 //sb.delete(0, sb.length());
579                 break;
580             }
581             prev = read;
582         }
583         return sb.toString();
584     }

587     public void run() {
588         Socket s;
589         try {
590             loadMimes();
591             String fileName;
592             while (run) {
593                 s = ss.accept(); // accept incoming requests
594                 if (soTimeout > 0) {
595                     s.setSoTimeout(soTimeout);
596                 }
597                 String line;
598                 Properties header = new Properties();
599                 DataInputStream inputStream = null;
600                 try {
601                     inputStream = new DataInputStream(s.getInputStream());
602                     StringBuilder lineBuilder = new StringBuilder();
603                     StringTokenizer tokenizer;
604                     while ((line = readRequest(inputStream)).length() != 0) {
605                         if (lineBuilder.length() > 0)
606                             lineBuilder.append("\n");
607                         lineBuilder.append(line);
608                         tokenizer = new StringTokenizer(line, ":");
609                         String aToken = tokenizer.nextToken().trim();
610                         if (tokenizer.hasMoreTokens()) {
611                             header.setProperty(aToken, tokenizer.nextToken().trim());
612                         }
613                     }
614                     line = lineBuilder.toString();
615                     int port = s.getPort();
616                     String from = s.getInetAddress().getHostAddress() + ":" + port;
617                     if (debug) {
618                         StringBuilder buff = new StringBuilder();
619                         buff.append("From: ").append(from).append(", ");
620                         if (soTimeout > 0)
621                             buff.append("SO_TIMEOUT: ").append(soTimeout).append(", ");
622                         buff.append("Request: ").append(line);
623                         System.out.println("\n"+buff.toString());
624                     }
625                     if (logger.isDebugEnabled()) {
626                         StringBuilder buff = new StringBuilder();
627                         buff.append("From: ").append(from).append(", ");
628                         if (soTimeout > 0)
629                             buff.append("SO_TIMEOUT: ").append(soTimeout).append(", ");
630                         buff.append("Request: ").append(line);
631                         if (logger.isDebugEnabled())
632                             logger.debug(buff.toString());
633                     }
634                     if (line != null && line.length()>0) {
635                         tokenizer = new StringTokenizer(line, " ");
636                         if (!tokenizer.hasMoreTokens())
637                             break;
638                         String token = tokenizer.nextToken();
639                         fileName = tokenizer.nextToken();
640                         if (fileName.startsWith("/"))
641                             fileName = fileName.substring(1);
642                         if (token.equals("GET")) {
643                             header.setProperty("GET", fileName);
644                         } else if (token.equals("PUT")) {
645                             header.setProperty("PUT", fileName);
646                         } else if (token.equals("DELETE")) {
647                             header.setProperty("DELETE", fileName);
648                         } else if (token.equals("HEAD")) {
649                             header.setProperty("HEAD", fileName);
650                         }
651                         while (tokenizer.hasMoreTokens()) {
652                             String aToken = tokenizer.nextToken().trim();
653                             if (tokenizer.hasMoreTokens()) {
654                                 header.setProperty(aToken,
655                                                    tokenizer.nextToken().trim());
656                             }
657                         }
658                         if (header.getProperty("GET") != null) {
659                             pool.execute(new GetFile(s, fileName));
660                         } else if (header.getProperty("PUT") != null) {
661                             pool.execute(new PutFile(s, fileName, header, inputStream));
662                         } else if (header.getProperty("DELETE") != null) {
663                             pool.execute(new DelFile(s, fileName));
664                         } else if (header.getProperty("HEAD") != null) {
665                             pool.execute(new Head(s, fileName));
666                         } else {
667                             if (debug)
668                                 System.out.println(
669                                         "bad request [" + line + "] from " + from);
670                             if (logger.isDebugEnabled())
671                                 logger.debug("bad request [" + line + "] " +
672                                                    "from " + from);
673                             DataOutputStream clientStream =
674                                     new DataOutputStream(
675                                             new BufferedOutputStream(
676                                                     s.getOutputStream()));
677                             clientStream.writeBytes(
678                                     "HTTP/1.0 400 Bad Request\r\n\r\n");
679                             clientStream.flush();
680                             clientStream.close();
681                         }
682                     } /* if line != null */
683                 } catch (Exception e) {
684                     DataOutputStream clientStream =
685                             new DataOutputStream(
686                                     new BufferedOutputStream(
687                                             s.getOutputStream()));
688                     clientStream.writeBytes(
689                             "HTTP/1.0 500 Internal Server Error\n" +
690                                     "MIME-Version: 1.0\n" +
691                                     "Server: " + SERVER_DESCRIPTION + "\n" +
692                                     "\n\n<H1>500 Internal Server Error</H1>\n"
693                                     + e);
694                     clientStream.flush();
695                     clientStream.close();
696                     inputStream.close();
697                     logger.warn("Getting Request", e);
698                 }
699             }
700         } catch (Exception e) {
701             if (run) {
702                 e.printStackTrace();
703                 logger.warn("Processing HTTP Request", e);
704             }
705         }
706     }

708     /**
709      * Get an anonymous port
710      *
711      * @return An available port created by instantiating a
712      *         <code>java.net.ServerSocket</code> with a port of 0
713      * @throws IOException If an available port cannot be obtained

773     protected File parseFileName(String filename) {
774         StringBuffer fn = new StringBuffer(filename);
775         for (int i = 0; i < fn.length(); i++) {
776             if (fn.charAt(i) == '/')
777                 fn.replace(i, i + 1, File.separator);
778         }
779         File f = null;

781         for (String root : roots) {

784             //System.out.println("root: "+root+", looking for "+f.getPath());
785             if (f.exists()) {
786                 return (f);

792     protected String[] expandRoots() {
793         List<String> expandedRoots = new LinkedList<String>();
794         if (hasWildcard()) {
795             String[] rawRoots = websterRoot;
796             for (String root : rawRoots) {
797                 int wildcard;
798                 if ((wildcard = root.indexOf('*')) != -1) {
799                     String prefix = root.substring(0, wildcard);
800                     File prefixFile = new File(prefix);
801                     if (prefixFile.exists()) {
802                         String suffix =
803                                 (wildcard < (root.length() - 1)) ?
804                                         root.substring(wildcard + 1) : "";
805                         String[] children = prefixFile.list();
806                         for (String aChildren : children) {
807                             expandedRoots.add(prefix + aChildren + suffix);
808                         }
809                     } else {
810                         // Eat the root entry if it's wildcarded and doesn't
811                         // exist
812                     }
813                 } else {
814                     expandedRoots.add(root);
815                 }
816             }
817         }
818         String[] roots;
819         if (expandedRoots.size() > 0) {
820             roots = expandedRoots.toArray(new String[expandedRoots.size()]);
821         } else {
822             roots = websterRoot;
823         }
824         return (roots);
825     }
