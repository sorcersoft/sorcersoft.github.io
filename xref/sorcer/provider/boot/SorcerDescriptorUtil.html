62      /**
63       * Get the {@link com.sun.jini.start.ServiceDescriptor} instance for
64       * Webster.
65       *
66       * @param policy
67       *            The security policy file to use
68       *            The roots webster should serve
69       * @return The {@link com.sun.jini.start.ServiceDescriptor} instance for
70       *         webster using an anonymous port. The <tt>webster.jar</tt> file
71       *         will be loaded from <tt>sorcer.home/common/sorcer/webster.com</tt>
72       *
73       * @throws IOException
74       *             If there are problems getting the anonymous port
75       * @throws RuntimeException
76       *             If the <tt>sorcer.home</tt> system property is not set
77       */
78      public static ServiceDescriptor getWebster(String policy, int port)
79              throws IOException {
80          if (Resolver.isMaven()) {
81              String[] roots = {
82                      sorcerHome + "/deploy",
83                      sorcerHome + "/lib/river",
84                      Resolver.getRepoDir()
85              };
86              return (getWebster(policy, port, roots));
87          } else {
88              String[] roots = {
89                      sorcerHome + "/deploy",
90                      sorcerHome + "/lib/river",
91                      Resolver.getRootDir(),
92                      Resolver.getRepoDir()
93              };
94              return (getWebster(policy, port, roots));
95          }
96      }

186 	 * Get the {@link com.sun.jini.start.ServiceDescriptor} instance for
187 	 * {@link sorcer.tools.webster.Webster}.
188 	 * 
189 	 * @param policy
190 	 *            The security policy file to use
191 	 * @param port
192 	 *            The port webster should use
193 	 * @param roots
194 	 *            The roots webster should serve
195 	 * @param debug
196 	 *            If true, set the <tt>sorcer.tools.debug</tt> property
197 	 * @return The {@link com.sun.jini.start.ServiceDescriptor} instance for
198 	 *         webster using a specified port. The <tt>webster.jar</tt> file
199 	 *         will be loaded from <tt>${sorcer.home}/common/sorcer</tt>
200 	 * 
201 	 * @throws IOException
202 	 *             If there are problems getting the anonymous port
203 	 */
204 	public static ServiceDescriptor getWebster(String policy, int port,
205 			String[] roots, String address, int startPort, int endPort,
206 			boolean debug, boolean isDaemon) throws IOException {
207 		logger.info("policy: " + policy + ", port: " + port + ", roots: "
208 				+ Arrays.toString(roots) + ", address: " + "startPort: "
209 				+ startPort + ", endPort:" + endPort 
210 				+ ", debug: " + debug + ", isDaemon: " + isDaemon);
211 		int websterPort = 0;
212 		if (sorcerHome == null)
213 			throw new RuntimeException("'sorcer.home' property not declared");
214 		// anonymous case
215 		if (port == 0) {
216 			websterPort = Booter.getAnonymousPort(); 
217 			System.setProperty("webster.port", "" + port);
218 		}
219 		else if (port > 0) {
220 			websterPort = port;
221 		} else {
222 			// use SORCER environment configuration
223 			websterPort = Booter.getPort();
224 		}
225 		
226 		String importCodeBase = Resolver.resolveAbsolute(WEBSTER);
227 		String websterRoots = StringUtils.join(roots, ';');
228 		String websterClass = "sorcer.tools.webster.Webster";
229 		if (debug) {
230 			System.setProperty("sorcer.tools.webster.debug", "true");
231 		}
232 		String websterAddress = address;
233 		if (address == null || address.length() == 0) {
234 			websterAddress = Booter.getWebsterInterface();
235 			//Booter.getHostAddressFromProperty("java.rmi.server.hostname");
236 		}
237 		System.setProperty("provider.webster.interface", websterAddress);
238 		logger.fine("webster started at: " + websterAddress + ":" + websterPort + ", hostname: " + address);
239 		return (new NonActivatableServiceDescriptor("", policy, importCodeBase,
240 				websterClass,
241 				new String[] { "-port", "" + websterPort, "-roots", websterRoots,
242 						"-bindAddress", websterAddress, "-startPort",
243 						"" + startPort, "-endPort", "" + endPort, 
244 						"-isDaemon", "" + isDaemon, "-debug", "" + debug}));
245 	}
