42  /**
43   * A SorcerILFactory can be used with object interfaces as its services. Those
44   * services exposed as interfaces do need implement Remote. A
45   * {@link sorcer.core.provider.ServiceProvider} or
46   * {@link sorcer.core.provider.bean.ProviderBean} using this factory
47   * should tell the factory what objects should be exposed as services and the
48   * invocation dispatcher created by this factory will manage the delegation of
49   * the method calls to the right exposed object. SORCER service beans (objects
50   * with methods taking a parameter {@link sorcer.service.Context} and returning
51   * {@link sorcer.service.Context} can be used transparently as
52   * {@link sorcer.service.Service}s with either
53   * {@link sorcer.core.provider.ServiceProvider} or
54   * {@link sorcer.core.provider.bean.ProviderBean}.
55   */

111 	/**
112 	 * Creates a <code>SorcerILFactory</code> with the specified server
113 	 * constraints, permission class, and class loader. The server constraints,
114 	 * if not <code>null</code>, are used to enforce minimum constraints for
115 	 * remote calls. The permission class, if not <code>null</code>, is used to
116 	 * perform server-side access control on incoming remote calls. The class
117 	 * loader, which may be <code>null</code>, is passed to the superclass
118 	 * constructor and is used by the {@link #createInstances createInstances}
119 	 * method.
120 	 * 
121 	 * @param serverConstraints
122 	 *            the server constraints, or <code>null</code>
123 	 * @param permissionClass
124 	 *            the permission class, or <code>null</code>
125 	 * @param loader
126 	 *            the class loader, or <code>null</code>
127 	 * @param services
128 	 *            the object to be exposed as a service by the dispatcher of
129 	 *            this ILFactory
130 	 * 
131 	 * @throws IllegalArgumentException
132 	 *             if the permission class is abstract, is not a subclass of
133 	 *             {@link Permission}, or does not have a public constructor
134 	 *             that has either one <code>String</code> parameter or one
135 	 *             {@link Method}parameter and has no declared exceptions
136 	 */
137 	public SorcerILFactory(MethodConstraints serverConstraints,
138 			Class permissionClass, ClassLoader loader, Map serviceBeans)
139 			throws IllegalArgumentException {
140 		super(serverConstraints, permissionClass, loader);
141 		serviceBeanMap = serviceBeans;
142         Map<Class, Object> superIfaces = new HashMap<Class, Object>();
143         for (Map.Entry<Class, Object> e : serviceBeanMap.entrySet()) {
144             for (Class iface : e.getKey().getInterfaces()) {
145                 superIfaces.put(iface, e.getValue());
146             }
147         }
148         serviceBeanMap.putAll(superIfaces);
149     }

151 	/**
152 	 * Returns a new array containing any additional interfaces that the proxy
153 	 * should implement, beyond the interfaces obtained by passing
154 	 * <code>impl</code> to the {@link #getRemoteInterfaces getRemoteInterfaces}
155 	 * method.
156 	 * 
157 	 * <p>
158 	 * <code>SorcerILFactory</code> implements this method to return a new array
159 	 * containing the interfaces of all services to be exposed and
160 	 * {@link RemoteMethodControl}and {@link TrustEquivalence}interfaces, in
161 	 * that order.
162 	 * 
163 	 * @throws NullPointerException
164 	 *             {@inheritDoc}
165 	 */
166 	protected Class[] getExtraProxyInterfaces(Remote impl)
167 			throws NullPointerException {
168 		if (impl == null) {
169 			throw new NullPointerException("impl is null");
170 		}
171 
172 		List<Class> exposedInterfaces = new ArrayList<Class>();
173 		Class curr = null;
174 		Iterator it = serviceBeanMap.keySet().iterator();
175 		while (it.hasNext()) {
176 			curr = (Class) it.next();
177 			if (curr != null && !exposedInterfaces.contains(curr))
178 				exposedInterfaces.add(curr);
179 		}
180 		exposedInterfaces.add(Service.class);
181 		exposedInterfaces.add(RemoteMethodControl.class);
182 		exposedInterfaces.add(TrustEquivalence.class);
183 		// exposedInterfaces.add(net.jini.admin.Administrable.class);
184         return exposedInterfaces.toArray(new Class[exposedInterfaces.size()]);
185 	}

187 	/** {@inheritDoc} */
188 	protected InvocationDispatcher createInvocationDispatcher(
189 			Collection methods, Remote impl, ServerCapabilities caps)
190 			throws ExportException {
191 		// pass on the method of the interfaces so as to make the dispatcher
192 		// aware that our custom methods need to be included
193 		Method[] additionalMethods;
194 		Class curr = null;
195 		Iterator it = serviceBeanMap.keySet().iterator();
196 		while (it.hasNext()) {
197 			curr = (Class) it.next();
198 			additionalMethods = curr.getDeclaredMethods();
199 			for (int j = 0; j < additionalMethods.length; j++)
200 				methods.add(additionalMethods[j]);
201 		}
202 		return new SorcerInvocationDispatcher(methods, caps,
203 				getServerConstraints(), getPermissionClass(), getClassLoader());
204 	}

206 	/**
207 	 * Our custom dispatcher to be used for exporting SORCER service beans.
208 	 */
209 	private class SorcerInvocationDispatcher extends BasicInvocationDispatcher {
210 
211 		public SorcerInvocationDispatcher(Collection methods,
212 				ServerCapabilities serverCapabilities,
213 				MethodConstraints serverConstraints, Class permissionClass,
214 				ClassLoader loader) throws ExportException {
215 			super(methods, serverCapabilities, serverConstraints,
216 					permissionClass, loader);
217 		}
218 
219 		protected Object invoke(Remote impl, Method method, Object[] args,
220 				Collection context) throws Throwable {
221 			if (impl == null || args == null || context == null)
222 				throw new NullPointerException();
223 
224 			if (!method.isAccessible()
225 					&& !(Modifier.isPublic(method.getDeclaringClass()
226 							.getModifiers()) && Modifier.isPublic(method
227 							.getModifiers())))
228 				throw new IllegalArgumentException(
229 						"method not public or set accessible");
230 
231 			Class decl = method.getDeclaringClass();
232 			if (decl == ProxyTrust.class
233 					&& method.getName().equals("getProxyVerifier")
234 					&& impl instanceof ServerProxyTrust) {
235 				if (args.length != 0)
236 					throw new IllegalArgumentException("incorrect arguments");
237 
238 				return ((ServerProxyTrust) impl).getProxyVerifier();
239 			}
240 			Object obj = null;
241 			try {
242 				// handle context management by the containing provider
243 				if (decl == ContextManagement.class) {
244 					obj = method.invoke((ContextManagement) impl, args);
245 					return obj;
246 				}
247 				// Check if the invocation is to be made on provider's service
248 				// beans
249 				Object service = serviceBeanMap.get(method.getDeclaringClass());
250 				if (service != null) {
251 					obj = method.invoke(service, args);
252 				} else {
253 					obj = method.invoke(impl, args);
254 				}
255 			} catch (Throwable t) {
256 				throw new ExertionException(t);
257 			}
258 			return obj;
259 		}
260 	}
