132 	public static Invocation invoker(Mappable mappable, String path)
133 			throws ContextException {
134 		Object obj = mappable.asis(path);
135 		while (obj instanceof Mappable || obj instanceof Par) {
136 			try {
137 				obj = ((Evaluation) obj).asis();
138 			} catch (RemoteException e) {
139 				throw new ContextException(e);
140 			}
141 		}
142 		if (obj instanceof Invocation)
143 			return (Invocation) obj;
144 		else
145 			throw new NoneException("No such invoker at: " + path + " in: " + mappable.getName());
146 	}

257     public static Object invoke(ParModel parModel, String parname, Arg... parameters)
258             throws RemoteException, InvocationException {
259         try {
260             Object obj = parModel.asis(parname);
261             Context scope = null;
262             // assume that the first argument is always context if provided
263             if (parameters.length > 0 && parameters[0] instanceof Context)
264                 scope = (Context)parameters[0];
265             if (obj instanceof Par
266                     && ((Par) obj).asis() instanceof Invocation) {
267                 Invocation invoker = (Invocation) ((Par) obj).asis();
268                 //return invoker.invoke(parModel, parameters);
269                 if (scope != null)
270                     return invoker.invoke(scope, parameters);
271                 else
272                     return invoker.invoke(parModel, parameters);
273             } else if (obj instanceof Invocation) {
274                 if (scope != null)
275                     return ((Invocation) obj).invoke(scope, parameters);
276                 else
277                     return ((Invocation) obj).invoke(null, parameters);
278             } else if (obj instanceof Agent) {
279                 return ((Agent)obj).getValue(parameters);
280             }
281 
282             else {
283                 throw new InvocationException("No invoker for: " + parname);
284             }
285         } catch (ContextException e) {
286             throw new InvocationException(e);
287         }
288     }
