18  package sorcer.core.exertion;
19  
20  import java.net.MalformedURLException;
21  import java.rmi.server.RMIClassLoader;
22  import java.util.List;
23  
24  import sorcer.core.signature.MethodFactory;
25  import sorcer.core.signature.NetSignature;
26  import sorcer.service.Signature;
27  import sorcer.service.SignatureException;
28  
29  /**
30   * This class creates instances of appropriate subclasses of
31   * ServiceMethod. The appropriate subclass is determined by calling the
32   * exertion signatures.
33   */
34  public class ExertionMethodFactory implements MethodFactory {
35  	private static ExertionMethodFactory factory;
36  
37  	private ExertionMethodFactory() {
38  		// do nothing
39  	}
40  
41  	public static ExertionMethodFactory getFactory() {
42  		if (factory == null)
43  			factory = new ExertionMethodFactory();
44  
45  		return factory;
46  	}
47  
48  	/**
49  	 * This method returns an instnace of the appropriate subclass of
50  	 * RemmoteServiceMethod as determined from information provided by the given
51  	 * ServiceMethod instance.
52  	 * 
53  	 * @param method
54  	 *            The remote method that will be used to perform
55  	 *            EntryServiceTask execution
56  	 */
57  	public Signature createMethod(Signature method) throws SignatureException {
58  		if (method instanceof NetSignature)
59  			return method;
60  		String agentClass = ((NetSignature) method).getAgentClass();
61  		try {
62  			if (agentClass != null)
63  				return getRemoteMethod(method);
64  		} catch (Throwable ex) {
65  			ex.printStackTrace();
66  			throw new SignatureException(
67  					"Failed to create SORCER mobile method: " + method
68  							+ " as agent: " + agentClass);
69  		}
70  		return method;
71  	}
72  
73  	private NetSignature getRemoteMethod(Signature method)
74  			throws MalformedURLException, InstantiationException,
75  			ClassNotFoundException, IllegalAccessException {
76  		Class methodClass = RMIClassLoader.loadClass(
77  				((NetSignature) method).getAgentCodebase(),
78  				((NetSignature) method).getAgentClass());
79  		NetSignature rfm = (NetSignature) methodClass
80  				.newInstance();
81  		rfm.copySignature(method);
82  		return rfm;
83  	}
84  
85  	private NetSignature getRemoteMethod(NetSignature method,
86  			String codebase, String className) throws MalformedURLException,
87  			InstantiationException, ClassNotFoundException,
88  			IllegalAccessException {
89  		Class methodClass = RMIClassLoader.loadClass(codebase, className);
90  		NetSignature rfm = (NetSignature) methodClass
91  				.newInstance();
92  		rfm.copySignature(method);
93  		return rfm;
94  	}
95  
96  	public List<Signature> createMethods(List<Signature> signatures)
97  			throws SignatureException {
98  		for (int i = 0; i < signatures.size(); i++)
99  			signatures.set(i, createMethod(signatures.get(i)));
100 
101 		return signatures;
102 	}
103 }
