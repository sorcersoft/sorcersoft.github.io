135 	public static List<Exertion> getInputExertions(Job job) throws ContextException {
136 		if (job == null || job.size() == 0)
137 			return null;
138 		List<Exertion> exertions = new ArrayList<Exertion>();
139 		Exertion master = job.getMasterExertion();
140 		for (int i = 0; i < job.size(); i++)
141 			if (!(job.get(i).equals(master) || job
142 					.getControlContext().isSkipped(job.get(i))))
143 				exertions.add(job.get(i));
144 		return exertions;
145 	}

230 	public static void replaceNullIDs(Exertion ex) throws ContextException {
231 		if (ex == null)
232 			return;
233 		if (ex.isJob()) {
234 			Job job = (Job) ex;
235 			if (job.getId() == null)
236 				job.setId(getId());
237 			if (job.getContext().getId() == null)
238 				job.getContext().setId(getId());
239 			for (int i = 0; i < job.size(); i++)
240 				replaceNullIDs(job.get(i));
241 		} else
242 			replaceNullIDs((ServiceExertion) ex);
243 	}

268 			throws ExertionException {
269 		if (ex == null)
270 			return null;
271 		else if (ex.getProcessSignature() != null)
272 			throw new ExertionException("No Method For Exertion e=" + ex);
273 
274 		ExertionEnvelop eenv = ExertionEnvelop.getTemplate();
275 		eenv.serviceType = ((NetSignature) ex.getProcessSignature()).getServiceType();
276 		eenv.providerName = ex.getProcessSignature().getProviderName();
277 		eenv.exertion = ex;
278 		eenv.exertionID = ex.getId();
279 		eenv.isJob = new Boolean(ex.isJob());
280 		eenv.state = new Integer(Exec.INITIAL);
281 		return eenv;
