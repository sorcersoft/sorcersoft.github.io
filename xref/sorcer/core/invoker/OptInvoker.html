18  package sorcer.core.invoker;
19  
20  import java.rmi.RemoteException;
21  
22  import sorcer.service.*;
23  
24  /**
25   * The option Var. There is a single target invoker that executes if the condition is true (like if... then).
26   * 
27   * @author Mike Sobolewski
28   */
29  public class OptInvoker<T> extends ServiceInvoker<T> {
30  	
31  	protected Condition condition;
32  	
33  	protected ServiceInvoker<T> target;
34  	
35  	public OptInvoker(String name) {
36  		super(name);
37  	}
38  		
39  	public OptInvoker(String name, ServiceInvoker<T> invoker) {
40  		super(name);
41  		this.condition = new Condition(true);
42  		this.target = invoker;
43  	}
44  	
45  	public OptInvoker(String name, Condition condition, ServiceInvoker<T> invoker) {
46  		super(name);
47  		this.condition = condition;
48  		this.target = invoker;
49  	}
50  
51  	public ServiceInvoker<T> getTarget() {
52  		return target;
53  	}
54  
55  	public void setTarget(ServiceInvoker<T> target) {
56  		this.target = target;
57  	}
58  
59  	@Override
60  	public T getValue(Arg... entries) throws EvaluationException, RemoteException {
61  		checkInvokeContext();
62  		try {
63  			if (condition.isTrue())
64  				return target.getValue(entries);
65  			else 
66  				return null;
67  		} catch (ContextException e) {
68  			throw new InvocationException(e);
69  		}
70  	}
71  	
72  	public T invoke(Arg... entries) throws RemoteException,
73  			InvocationException {
74  		checkInvokeContext();
75  		try {
76  			if (condition.isTrue())
77  				return target.invoke(entries);
78  			else 
79  				return null;
80  		} catch (ContextException e) {
81  			throw new InvocationException(e);
82  		}
83  	}
84  
85  	public T invoke(Context context, Arg... entries)
86  			throws RemoteException, InvocationException {
87  		checkInvokeContext();
88  		try {
89  			if (condition.isTrue())
90  				return target.invoke(context, entries);
91  			else 
92  				return null;
93  		} catch (ContextException e) {
94  			throw new InvocationException(e);
95  		}
96  	}
97  	
98  	private void checkInvokeContext() throws RemoteException {
99  		if (target.getScope().size() == 0 && invokeContext.size() > 0) {
100 			target.setScope(invokeContext);
101 		}
102 	}
103 }
