104 	public CmdInvoker(String argarray[], File script, boolean background,
105 			InputStream stdin, File logFile, Par... pars) {
106 		cmdarray = new String[argarray.length + 1];
107 		this.scriptFile = script;
108 		this.pars = new ArgSet(pars);
109 		if (!scriptFile.canExecute()) {
110 			scriptFile.setExecutable(true);
111 			logger.warn("script exec permission corrected for: " + scriptFile);
112 		}
113 
114 		System.arraycopy(argarray, 0, cmdarray, 0, argarray.length);
115 
116 		cmdarray[cmdarray.length - 1] = scriptFile.getAbsolutePath();
117 		this.stdin = stdin;
118 		this.logFile = logFile;
119 		this.background = background;

153 	@Override
154 	public CmdResult getValue(Arg... entries) throws EvaluationException,
155 			RemoteException {
156 		CmdResult out = null;
157 		if (scriptFile != null) {
158 			try {
159 				return execScript();
160 			} catch (Exception se) {
161 				throw new EvaluationException("Script invocation failed: "
162 						+ cmd);
163 			}
164 		} else {
165 			try {
166 				if (cmd == null && cmdarray == null)
167 					throw new EvaluationException("No args for CmdEvaluator!");
168 
169 				if (cmd != null) {
170 					if (stdin != null) {
171 						out = ExecUtils.execCommand(Runtime.getRuntime().exec(
172 								cmd), stdin);
173 					} else {
174 						out = ExecUtils.execCommand(cmd);
175 					}
176 				} else if (cmdarray != null) {
177 					if (stdin != null) {
178 						out = ExecUtils.execCommand(Runtime.getRuntime().exec(
179 								cmdarray), stdin);
180 					} else {
181 						out = ExecUtils.execCommand(cmdarray);
182 					}
183 				}
184 			} catch (Exception e) {
185 				e.printStackTrace();
186 				throw new EvaluationException("Command invocation failed: "
187 						+ Arrays.toString(cmdarray));
188 			}
189 		}
190 		return out;
191 	}

193 	public CmdResult execScript() throws IOException, InterruptedException {
194 		if (cmdarray != null) {
195 			StringBuilder sb = new StringBuilder(cmdarray[0]);
196 			for (int i = 1; i < cmdarray.length; i++)
197 				sb.append(" ").append(cmdarray[i]);
198 			if (background)
199 				sb.append(" &");
200 			cmd = sb.toString();
201 		} else {
202 			cmd = cmd + " " + scriptFile.getAbsolutePath();
203 			if (background)
204 				cmd = cmd + " &";
205 		}
206 
207 		logger.info("executing script: " + cmd);
208 
209 		final Process process = Runtime.getRuntime().exec(cmd);				
210 		final PrintWriter logOut = new PrintWriter(logFile);
211 		Thread scriptLogger = new Thread(new Runnable() {
212 			public void run() {
213 				String line;
214 				BufferedReader in = new BufferedReader(new InputStreamReader(
215 						process.getInputStream()));
216 				try {
217 					while ((line = in.readLine()) != null) {
218 						logOut.println(line);
219 						if (logOut.checkError()) {
220 							System.err
221 									.println("scipt exec log file encountered check error"
222 											+ logFile);
223 						}
224 						logOut.flush();
225 					}
226 				} catch (IOException e) {
227 					System.err
228 							.println("scipt exec log file encountered IO error"
229 									+ logFile);
230 				}
231 			}
232         }, tName("STDIN-" + process));
233 		scriptLogger.start();
234 		CmdResult result = null;
235 		if (stdin != null)
236 			result = ExecUtils.execCommand(process, stdin);
237 		else
238 			result = ExecUtils
239 					.execCommand(process, new NullInputStream());
240 
241 		logOut.close();
242 		logger.info(Arrays.toString(cmdarray) + " completed with status = "
243 				+ process.exitValue());
244 		return result;
245 	}
