44  	public void dispatchExertions() throws ExertionException,
45  			SignatureException {
46          checkAndDispatchExertions();
47  		try {
48  			reconcileInputExertions(xrt);
49  		} catch (ContextException ex) {
50  			throw new ExertionException(ex);
51  		}
52  
53  		logger.debug("exertion count: " + inputXrts.size());
54  		for (int i = 0; i < inputXrts.size(); i++) {
55  			ServiceExertion exertion = (ServiceExertion) inputXrts
56  					.get(i);
57  			logger.debug("exertion #: " + i + ", exertion:\n" + exertion);
58  			try {
59  				writeEnvelop(exertion);
60  				logger.debug("generateTasks ==> SPACE SEQUENIAL EXECUTE EXERTION: "
61                          + exertion.getName());
62  			}  catch (ProvisioningException pe) {
63                  xrt.setStatus(FAILED);
64                  throw new ExertionException(pe.getLocalizedMessage());
65              } catch (RemoteException re) {
66  				logger.warn("Space not reachable....resetting space", re);
67  				space = SpaceAccessor.getSpace();
68  				if (space == null) {
69  					xrt.setStatus(FAILED);
70  					throw new ExertionException("NO exertion space available!");
71  				}
72  				if (masterXrt != null) {
73  					try {
74  						writeEnvelop(masterXrt);
75  					} catch (Exception e) {
76  						e.printStackTrace();
77  						xrt.setStatus(FAILED);
78  						throw new ExertionException(
79  								"Wrting master exertion into exertion space failed!",
80  								e);
81  					}
82  				}
83  			}
84  			logger.debug("waiting for exertion " + i + ", id="
85                      + exertion.getId() + "...");
86  			waitForExertion(exertion.getIndex());
87  		}
88  		dThread.stop = true;
89  	}

91  	public void collectResults() throws ExertionException, SignatureException {
92  		int count = 0;
93  		ExertionEnvelop temp;
94  		temp = ExertionEnvelop.getTemplate();
95  		temp.parentID = xrt.getId();
96  		temp.state = new Integer(DONE);
97  
98  		logger.debug("collect exertions for template: \n"
99                  + temp.describe());
100 		while (count < inputXrts.size() && state != FAILED) {
101 			ExertionEnvelop resultEnvelop = takeEnvelop(temp);
102 			logger.debug("collected result envelope "
103                     + (resultEnvelop != null ? resultEnvelop.describe() : "NULL"));
104 
105 			if (resultEnvelop != null && resultEnvelop.exertion != null) {
106 				ServiceExertion input = (ServiceExertion) ((NetJob)xrt)
107 						.get(resultEnvelop.exertion
108 								.getIndex());
109 				ServiceExertion result = (ServiceExertion) resultEnvelop.exertion;
110 				postExecExertion(input, result);
111 				count++;
112 			}
113 		}
114 		executeMasterExertion();
115 		dispatchers.remove(xrt.getId());
116 		state = DONE;
117 	}
