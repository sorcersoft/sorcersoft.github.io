50      protected Exertion execExertion(Exertion ex) throws SignatureException,
51              ExertionException {
52          beforeExec(ex);
53          // set subject before task goes out.
54          // ex.setSubject(subject);
55          ServiceExertion result = null;
56          try {
57  			if (ex.isTask()) {
58  				result = execTask((Task) ex);
59  			} else if (ex.isJob()) {
60  				result = execJob((Job) ex);
61  			} else if (ex.isBlock()) {
62  				result = execBlock((Block) ex);
63  			} else {
64  				logger.warn("Unknown ServiceExertion: {}", ex);
65  			}
66              afterExec(ex, result);
67  		} catch (Exception e) {
68              logger.warn("Error while executing exertion");
69  			// return original exertion with exception
70  			result = (ServiceExertion) ex;
71              result.reportException(e);
72  			result.setStatus(FAILED);
73  			setState(Exec.FAILED);
74  			return result;
75  		}
76  		// set subject after result is received
77  		// result.setSubject(subject);
78  		return result;
79      }

118     protected Task execServiceTask(Task task) throws ExertionException {
119         logger.info("Starting execServiceTask for: " + task.getName());
120         Task result = null;
121         int maxTries = 6;
122         int tried=0;
123         try {
124             if (((NetSignature) task.getProcessSignature()).getService()!=null) {
125                 logger.info("\n*** service is set in signature testing if it is the same provider ***\n");
126                 try {
127                     if (((NetSignature) task.getProcessSignature()).getService().equals(provider)) {
128                         logger.info("\n*** getting result from delegate of "
129                                 + provider.getProviderName() + "... ***\n");
130                         result = ((ServiceProvider) provider).getDelegate().doTask(
131                                 task, null);
132                         result.getControlContext().appendTrace(
133                                 "delegate of: " + this.provider.getProviderName()
134                                         + "=>" + this.getClass().getName());
135                         return result;
136                     }
137                 } catch (RemoteException re) {
138                     logger.warn("Got exception trying to run using the provider in the signature: " + re.getMessage());
139                     re.printStackTrace();
140                 }
141             }
142 
143             NetSignature sig = (NetSignature) task.getProcessSignature();
144             // Catalog lookup or use Lookup Service for the particular
145             // service
146             Service service = (Service) Accessor.getService(sig);
147             if (service == null && task.isProvisionable()) {
148                 Provisioner provisioner = Accessor.getService(Provisioner.class);
149                 if (provisioner != null) {
150                     try {
151                         logger.info("Provisioning "+sig);
152                         service = provisioner.provision(sig.getServiceType().getName(), sig.getProviderName(), sig.getVersion());
153                     } catch (ProvisioningException pe) {
154                         String msg = "Problem provisioning " + sig + " in task " + task.getName() + ": " +pe.getMessage();
155                         logger.warn(msg, pe);
156                         throw new ExertionException(msg, task);
157                     } catch (RemoteException re) {
158                         String msg = "Problem provisioning " + sig + " in task " + task.getName() + ": " +re.getMessage();
159                         logger.warn(msg, re);
160                         throw new ExertionException(msg, task);
161                     }
162                 }
163             }
164             if (service == null) {
165                 String msg;
166                 // get the PROCESS Method and grab provider name + interface
167                 msg = "No Provider Available\n" + "Provider Name:      "
168                         + sig.getProviderName() + "\n"
169                         + "Provider Interface: " + sig.getServiceType();
170 
171                 logger.info(msg);
172                 throw new ExertionException(msg, task);
173             } else {
174                 tried=0;
175                 while (result==null && tried < maxTries) {
176                     tried++;
177                     try {
178                         // setTaskProvider(task, provider.getProviderName());
179                         task.setService(service);
180                         // client security
181                         /*
182                          * ClientSubject cs = null;
183                          * try{ // //cs =

185                          * (ClientSubject.class); }catch (Exception ex){
186                          * Util.debug(this, ">>>No Subject in the server call");
187                          * cs=null; } Subject client = null; if(cs!=null){

189                          * "Abhijit::>>>>> CS was not null"); if(client!=null){

191                          * null"+client); }else{ Util.debug(this,"Abhijit::>>>>>>
192                          * CLIENT SUBJECT WAS
193                          * NULL!!"); } }else{ Util.debug(this, "OOPS! NULL CS"); }
194                          * if(client!=null&&task.getPrincipal()!=null){
195                          * Util.debug(this,"Abhijit:: >>>>>--------------Inside
196                          * Client!=null, PRINCIPAL != NULL, subject="+client);
197                          * result = (RemoteServiceTask)provider.service(task);
198                          * }else{ Util.debug(this,"Abhijit::
199                          * >>>>>--------------Inside null Subject"); result =
200                          * (RemoteServiceTask)provider.service(task); }
201                          */
202                         logger.debug("getting result from provider...");
203                         result = (Task) service.service(task, null);
204 
205                     } catch (Exception re) {
206                         if (tried >= maxTries) {
207                             System.out.println("+++++++++++++++Problem exerting task, already tried " + tried + " times for: " + xrt.getName() + " " + re.getMessage());
208                             logger.error("+++++++++++++++Problem exerting task, already tried " + tried + " times for: " + xrt.getName() + " " + re.getMessage());
209                             throw re;
210                         }
211                         else {
212                             System.out.println("+++++++++++++++Problem exerting task, retrying " + tried + " time: " + xrt.getName() + " " + re.getMessage());
213                             logger.info("Problem exerting task, retrying " + tried + " time: " + xrt.getName() + " " + re.getMessage());
214                             service = (Service) Accessor.getService(sig);
215                             try {
216                                 System.out.println("+++++++++++++++Got service: " + ((Provider)service).getProviderID());
217                                 logger.info("+++++++++++++++Got service: " + ((Provider)service).getProviderID());
218                             } catch (Exception e) {
219                                 System.out.println("+++++++++++++++The service we got is not valid");
220                             }
221                         }
222                     }
223                 }
224                 if (result!=null)
225                     result.getControlContext().appendTrace(
226                                ((Provider)service).getProviderName() + " dispatcher: "
227                                         + getClass().getName());
228             }
229             logger.debug("got result: {}", result);
230         } catch (Exception re) {
231             System.out.println("+++++++++++++++Dispatcher failed for task, tried: " + tried + " : "

233             logger.error("+++++++++++++++Dispatcher failed for task, tried: " + tried + " : "
234                     + xrt.getName());
235             task.reportException(re);
236             throw new ExertionException("Dispatcher failed for task, tried: " + tried + " : "
237                     + xrt.getName(), re);
238         }
239         return result;
240     }

242     protected Job execJob(Job job)
243             throws DispatcherException, InterruptedException,
244             RemoteException {
245 
246         runningExertionIDs.add(job.getId());
247 
248         // create a new instance of a dispatcher
249         Dispatcher dispatcher = ExertionDispatcherFactory.getFactory()
250                 .createDispatcher(job, sharedContexts, true, provider);
251         dispatcher.exec();
252         // wait until serviceJob is done by dispatcher

254         // Not sure if good place

262 	private Block execBlock(Block block)
263 			throws DispatcherException, InterruptedException,
264 			ExertionException, RemoteException {
265 
266 		try {
267 			ServiceTemplate st = Accessor.getServiceTemplate(null,
268 					null, new Class[] { Concatenator.class }, null);
269 			ServiceItem[] concatenators = Accessor.getServiceItems(st, null,
270 					SorcerEnv.getLookupGroups());
271 			/*
272 			 * check if there is any available concatenator in the network and
273 			 * delegate the inner block to the available Concatenator. In the future, a
274 			 * efficient load balancing algorithm should be implemented for
275 			 * dispatching inner jobs. Currently, it only does round robin.
276 			 */
277 			for (int i = 0; i < concatenators.length; i++) {
278 				if (concatenators[i] != null) {
279 					if (!provider.getProviderID().equals(
280 							concatenators[i].serviceID)) {
281 						logger.trace("Concatenator: " + i + " ServiceID: "
282 								+ concatenators[i].serviceID);
283 						Provider rconcatenator = (Provider) concatenators[i].service;
284 
285 						return (Block) rconcatenator.service(block, null);
286 					}
287 				}
288 			}
289 
290 			/*
291 			 * Create a new dispatcher thread for the inner job, if no available
292 			 * Jobber is found in the network
293 			 */
294 			Dispatcher dispatcher;
295 			runningExertionIDs.add(block.getId());
296 
297 			// create a new instance of a dispatcher
298 			dispatcher = ExertionDispatcherFactory.getFactory()
299 					.createDispatcher(block, sharedContexts, true, provider);
300             dispatcher.exec();
301 			// wait until serviceJob is done by dispatcher
302 			Block out = (Block) dispatcher.getResult().exertion;
303 			out.getControlContext().appendTrace(provider.getProviderName() 
304 					+ " dispatcher: " + getClass().getName());
305 			return out;
306 		} catch (RemoteException re) {
307 			re.printStackTrace();
308 			throw re;
309 		} catch (ExertionException ee) {
310 			ee.printStackTrace();
311 			throw ee;
312 		} catch (DispatcherException de) {
313 			de.printStackTrace();
314 			throw de;
315 		} catch (TransactionException te) {
316 			te.printStackTrace();
317 			throw new ExertionException("transaction failure", te);
318 		}
319 	}
