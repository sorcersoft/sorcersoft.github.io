45  	public void dispatchExertions() throws ExertionException,
46  			SignatureException {
47  		reconcileInputExertions(xrt);
48  
49  		int index = inputXrts.size() - 1;
50  		ServiceExertion exertion = null;
51  		try {
52  			while (index >= 0) {
53  				exertion = (ServiceExertion) inputXrts.elementAt(index);
54  				logger
55  						.info("generateTasks ==> SPACE PARALLEL EXECUTE EXERTION: "
56  								+ exertion.getName());
57  				writeEnvelop(exertion);
58  				index--;
59  			}
60  			dThread.stop = true;
61  		} catch (Exception re) {
62  			re.printStackTrace();
63  			logger.severe("Space not reachable....resetting space");
64  			space = SpaceAccessor.getSpace();
65  			if (space == null) {
66  				xrt.setStatus(FAILED);
67  				throw new ExertionException("NO exertion space available!");
68  			}
69  			if (masterXrt != null) {
70  				try {
71  					writeEnvelop(masterXrt);
72  				} catch (Exception e) {
73  					e.printStackTrace();
74  					xrt.setStatus(FAILED);
75  					throw new ExertionException(
76  							"Wrting master exertion into exertion space failed!",
77  							e);
78  				}
79  			}
80  		}
81  	}

83  	public void collectResults() throws ExertionException, SignatureException {
84  		int count = 0;
85  		// get all children of the underlying parent job
86  		ExertionEnvelop template = ExertionEnvelop.getTakeTemplate(xrt.getId(),
87  				null);
88  		logger
89  				.finer("<===================== collect exertions for template: \n"
90  						+ template.describe());
91  		while (count < inputXrts.size() && state != FAILED) {
92  			ExertionEnvelop resultEnvelop = (ExertionEnvelop) takeEnvelop(template);
93  			if (resultEnvelop != null && resultEnvelop.exertion != null) {
94  				ServiceExertion input = (ServiceExertion) ((NetJob)xrt)
95  						.exertionAt(((ServiceExertion) resultEnvelop.exertion)
96  								.getIndex());
97  				logger
98  						.finer("collected result envelope  <===================== \n"
99  								+ resultEnvelop.describe());
100 				ServiceExertion result = (ServiceExertion) resultEnvelop.exertion;
101 				postExecExertion(input, result);
102 				count++;
103 			} else {
104 				logger.finest("continue for envelop: " + resultEnvelop);
105 				continue;
106 			}
107 		}
108 		executeMasterExertion();
109 		dispatchers.remove(xrt.getId());
110 		state = DONE;
111 	}
