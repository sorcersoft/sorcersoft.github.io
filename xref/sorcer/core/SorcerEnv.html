113 	 * @return Sorcer Local jar repo location 
114 	 */
115 	public static String getRepoDir() {
116 		String repo = props.getProperty(SorcerConstants.S_SORCER_REPO);
117         if (repo!=null) repo = repo.trim();
118 		
119 		if (repo != null && !repo.isEmpty()) {
120 			try {
121 				File repoDir = new File(repo);
122 				if (repoDir.exists() && repoDir.isDirectory())
123 					return repo;
124 			} catch (Throwable t) {
125 				logger.throwing(
126 						SorcerEnv.class.getName(),
127 						"The given Sorcer Jar Repo Location: " + repo + " does not exist or is not a directory!",
128 						t);
129 			}			
130 		} else {
131 			// Fall back to default location in user's home/.m2
132 			try {
133 				File repoDir = new File(System.getProperty("user.home")+"/.m2/repository");
134                 if (repoDir.exists() && repoDir.isDirectory())
135 					return repoDir.getAbsolutePath();
136                 else {
137                     FileUtils.forceMkdir(repoDir);
138                     return repoDir.getAbsolutePath();
139                 }
140             } catch (Throwable t) {
141 				logger.throwing(
142 						SorcerEnv.class.getName(),
143 						"The given Sorcer Jar Repo Location: " + repo + " does not exist or is not a directory!",
144 						t);
145 			}						
146 		}
147 		return null;
148 	}

158 		// Try and load from path given in system properties
159 		String envFile = null;
160 		props = new Properties();
161 
162 		try {

192 	 * Tries to load properties from a <code>filename</code>, first in a local
193 	 * directory. If there is no file in the local directory then load the file
194 	 * from the classpath at sorcer/util/sorcer.env.
195 	 * 
196 	 * @throws IOException
197 	 * @throws ConfigurationException
198 	 */
199 	public static Properties loadProperties(String filename)
200 			throws ConfigurationException {
201 		Properties properties = new Properties();
202 		try {
203 			// Try in user home directory first
204 			properties.load((new FileInputStream(new File(filename))));
205 			logger.fine("loaded properties from: " + filename);
206 
207 		} catch (Exception e) {
208 			try {
209 				// try to look for sorcer.env in SORCER_HOME/configs
210 				properties.load((new FileInputStream(new File(System.getenv("SORCER_HOME")+"/configs/" + filename))));
211 				logger.fine("loaded properties from: " + System.getenv("SORCER_HOME") +"/configs/" + filename);
212 			} catch (Exception ee) {
213 				try {
214 					// No file give, try as resource sorcer/util/sorcer.env
215 					InputStream stream = SorcerEnv.class.getResourceAsStream(filename);
216 					if (stream != null)
217 						properties.load(stream);
218 					else
219 						logger.severe("could not load properties as Sorcer resource file>"
220 								+ filename + "<" );
221 				} catch (Throwable t2) {
222 					throw new ConfigurationException(e);
223 				}
224 			}
225 		}
226 		reconcileProperties(properties);
227 
228 		return properties;
229 	}

232 			throws ConfigurationException {
233 
234 		update(properties);
235 
236 		// set the document root for HTTP server either for provider or
237 		// requestor
238 
239 		String rootDir = null, dataDir = null;
240 		rootDir = properties.getProperty(P_DATA_ROOT_DIR);
241 		dataDir = properties.getProperty(P_DATA_DIR);
242 		if (rootDir != null && dataDir != null) {
243 			System.setProperty(DOC_ROOT_DIR, rootDir + File.separator + dataDir);
244 		} else {
245 			rootDir = properties.getProperty(R_DATA_ROOT_DIR);
246 			dataDir = properties.getProperty(R_DATA_DIR);
247 			if (rootDir != null && dataDir != null) {
248 				System.setProperty(DOC_ROOT_DIR, rootDir + File.separator
249 						+ dataDir);
250 			}
251 		}
252 		dataDir = properties.getProperty(P_SCRATCH_DIR);
253 		if (dataDir != null) {
254 			System.setProperty(SCRATCH_DIR, dataDir);
255 		} else {
256 			dataDir = properties.getProperty(R_SCRATCH_DIR);
257 			if (dataDir != null) {
258 				System.setProperty(SCRATCH_DIR, dataDir);
259 			}
260 		}
261 
262 		String httpInterface = null, httpPort = null;
263 		httpInterface = properties.getProperty(P_DATA_SERVER_INTERFACE);
264 		httpPort = properties.getProperty(P_DATA_SERVER_PORT);
265 		if (httpInterface != null) {
266 			System.setProperty(DATA_SERVER_INTERFACE, httpInterface);
267 			System.setProperty(DATA_SERVER_PORT, httpPort);
268 		} else {
269 			httpInterface = properties.getProperty(R_DATA_SERVER_INTERFACE);

272 				System.setProperty(DATA_SERVER_INTERFACE, httpInterface);
273 				System.setProperty(DATA_SERVER_PORT, httpPort);

275 		}

303 	 * Overwrites defined properties in sorcer.env (sorcer.home,
304 	 * provider.webster.interface, provider.webster.port) with those defined as
305 	 * JVM system properties.
306 	 * 
307 	 * @param properties
308 	 * @throws ConfigurationException
309 	 */
310 	private static void update(Properties properties)
311 			throws ConfigurationException {
312 		Enumeration<?> e = properties.propertyNames();
313 		String key, value, evalue = null;
314 		String pattern = "${" + "localhost" + "}";
315 //		String userDirPattern = "${user.home}";
316 		// first substitute for this localhost
317 		while (e.hasMoreElements()) {
318 			key = (String) e.nextElement();
319 			value = properties.getProperty(key);
320 			if (value.equals(pattern)) {
321 				try {
322 					value = getHostAddress();
323 				} catch (UnknownHostException ex) {
324 					ex.printStackTrace();
325 				}
326 				properties.put(key, value);
327 			}
328 	/*		if (value.equals(userDirPattern)) {
329 				value = System.getProperty("user.home");				
330 				properties.put(key, value);
331 			}*/
332 		}
333 		// now substitute other entries accordingly 
334 		e = properties.propertyNames();
335 		while (e.hasMoreElements()) {
336 			key = (String) e.nextElement();
337 			value = properties.getProperty(key);
338 			evalue = expandStringProperties(value, true);
339 			// try SORCER env properties
340 			if (evalue == null)
341 				evalue = expandStringProperties(value, false);
342 			if (evalue != null)
343 				properties.put(key, evalue);
344 			if (value.equals(pattern)) {
345 				try {
346 					evalue = getHostAddress();
347 				} catch (UnknownHostException e1) {
348 					e1.printStackTrace();
349 				}
350 				properties.put(key, evalue);
351 			}
352 		}
353 	}

365 	/**
366 	 * Expands properties embedded in a string with ${some.property.name}. Also
367 	 * treats ${/} as ${file.separator}.
368 	 */
369 	public static String expandStringProperties(String value,
370 			boolean isSystemProperty) throws ConfigurationException {
371 		int p = value.indexOf("${", 0);
372 		if (p == -1) {
373 			return value;
374 		}
375 		int max = value.length();
376 		StringBuffer sb = new StringBuffer(max);
377 		int i = 0; /* Index of last character we copied */
378 		while (true) {
379 			if (p > i) {
380 				/* Copy in anything before the special stuff */
381 				sb.append(value.substring(i, p));
382 				i = p;
383 			}
384 			int pe = value.indexOf('}', p + 2);
385 			if (pe == -1) {
386 				/* No matching '}' found, just add in as normal text */
387 				sb.append(value.substring(p, max));
388 				break;
389 			}
390 			String prop = value.substring(p + 2, pe);
391 			if (prop.equals("/")) {
392 				sb.append(File.separatorChar);
393 			} else {
394 				try {
395 					String val = null;
396 					if (isSystemProperty) {
397 						val = prop.length() == 0 ? null : System
398 								.getProperty(prop);
399 						if (val == null) {
400 							// try System env
401 							val = prop.length() == 0 ? null : System
402 									.getenv(prop);
403 						}
404 					} else {
405 						val = prop.length() == 0 ? null : props
406 								.getProperty(prop);
407 					}
408 					if (val != null) {
409 						sb.append(val);
410 					} else {
411 						return null;
412 					}
413 				} catch (SecurityException e) {
414 					throw new ConfigurationException(e);
415 				}
416 			}
417 			i = pe + 1;
418 			p = value.indexOf("${", i);
419 			if (p == -1) {
420 				/* No more to expand -- copy in any extra. */
421 				if (i < max) {
422 					sb.append(value.substring(i, max));
423 				}
424 				break;
425 			}
426 		}
427 		return sb.toString();
428 	}
