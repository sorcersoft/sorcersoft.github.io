215     static ServiceElement create(final ServiceDetails serviceDetails,
216         final ServiceDeployment deployment,
217         String configurationFilePath) throws IOException {
218             ServiceElement service = new ServiceElement();
219 
220             String websterUrl;
221             if(serviceDetails.webster==null) {
222                 if(deployment.getWebsterUrl()==null) {
223                     websterUrl = Sorcer.getWebsterUrl();
224                     logger.debug("Set code base derived from Sorcer.getWebsterUrl: "+websterUrl);
225                 } else {
226                     websterUrl = deployment.getWebsterUrl();
227                     logger.debug("Set code base derived from Deployment: "+websterUrl);
228                 }
229             } else {
230                 websterUrl = serviceDetails.webster;
231             }
232         /* Create client (export) ClassBundle */
233             List<ClassBundle> exports = new ArrayList<ClassBundle>();
234             for(String s : serviceDetails.interfaces) {
235                 ClassBundle export = new ClassBundle(s);
236                 if(serviceDetails.codebaseJars.length==1 && Artifact.isArtifact(serviceDetails.codebaseJars[0])) {
237                     export.setArtifact(serviceDetails.codebaseJars[0]);
238                 } else {
239                     export.setJARs(appendJars(commonDLJars, serviceDetails.codebaseJars));
240                     export.setCodebase(websterUrl);
241                 }
242                 exports.add(export);
243             }
244 
245 		/* Create service implementation ClassBundle */
246             ClassBundle main = new ClassBundle(serviceDetails.providerClass==null?deployment.getImpl():serviceDetails.providerClass);
247             if(serviceDetails.implJars.length==1 && Artifact.isArtifact(serviceDetails.implJars[0])) {
248                 main.setArtifact(serviceDetails.implJars[0]);
249             } else {
250                 main.setJARs(serviceDetails.implJars);
251                 main.setCodebase(websterUrl);
252             }
253 
254 		/* Set ClassBundles to ServiceElement */
255             service.setComponentBundle(main);
256             service.setExportBundles(exports.toArray(new ClassBundle[exports.size()]));
257 
258             String serviceName;
259             if(serviceDetails.name==null) {
260 		    /* Get the (simple) name from the fully qualified interface */
261                 if(deployment.getName()==null) {
262                     StringBuilder nameBuilder = new StringBuilder();
263                     for(String s : serviceDetails.interfaces) {
264                         String value;
265                         int ndx = s.lastIndexOf(".");
266                         if (ndx > 0) {
267                             value = s.substring(ndx + 1);
268                         } else {
269                             value = s;
270                         }
271                         if(nameBuilder.length()>0) {
272                             nameBuilder.append(" | ");
273                         }
274                         nameBuilder.append(value);
275                     }
276                     serviceName = nameBuilder.toString();
277                 } else {
278                     serviceName = deployment.getName();
279                 }
280             } else {
281                 serviceName = serviceDetails.name;
282             }
283 
284             if(serviceDetails.maxPerNode>0) {
285                 service.setMaxPerMachine(serviceDetails.maxPerNode);
286             }
287             if(serviceDetails.architecture!=null || serviceDetails.operatingSystems.length>0) {
288                 ServiceLevelAgreements slas = new ServiceLevelAgreements();
289                 SystemRequirements systemRequirements = new SystemRequirements();
290                 if(serviceDetails.architecture!=null) {
291                     Map<String, Object> attributeMap = new HashMap<String, Object>();
292                     attributeMap.put(ProcessorArchitecture.ARCHITECTURE, serviceDetails.architecture);
293                     SystemComponent systemComponent = new SystemComponent("Processor",
294                             ProcessorArchitecture.class.getName(),
295                             attributeMap);
296                     systemRequirements.addSystemComponent(systemComponent);
297                 }
298                 for(String s : serviceDetails.operatingSystems) {
299                     String opSys = checkAndMaybeFixOpSys(s);
300                     Map<String, Object> attributeMap = new HashMap<String, Object>();
301                     attributeMap.put(OperatingSystem.NAME, opSys);
302                     SystemComponent operatingSystem =
303                             new SystemComponent("OperatingSystem", OperatingSystem.class.getName(), attributeMap);
304                     systemRequirements.addSystemComponent(operatingSystem);
305                 }
306                 slas.setServiceRequirements(systemRequirements);
307                 service.setServiceLevelAgreements(slas);
308             }
309             if(serviceDetails.ips.length>0) {
310                 SystemRequirements systemRequirements = service.getServiceLevelAgreements().getSystemRequirements();
311                 systemRequirements.addSystemComponent(getSystemComponentAddresses(false, serviceDetails.ips));
312             }
313 
314             if(serviceDetails.excludeIps.length>0) {
315                 SystemRequirements systemRequirements = service.getServiceLevelAgreements().getSystemRequirements();
316                 systemRequirements.addSystemComponent(getSystemComponentAddresses(true, serviceDetails.excludeIps));
317             }
318 
319 		/* Create simple ServiceBeanConfig */
320             Map<String, Object> configMap = new HashMap<String, Object>();
321             configMap.put(ServiceBeanConfig.NAME, serviceName);
322             configMap.put(ServiceBeanConfig.GROUPS, Sorcer.getLookupGroups());
323             ServiceBeanConfig sbc = new ServiceBeanConfig(configMap, new String[]{configurationFilePath});
324             sbc.addAdditionalEntries(new DeployInfo(deployment.getType().name(), deployment.getUnique().name(), deployment.getIdle()));
325             service.setServiceBeanConfig(sbc);
326             service.setPlanned(deployment.getMultiplicity());
327 
328         /* If the service is to be forked, create an ExecDescriptor */
329             if(serviceDetails.fork) {
330                 service.setFork(true);
331                 if(serviceDetails.jvmArgs!=null) {
332                     ExecDescriptor execDescriptor = new ExecDescriptor();
333                     execDescriptor.setInputArgs(serviceDetails.jvmArgs);
334                     service.setExecDescriptor(execDescriptor);
335                 }
336             }
337             logger.debug("Generated Service Element :"+service);
338             return service;
339         }

341         private static SystemComponent[] getSystemComponentAddresses(boolean exclude, String[] addresses) {
342             List<SystemComponent> machineAddresses = new ArrayList<SystemComponent>();
343             for(String ip : addresses) {
344                 SystemComponent machineAddress = new SystemComponent(TCPConnectivity.ID);
345                 machineAddress.setExclude(exclude);
346                 if(isIpAddress(ip)) {
347                     machineAddress.put(TCPConnectivity.HOST_ADDRESS, ip);
348                 } else {
349                     machineAddress.put(TCPConnectivity.HOST_NAME, ip);
350                 }
351                 machineAddresses.add(machineAddress);
352             }
353             return machineAddresses.toArray(new SystemComponent[machineAddresses.size()]);
354         }

356         private static String checkAndMaybeFixOpSys(final String opSys) {
357             String fixed;
358             if(opSys.equalsIgnoreCase("Mac") || opSys.equalsIgnoreCase("OSX")) {
359                 fixed = "Mac OS X";
360             } else if(opSys.equalsIgnoreCase("Win")) {
361                 fixed = "Windows";
362             } else {
363                 fixed = opSys;
364             }
365             return fixed;
366         }

368         private static String[] appendJars(final List<String> base, final String... jars) {
369             List<String> jarList = new ArrayList<String>();
370             jarList.addAll(base);
371             for(String jar : jars) {
372                 if(jarList.contains(jar)) {
373                     continue;
374                 }
375                 jarList.add(jar);
376             }
377             return jarList.toArray(new String[jarList.size()]);
378         }

380         public static boolean isIpAddress(String ip) {
381             String [] parts = ip.split ("\\.");
382             for (String s : parts){
383                 try {
384                     int i = Integer.parseInt(s);
385                     if (i < 0 || i > 255){
386                         return false;
387                     }
388                 } catch(NumberFormatException e) {
389                     return false;
390                 }
391             }
392             return true;
393         }

395         private static String getConfigurationAsString(final String configArg) throws IOException {
396             if(configArg.equalsIgnoreCase("-")) {
397                 return configArg;
398             }
399             StringBuilder stringBuilder = new StringBuilder();
400             String line;
401             String ls = System.getProperty("line.separator");
402             if(configArg.startsWith("http")) {
403                 URL oracle = new URL(configArg);
404                 BufferedReader in = new BufferedReader(new InputStreamReader(oracle.openStream()));
405                 while ((line = in.readLine()) != null) {
406                     stringBuilder.append(line);
407                     stringBuilder.append(ls);
408                 }
409                 in.close();
410             } else {
411                 File configFile = new File(configArg);
412                 BufferedReader reader = null;
413                 try {
414                     reader = new BufferedReader(new FileReader(configFile));
415                     while ((line = reader.readLine()) != null) {
416                         stringBuilder.append(line);
417                         stringBuilder.append(ls);
418                     }
419                 } finally {
420                     if(reader!=null)
421                         reader.close();
422                 }
423             }
424             return stringBuilder.toString();
425         }
