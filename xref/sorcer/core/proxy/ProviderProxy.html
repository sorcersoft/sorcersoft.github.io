50  /**
51   * The Service provider should wrap up the smart proxy/stub. The
52   * java.lang.reflect.Proxy returned from this object will have the following
53   * features. If the server implements RemoteMethodControl, the proxy returned
54   * would implement the functionality of Jini2.0(TM) Security and semantics to
55   * ReferentUuid. If not it implements the semantics of ReferentUuid.
56   * 
57   * Functionality of Jini2.0 Security implemented by this ProxyWrapper Ability to
58   * set and get client constraints Ability to getProxyTrustIterator
59   * 
60   * Functionality of ReferentUuid The unique identifier assigned to the current
61   * instance of this proxy class by the lookup discovery service. This ID is used
62   * to determine equality between proxies.
63   *  * 
64   * @author Mike Sobolewski
65   */
66  
67  @SuppressWarnings("rawtypes")
68  public class ProviderProxy implements Serializable {
69  	private static final long serialVersionUID = -242006752320266252L;
70  	protected final static Logger logger = LoggerFactory.getLogger(ProviderProxy.class);
71  
72  	/**
73  	 * Public static factory method that creates and returns an instance of
74  	 * <code>java.lang.reflect.Proxy</code>. This proxy will implement the
75  	 * semantics of ReferentUuid and jini2.0 security semantics if the server
76  	 * passed onto the createServiceProxy method will implement
77  	 * RemoteMethodControl.
78  	 * 
79  	 * @param proxy
80  	 *            reference to the server object through which communication
81  	 *            occurs between the client-side and server-side of the
82  	 *            associated service.
83  	 * @param proxyID
84  	 *            the unique identifier assigned by the service to each instance
85  	 *            of this proxy
86  	 * 
87  	 * @return an instance of <code>java.lang.reflect.Proxy</code> that
88  	 *         implements <code>RemoteMethodControl</code> if the given
89  	 *         <code>server</code> does.
90  	 */
91  	public static Remote wrapServiceProxy(Object proxy, Uuid proxyID, Object adminProxy, Class... additionalInterfaces) {
92  
93  		if (proxy == null)
94  			throw new NullPointerException("Cannot have a server which is null");
95  
96  		ReferentUuidInvocationHandler handler =
97                  (proxy instanceof RemoteMethodControl) ?
98                          new ConstrainableInvocationHandler(proxy, proxyID, adminProxy) :
99                          new ReferentUuidInvocationHandler(proxy, proxyID, adminProxy);
100 
101         return (Remote)Proxy.newProxyInstance(proxy.getClass().getClassLoader(),
102                                       handler.getInterfaces(proxy, additionalInterfaces),
103                                       handler);
104     }
105 	
106 	public static Remote wrapAdminProxy(Object adminProxy, Uuid adminProxyID, Class... additionalInterfaces) {
107         return wrapServiceProxy(adminProxy, adminProxyID, adminProxy, additionalInterfaces);
108     }

122 		public Class[] getInterfaces(Object proxy, Class... additionalInterfaces) {
123             Set<Class> additionalIfacesSet = new HashSet<Class>();
124             Collections.addAll(additionalIfacesSet, additionalInterfaces);
125             return getInterfaces(proxy, additionalIfacesSet);

128         protected Class[] getInterfaces(Object proxy, Set<Class> list) {
129 			if (proxy == adminProxy) {
130 				// admin interfaces
131 				list.add(ReferentUuid.class);
132 				list.add(DestroyAdmin.class);
133 				list.add(JoinAdmin.class);
134 				list.add(RemoteMethodControl.class);
135 				list.add(RemoteContextManagement.class);
136 			} else {
137 				// provider interfaces
138                 Collections.addAll(list, proxy.getClass().getInterfaces());
139 					list.add(ReferentUuid.class);
140                     list.add(ServiceActivityProvider.class);
141 
142 				if (list.contains(Provider.class)) {
143 					list.remove(AdministratableProvider.class);
144 					list.remove(RemoteContextManagement.class);
145 				}
146 			}
147 
148 			return list.toArray(new Class[list.size()]);
149 		}

151         public Object invoke(Object server, Method m, Object[] args) throws Throwable {
152 			String selector = m.getName();
153             if ("getReferentUuid".equals(selector))
154                 return proxyID;
155             if ("hashCode".equals(selector)) {
156                 return proxyID.hashCode();
157             } else if ("equals".equals(selector)) {
158                 return !(args.length != 1 || !(args[0] instanceof ReferentUuid)) && proxyID.equals(((ReferentUuid) args[0]).getReferentUuid());
159             } else if ("toString".equals(selector)) {
160                 return "refID=" + proxyID + " : proxy=" + proxy;
161             } 
162 
163             try {
164             	return doInvoke(server, selector, m, args);
165             } catch (InvocationTargetException ie) {
166                 Throwable cause = ie.getCause();
167                 // disable logging for calls coming from ServiceCataloger
168                 if (cause instanceof ConnectException && MDC.get("java.net.ConnectException.ignore") == null) {
169                     logger.warn("Proxy Connection problem to : {} to perform: {} for args: {}", proxyID, m, Arrays.toString(args), ie);
170                 }
171                 for (Class<?> throwableType : m.getExceptionTypes()) {
172                     if(throwableType.isInstance(cause))
173                         throw cause;
174                 }
175                 throw ie;
176             } catch (Throwable e) {
177                 // this block is for debugging, can be deleted
178                 // do not report broken network connection on destruction
179                 logger.warn("proxy method: {} for args: {}", m, Arrays.toString(args), e);
180                 if (!(selector.equals("destroyNode") || selector.equals("destroy"))) {
181                     throw e;
182                 } else
183                     return null;
184             }
185         }
186 
187         protected Object doInvoke(Object server, String selector, Method m, Object[] args) throws IllegalAccessException, InvocationTargetException, RemoteException {
188             return m.invoke(proxy, args);
189         }

218         private static final long serialVersionUID = 6152077538535766390L;

221 			super(server, proxyID, adminProxy);
222 		}

224 		@Override
225 		public Class[] getInterfaces(Object proxy, Set<Class> additionalInterfaces) {
226             additionalInterfaces.add(RemoteMethodControl.class);
227             return super.getInterfaces(proxy, additionalInterfaces);
228 		}

230 		public Object doInvoke(Object server, String selector, Method m, Object[] args) throws RemoteException, InvocationTargetException, IllegalAccessException {
231             if ("getConstraints".equals(selector)) {
232                     return ((RemoteMethodControl) proxy).getConstraints();
233                 } else if ("setConstraints".equals(selector)) {
234                     return server;
235                 } else if ("getProxyTrustIterator".equals(selector)) {
236                     return new SingletonProxyTrustIterator(server);
237                 } else if ("isActive".equals(selector)) {
238                     return ((Provider) proxy).isBusy();
239                 } else if ("getAdmin".equals(selector)) {
240                     return ((Administrable) proxy).getAdmin();
241                 } else {
242                     return super.doInvoke(server, selector, m, args);
243                 }
244 		}
