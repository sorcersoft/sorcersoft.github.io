18  package sorcer.core;
19  
20  import sorcer.util.DocumentDescriptor;
21  
22  import javax.transaction.InvalidTransactionException;
23  import java.io.IOException;
24  import java.rmi.RemoteException;
25  
26  /**
27   * The file store interface to access remote directories.
28   */
29  public interface FileStorer {
30  
31  	/**
32  	 * Returns a document descriptor intialized for file read operations.
33  	 * 
34  	 * @param desc
35  	 *            an initial document descriptor
36  	 * @return a document descriptor initialized for read operations
37  	 * @throws RemoteException
38  	 * @throws IOException
39  	 * @throws InvalidTransactionException
40  	 */
41  	public DocumentDescriptor getInputDescriptor(DocumentDescriptor desc)
42  			throws RemoteException, IOException, InvalidTransactionException;
43  
44  	/**
45  	 * Returns a document descriptor intialized for file appending operations.
46  	 * 
47  	 * @param desc
48  	 *            an initial document descriptor
49  	 * @return a document descriptor initialized for append operations
50  	 * @throws RemoteException
51  	 * @throws IOException
52  	 * @throws InvalidTransactionException
53  	 */
54  	public DocumentDescriptor getAppendDescriptor(DocumentDescriptor desc)
55  			throws RemoteException, IOException, InvalidTransactionException;
56  
57  	/**
58  	 * Returns a document descriptor intialized for file writing operations.
59  	 * 
60  	 * @param desc
61  	 *            an initial document descriptor
62  	 * @return a document descriptor initialized for write operation
63  	 * @throws RemoteException
64  	 * @throws IOException
65  	 * @throws InvalidTransactionException
66  	 */
67  	public DocumentDescriptor getOutputDescriptor(DocumentDescriptor desc)
68  			throws RemoteException, IOException, InvalidTransactionException;
69  
70  	/**
71  	 * Lists all directories available on the file store.
72  	 * <p>
73  	 * The format is:
74  	 * <ul>
75  	 * <li># is the path separator</li>
76  	 * <li>/ is the separator between entries</li>
77  	 * </ul>
78  	 * <p>
79  	 * Example: <tt>/test/test#bla
80  	 * </p>
81  	 * 
82  	 * @param docDesc
83  	 *            Only Principal is used.
84  	 * @return a list of all directories.
85  	 */
86  	public String getDirectories(DocumentDescriptor docDesc)
87  			throws RemoteException, IOException;
88  
89  	/**
90  	 * Returns the content of a directory.
91  	 * <p>
92  	 * The components are separeted with a /. Only files and no directories are
93  	 * listed.
94  	 * 
95  	 * @param desc
96  	 *            DocumentDescriptor. FolderPath and Principal are user.
97  	 *            FolderPath uses # as a Path separator.
98  	 * @return
99  	 * @throws RemoteException
100 	 * @throws IOException
101 	 */
102 	public String getDirectoryContent(DocumentDescriptor desc)
103 			throws RemoteException, IOException;
104 
105 	public String listDirectories(DocumentDescriptor desc)
106 			throws RemoteException, IOException;
107 
108 }
