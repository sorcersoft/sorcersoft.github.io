176 	 * This function recursively goes thru the tree adding items to the dataContext
177 	 * using the tree path as the path for the dataContext and the datanode as the
178 	 * data. The function also marks the node for the type of data it is
179 	 * (input/output).
180 	 * 
181 	 * @param theContext

185 	 * @param currentPath

187 	 *            to with the dataContext path separator of /

197 		// have kids and

199 		// not at root
200 		{ // dont want root because it is technically the name of the dataContext
201 			// and not a valid path
202 
203 			// we have no kids so we just output the path.
204 			try {
205 				Object child = ((DefaultMutableTreeNode) current)
206 						.getUserObject();
207 				if (child instanceof ContextNodeType) // check if is of our
208 				// ContextNodeType
209 				{
210 					ContextNodeType cnt = (ContextNodeType) child;
211 
212 					// output the path with the correct type
213 					if (cnt.getDirection().equals(ContextNodeType.INPUTTYPE))
214 						theContext.putInValue(currentPath, "");
215 					if (cnt.getDirection().equals(ContextNodeType.OUTPUTTYPE))
216 						theContext.putOutValue(currentPath, "");
217 					if (cnt.getDirection().equals(ContextNodeType.INOUTPUTTYPE))
218 						theContext.putInoutValue(currentPath, "");
219 					if (cnt.getDirection().equals(ContextNodeType.DEFAULTTYPE))
220 						theContext.putValue(currentPath, "");
221 				}
222 			} catch (ContextException e) {

225 			}
226 		} else // has kids or is the root
227 		{
228 			for (int i = 0; i < childcount; i++) // go thru each kid
229 			{

231 				// add is datanode check

251 						// output it, as data nodes can

253 						{

255 							try {
256 								child = ((DefaultMutableTreeNode) current)
257 										.getUserObject();

260 									// output the data and direction marking
261 									if (cnt2.getDirection().equals(

297 					}

333 	 * removed too. Nothing happens if called on the root node.
334 	 */
335 	public void removeCurrentNode() {
336 		TreePath currentSelection = tree.getSelectionPath();
337 		if (currentSelection != null) {
338 			DefaultMutableTreeNode currentNode = (DefaultMutableTreeNode) (currentSelection
339 					.getLastPathComponent());
340 			MutableTreeNode parent = (MutableTreeNode) (currentNode.getParent());
341 
342 			if (parent != null) // check if root
343 			{
344 				treeModel.removeNodeFromParent(currentNode);
345 				return;
346 			}
347 		}
348 
349 		// Either there was no selection, or the root was selected.
350 		toolkit.beep();
351 	}

390 	/**
391 	 * Add child to the tree under parent, and set the visibility of it. If
392 	 * parent is incorrect the child will be added under the root. This method
393 	 * enforces rules prevents any item from appears as a child to a datanode,
394 	 * also only allows one data node for path.
395 	 * 
396 	 * 
397 	 * @param parent
398 	 *            DefaultMutableTreeNode node where the data should be added to
399 	 * @param child
400 	 *            Object representing data to be stored in the tree
401 	 * @param shouldBeVisible
402 	 *            Boolean indicating if the new node should be expanded or
403 	 *            collapsed
404 	 * @return DefaultMutableTreeNode of the newly created treenode
405 	 */
406 	public DefaultMutableTreeNode addObject(DefaultMutableTreeNode parent,
407 			Object child, boolean shouldBeVisible) {
408 		DefaultMutableTreeNode childNode = new DefaultMutableTreeNode(child);
409 
410 		if (parent == null) {
411 			parent = rootNode;
412 		}
413 
414 		if (parent.getUserObject() instanceof ContextNodeType) // check if the
415 		// current type
416 		// is our
417 		// special type
418 		{
419 			ContextNodeType parentObject = (ContextNodeType) parent
420 					.getUserObject(); // get the parent
421 
422 			if (parentObject.isDataNode()) // can't add to datanode object type
423 				return null;
424 
425 			if (child instanceof ContextNodeType) {
426 				if (((ContextNodeType) child).isDataNode()) {
427 					if (parent == rootNode)
428 						return null; // prevent datanode children on the root
429 					for (int i = 0; i < parent.getChildCount(); i++) // only one
430 					// datanode
431 					// per-path
432 					{
433 						DefaultMutableTreeNode cNode = (DefaultMutableTreeNode) treeModel
434 								.getChild(parent, i);
435 						if (cNode.getUserObject() instanceof ContextNodeType) {
436 							if (((ContextNodeType) cNode.getUserObject())
437 									.isDataNode())
438 								return null; // enforce only one datanode per
439 							// path
440 						}
441 					}
442 				}
443 			}
444 		}
445 
446 		// It is key to invoke this on the TreeModel, and NOT
447 		// DefaultMutableTreeNode
448 		treeModel.insertNodeInto(childNode, parent, parent.getChildCount());
449 
450 		// Make sure the user can see the lovely new node.
451 		if (shouldBeVisible) {
452 			tree.scrollPathToVisible(new TreePath(childNode.getPath()));
453 		}
454 		return childNode; // return the new node
455 	}

465 	public DefaultMutableTreeNode nodeExists(String path) {
466 		String splitPath[] = path.split("/"); // get each part of the path
467 		Object nodes = treeModel.getRoot();
468 		for (String indPath : splitPath) // go thru each part of the path
469 		{
470 			Boolean foundNode = false; // indicate we have not found the node
471 			// System.out.println("working on "+indPath);
472 			int count = treeModel.getChildCount(nodes);
473 			// System.out.println(nodes.toString()+" has "+count+" kids");
474 			for (int i = 0; i < count; i++) // go true each child
475 			{
476 				Object node = treeModel.getChild(nodes, i);
477 				// System.out.println(node.toString()+" is node number "+i);
478 				if (indPath.equals(node.toString())) // matched this part, keep
479 				// going
480 				{
481 					// System.out.println("nodes matched");
482 					nodes = node;
483 					foundNode = true;
484 					break;
485 				}
486 			}
487 			if (!foundNode) // no match kill it off
488 				return null;
489 
490 			// System.out.println("Children of path"+indPath+" num:"+treeModel.getChildCount(new
491 			// DefaultMutableTreeNode(indPath)));
492 		}
493 
494 		return (DefaultMutableTreeNode) nodes; // return the node found
495 	}
