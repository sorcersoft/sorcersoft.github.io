18  package sorcer.core.provider.cataloger.ui;
19  
20  import java.awt.BorderLayout;
21  import java.rmi.RMISecurityManager;
22  
23  import javax.swing.JPanel;
24  import javax.swing.JTabbedPane;
25  
26  import sorcer.core.Provider;
27  import sorcer.ui.exertlet.EditorView;
28  
29  /**
30   * This class implements the Context editing window, which is used in the
31   * Provider UI.
32   */
33  public class ContextScriptView extends JPanel {
34  
35  	/**
36  	 * Private copy of the dataContext used for various operations, like load other
37  	 * and save as.
38  	 */	
39  	private EditorView browser;
40  	private EditorView inViewer;
41  	private EditorView outViewer;
42  	private Provider provider; 
43  	private BrowserModel model;
44  	/**
45  	 * Tabbed pane to hold the input and output ContextTrees
46  	 */
47  	private JTabbedPane tabbedPane;
48  	
49  	/**
50  	 * Constructor for creating the dataContext view panel
51  	 * 
52  	 * @param dispatcher
53  	 *            Remote dispatcher to use.
54  	 */
55  	public ContextScriptView(SignatureDispatchment dispatcher, BrowserModel model) {
56  		super(new BorderLayout());
57  		// needed for exerting the service
58  		this.model = model;
59  		if (System.getSecurityManager() == null)
60  			System.setSecurityManager(new RMISecurityManager());
61  
62          provider = dispatcher.getProvider();
63  
64  		tabbedPane = new JTabbedPane();
65  
66  		inViewer = new EditorView("", false);
67  		
68  		tabbedPane.addTab("Browser", null,
69  				createBrowserPanel(null), "Browser");
70  		add(tabbedPane, BorderLayout.CENTER);
71  	}
72  
73  	  public JPanel createOutContextPanel(String text) {
74  			outViewer = new EditorView(null, false, false, false, false, true, model);
75  			outViewer.setProvider(provider);
76  			outViewer.setText(text);
77  			outViewer.setTabbedPane(tabbedPane);
78  			outViewer.setDelegate(browser);
79  			tabbedPane.addTab("Out Viewer", null,
80  					outViewer, "Out Viewer");
81  			tabbedPane.setSelectedComponent(outViewer);
82  			return outViewer;
83  		}
84  	  
85  	   private JPanel createBrowserPanel(String text) {
86  			browser = new EditorView(null, true, false, true, false, false, model);
87  			browser.setProvider(provider);
88  			if (text != null && text.length() >0)
89  				browser.setText(text);
90  			browser.setTabbedPane(tabbedPane);
91  			inViewer.setDelegate(browser);
92  			inViewer.setTabbedPane(tabbedPane);
93  			return browser;
94  		}
95  
96  }
