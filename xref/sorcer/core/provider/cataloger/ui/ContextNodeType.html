18  package sorcer.core.provider.cataloger.ui;
19  
20  /**
21   * This class represents a ContextNode and is used in the formation of the
22   * ContextTree for the UI. This class stores the current data object, a boolean
23   * indicating if it is a dataNode or path and optionally the direction
24   * (input/output). This data is used when the tree is converted back to a
25   * dataContext.
26   * 
27   * 
28   * @author Greg McChesney
29   * 
30   */
31  public class ContextNodeType {
32  	/**
33  	 * The value of the underlying data
34  	 */
35  	private Object value;
36  
37  	/**
38  	 * boolean indicating if this represents a datanode=true or path=false
39  	 */
40  	private boolean dataNode;
41  
42  	/**
43  	 * String representing the direction (input/output marking on the path),
44  	 * only used for paths.
45  	 */
46  	private String direction;
47  
48  	/**
49  	 * String for the INPUT path
50  	 */
51  	public static String INPUTTYPE = "Input";
52  
53  	/**
54  	 * String for the OUTPUT path
55  	 */
56  	public static String OUTPUTTYPE = "Output";
57  
58  	/**
59  	 * String for the INPUT and OUTPUT path
60  	 */
61  	public static String INOUTPUTTYPE = "InOutput";
62  
63  	/**
64  	 * String for the default paths, and all datanodes
65  	 */
66  	public static String DEFAULTTYPE = "Default";
67  
68  	/**
69  	 * Constructor for new nodes, accepts the original data from the dataContext and
70  	 * a boolean indicating if it is a data node. If it is a path then its a
71  	 * string with the path name
72  	 * 
73  	 * @param v
74  	 *            Object representing the data to be stored
75  	 * @param isDataNode
76  	 *            Boolean indicating if it is a data node or not
77  	 */
78  	ContextNodeType(Object v, boolean isDataNode) {
79  		value = v;
80  		dataNode = isDataNode;
81  		direction = DEFAULTTYPE;
82  	}
83  
84  	/**
85  	 * Returns a string representation of the value.
86  	 * 
87  	 * @return String representation of the value
88  	 */
89  	public String getValue() {
90  		return value.toString();
91  	}
92  
93  	/**
94  	 * Indicates if the current item is a data node or not.
95  	 * 
96  	 * @return Boolean indicating if it is a data node
97  	 */
98  	public Boolean isDataNode() {
99  		return dataNode;
100 	}
101 
102 	/**
103 	 * Overloading the toString operator to get the value of the data.
104 	 * 
105 	 * @return String representation of the current value of the inner data
106 	 */
107 	public String toString() {
108 		return value.toString();
109 	}
110 
111 	/**
112 	 * Used to update the internal value to a new one
113 	 * 
114 	 * @param value
115 	 *            Object representing the new value the node should contain
116 	 */
117 	public void setValue(Object value) {
118 		this.value = value;
119 	}
120 
121 	/**
122 	 * Used to get the internal data currently being stored, this returns it in
123 	 * the original form.
124 	 * 
125 	 * @return Object of the original data.
126 	 */
127 	public Object getObject() {
128 		return value;
129 	}
130 
131 	/**
132 	 * Returns the current direction of this node.
133 	 * 
134 	 * @return String representing the current direction
135 	 */
136 	public String getDirection() {
137 		return direction;
138 	}
139 
140 	/**
141 	 * Sets the current direction of the node
142 	 * 
143 	 * @param newDirection
144 	 *            String representing the current direction
145 	 */
146 	public void setDirection(String newDirection) {
147 		direction = newDirection;
148 	}
149 
150 }
