162 	public Exertion doJob(Exertion job) {
163 		setServiceID(job);
164 		try {
165 			if (job.getControlContext().isMonitorable()
166 					&& !job.getControlContext().isWaitable()) {
167 				replaceNullExertionIDs(job);
168 				notifyViaEmail(job);
169 				new JobThread((Job) job, this).start();
170 				return job;
171 			} else {
172 				JobThread jobThread = new JobThread((Job) job, this);
173 				jobThread.start();
174 				jobThread.join();
175 				Job result = jobThread.getResult();
176 				logger.finest("Result: " + result);
177 				return result;
178 			}
179 		} catch (Throwable e) {
180 			e.printStackTrace();
181 			return null;
182 		}
183 	}

285 		if (ex != null && ex.getId() == null) {
286 			((ServiceExertion) ex)
287 					.setId(UuidFactory.generate());
288 			if (ex.isJob()) {

290 					replaceNullExertionIDs(((Job) ex).exertionAt(i));

295 	private void notifyViaEmail(Exertion ex) throws ContextException {
296 		if (ex == null || ex.isTask())
297 			return;
298 		Job job = (Job) ex;
299 		Vector recipents = null;
300 		String notifyees = ((ControlContext) job.getDataContext()).getNotifyList();
301 		if (notifyees != null) {
302 			String[] list = StringUtils.tokenize(notifyees, SorcerConstants.MAIL_SEP);
303 			recipents = new Vector(list.length);
304 			for (int i = 0; i < list.length; i++)
305 				recipents.addElement(list[i]);
306 		}
307 		String to = "", admin = SorcerEnv.getProperty("sorcer.admin");
308 		if (recipents == null) {
309 			if (admin != null) {
310 				recipents = new Vector();
311 				recipents.addElement(admin);
312 			}
313 		} else if (admin != null && !recipents.contains(admin))
314 			recipents.addElement(admin);
315 
316 		if (recipents == null)
317 			to = to + "No e-mail notifications will be sent for this job.";
318 		else {
319 			to = to + "e-mail notification will be sent to\n";
320 			for (int i = 0; i < recipents.size(); i++)
321 				to = to + "  " + recipents.elementAt(i) + "\n";
322 		}
323 		String comment = "Your job '" + job.getName()
324 				+ "' has been submitted.\n" + to;
325 		((ControlContext) job.getDataContext()).setFeedback(comment);
326 		if (job.getMasterExertion() != null
327 				&& job.getMasterExertion().isTask()) {
328 			job.getMasterExertion().getDataContext()
329 					.putValue(Context.JOB_COMMENTS, comment);
330 
331 			Contexts.markOut(
332 					job.getMasterExertion().getDataContext(),
333 					Context.JOB_COMMENTS);
334 
335 		}
336 	}

354 		public void run() {
355 			logger.finest("*** JobThread Started ***");
356 			Dispatcher dispatcher = null;
357 
358 			try {
359 				dispatcher = ExertionDispatcherFactory.getFactory()
360 						.createDispatcher(job,
361 								new HashSet<Context>(), false, myMemberUtil, provider);
362 				while (dispatcher.getState() != ExecState.DONE
363 						&& dispatcher.getState() != ExecState.FAILED
364 						&& dispatcher.getState() != ExecState.SUSPENDED) {
365 					logger.fine("Dispatcher waiting for exertions... Sleeping for 250 milliseconds.");
366 					Thread.sleep(250);
367 				}
368 				logger.fine("Dispatcher State: " + dispatcher.getState());
369 			} catch (DispatcherException e) {
370 				e.printStackTrace();
371 			} catch (InterruptedException e) {
372 				e.printStackTrace();
373 			}
374 			result = (NetJob) dispatcher.getExertion();
375 			try {
376 				job.getControlContext().appendTrace(provider.getProviderName()  + " dispatcher: " 
377 						+ dispatcher.getClass().getName());
378 			} catch (RemoteException e) {
379 				// ignore it
380 			}
381 		}
