67  		landlord = new MonitorLandlord();
68  		String dbHome = getProperty("monitor.database.home");
69  		File dbHomeFile = null;
70  		if (dbHome == null || dbHome.length() == 0) {
71  			logger.severe("Session database home missing: " + dbHome);
72  			destroy();
73  		} else {
74  			dbHomeFile = new File(dbHome);
75  			if (!dbHomeFile.isDirectory() && !dbHomeFile.exists()) {			
76  				boolean done = dbHomeFile.mkdirs();
77  				if (!done) {
78  					logger.severe("Not able to create session database home: " 
79  							+ dbHomeFile.getAbsolutePath());
80  					destroy();
81  				}
82  			}
83  		}
84          logger.fine("Opening BDBJE environment in: " + dbHomeFile.getAbsolutePath());
85  		db = new SessionDatabase(dbHome);
86  		SessionDatabaseViews views = new SessionDatabaseViews(db);
87  		resources = views.getSessionMap();
88  
89  		// statically initialize
90  		MonitorSession.mLandlord = landlord;
91  		MonitorSession.sessionManager = (MonitoringManagement) getServiceProxy();
92  	}

174 	private MonitorSession findSessionResource(Uuid cookie)
175 			throws MonitorException {
176 
177 		MonitorSession resource;
178 
179 		// Check if landlord is keeping it in memory
180 		Hashtable lresources = landlord.getResources();
181 		if (lresources.get(cookie) != null)
182 			return (MonitorSession) lresources.get(cookie);
183 
184 		Uuid key;
185 		for (Enumeration e = lresources.keys(); e.hasMoreElements();) {
186 			key = (Uuid) e.nextElement();
187 			resource = ((MonitorSession) lresources.get(key))
188 			.getSessionResource(cookie);
189 			if (resource != null)
190 				return resource;
191 		}
192 
193 		// if (landlord.getResource(cookie)!=null) return
194 		// (SessionResource)landlord.getResource(cookie);
195 
196 		// Ok it's not with landlord. So we retrieve it from the database
197 		synchronized (resourcesWriteLock) {
198 			Iterator<Map.Entry<UuidKey, IMonitorSession>> si = resources.entrySet().iterator();
199 			Map.Entry<UuidKey, IMonitorSession> next;
200 			while (si.hasNext()) {
201 				next = si.next();
202 				try {
203 					resource = getSession(next.getKey()).getSessionResource(cookie);
204 				} catch (Exception e) {
205 					throw new MonitorException(e);
206 				} 
207 				if (resource != null)
208 					return resource;
209 			}
210 		}
211 			return null;
212 	}

215 	 * 
216 	 * If the Broker wants to drop the exertion to space, then the Broker has no
217 	 * idea who will pick up this exertion. In that case, it doesn't make sense
218 	 * for the broker to force leasing. However, it may activate the the session
219 	 * with the timeout marked and the lease duration specified so that if no
220 	 * provider picks out and the task gets timed out, then we can clean up the
221 	 * entry from space and notify the broker.
222 	 * 
223 	 * If the provider picks up before it timesout, then the provider must
224 	 * initialize this session by calling init(Monitorable) so that the monitor
225 	 * will now make sure that the leases are renewed properly for this session.
226 	 * 
227 	 * @param duration
228 	 *            Requested lease duration for the session.
229 	 * @param timeout
230 	 *            Timeout for execution of this task wich includes idle time in
231 	 *            space.
232 	 * 
233 	 * @throws MonitorException
234 	 *             1) If this session is already active 2) If there is no such
235 	 *             session
236 	 * 
237 	 * @throws RemoteException
238 	 *             if there is a communication error
239 	 * 
240 	 */

434 	public Exertion getMonitorableExertion(UEID cookie, Principal principal)
435 			throws RemoteException, MonitorException {
436 		UuidKey key;
437 		Exertion ex;
438 		Iterator<UuidKey> ki = resources.keySet().iterator();
439 		while (ki.hasNext()) {
440 			key = ki.next();
441 			ex = (getSession(key)).getRuntimeExertion();
442 
443 			if (cookie.exertionID.equals(ex.getId())
444 					&& ((ServiceExertion) ex).getPrincipal().getId()
445 							.equals(((SorcerPrincipal) principal).getId()))
446 				return ex;
447 			else
448 				return null;
449 		}
450 		return null;
451 	}
