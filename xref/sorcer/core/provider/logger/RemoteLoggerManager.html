79  	public void init(String[] args) {
80  		knownLoggers = new Vector<LoggingConfig>();
81  		final String[] argv = args;

83  				+ File.separator + "logs" + File.separator + "remote";
84  		new Thread() {
85  			public void run() {
86  				Configuration config;
87  				if (argv.length == 0) {
88  					logDir = defautDir;
89  				} else {
90  					try {
91  						config = ConfigurationProvider.getInstance(argv);
92  						logDir = (String) config.getEntry(
93  								"sorcer.core.provider.logger.loggerConfig",
94  								"loggerDir", String.class, defautDir);
95  					} catch (ConfigurationException ce) {
96  						ce.printStackTrace();
97  					}
98  				}
99  				logger.info("ServiceLogManager logDir:" + logDir);
100 				File dir = new File(logDir);
101 				if (!dir.exists()) {
102 					dir.getParentFile().mkdir();
103 					dir.mkdir();
104 				}
105 			}
106 		}.start();
107 	}

122 	public synchronized void publish(LogRecord record) throws RemoteException {
123 		boolean contains = false;
124 		Logger log = null;
125 
126 		Enumeration<String> e = LogManager.getLogManager().getLoggerNames();
127 
128 		while (e.hasMoreElements()) {
129 			String loggerName = e.nextElement();
130 			if (record.getLoggerName().equals(loggerName))
131 				contains = true;
132 		}
133 
134 		if (!contains) {
135 			log = Logger.getLogger(record.getLoggerName());
136 			log.setLevel(Level.ALL);
137 
138 			FileHandler h;
139 			try {
140 				String fs = File.separator;
141 				h = new FileHandler(SorcerEnv.getHomeDir() + fs
142 						+ "logs" + fs + "remote" + fs + "remote-logger-" + log.getName()
143 						+ "%g.log", 20000, 8, true);
144 
145 				if (h != null) {
146 					h.setFormatter(new SimpleFormatter());
147 					log.addHandler(h);
148 				}
149 				log.setUseParentHandlers(false);
150 			} catch (SecurityException e1) {
151 				e1.printStackTrace();
152 			} catch (IOException e1) {
153 				e1.printStackTrace();
154 			}
155 		} else {
156 			log = Logger.getLogger(record.getLoggerName());
157 		}
158 		if (log.isLoggable(record.getLevel()))
159 			log.log(record);
160 
161 		String name = record.getLoggerName();
162 		LoggingConfig lc = new LoggingConfig(name, null);
163 		if (!knownLoggers.contains(lc)) {
164 			Level lev = logger.getLevel();
165 			lc.setLevel(lev);
166 			knownLoggers.add(lc);
167 		}
168 	}

288 	public String getLogComments(String filename) {
289 		Pattern p = null;
290 		try {
291 			// The following pattern lets this extract multiline comments that
292 			// appear on a single line (e.g., /* same line */) and single-line
293 			// comments (e.g., // some line). Furthermore, the comment may
294 			// appear anywhere on the line.
295 			p = Pattern.compile(".*/\\*.*\\*/|.*//.*$");
296 		} catch (PatternSyntaxException e) {
297 			System.err.println("Regex syntax error: " + e.getMessage());
298 			System.err.println("Error description: " + e.getDescription());
299 			System.err.println("Error index: " + e.getIndex());
300 			System.err.println("Erroneous pattern: " + e.getPattern());
301 		}
302 		BufferedReader br = null;
303 		StringBuffer bw = new StringBuffer();
304 		try {
305 			FileReader fr = new FileReader(filename);
306 			br = new BufferedReader(fr);
307 			Matcher m = p.matcher("");
308 			String line;
309 			while ((line = br.readLine()) != null) {
310 				m.reset(line);
311 				if (m.matches()) /* entire line must match */
312 				{
313 					bw.append(line + "\n");
314 				}
315 			}
316 		} catch (IOException e) {
317 			System.err.println(e.getMessage());
318 		} finally // Close file.
319 		{
320 			try {
321 				if (br != null)
322 					br.close();
323 			} catch (IOException e) {
324 			}
325 		}
326 		return bw.toString();
327 	}
