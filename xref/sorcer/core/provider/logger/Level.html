204     public static Level toLevel(String sArg, Level defaultLevel) {
205         if (sArg == null) {
206             return defaultLevel;

209         if (sArg.equalsIgnoreCase("ALL")) {
210             return Level.ALL;
211         }
212         if (sArg.equalsIgnoreCase("TRACE")) {
213             return Level.TRACE;
214         }
215         if (sArg.equalsIgnoreCase("DEBUG")) {
216             return Level.DEBUG;
217         }
218         if (sArg.equalsIgnoreCase("INFO")) {
219             return Level.INFO;
220         }
221         if (sArg.equalsIgnoreCase("WARN")) {
222             return Level.WARN;
223         }
224         if (sArg.equalsIgnoreCase("ERROR")) {
225             return Level.ERROR;
226         }
227         if (sArg.equalsIgnoreCase("OFF")) {
228             return Level.OFF;
229         }
230         return defaultLevel;
231     }
