224     /**
225      * Process the Exertion accordingly if it is a job, task, or a Conditional
226      * Exertion.
227      *
228      * @return Exertion the result
229      * @see NetJob
230      * @see NetTask
231      * @see Conditional
232      * @throws ExertionException
233      *             exception from other methods
234      */
235     public Exertion process() throws ExertionException {
236         logger.info("process exertion: " + exertion.getName());
237         try {
238             Exertion result = null;
239             if (exertion.isConditional()) {
240                 logger.info("exertion Conditional");
241                 result = doConditional(exertion);
242                 logger.info("exertion Conditional; result: " + result);
243             } else if (exertion.isJob()) {
244                 logger.info("exertion isJob()");
245                 result = doRendezvousExertion((Job) exertion);
246                 logger.info("exertion isJob(); result: " + result);
247             } else if (exertion.isBlock()) {
248                 logger.info("exertion isBlock()");
249                 result = doBlock((Block) exertion);
250                 logger.info("exertion isBlock(); result: " + result);
251             } else if (exertion.isTask()) {
252                 logger.info("exertion isTask()");
253                 result = doTask((Task) exertion);
254                 logger.info("exertion isTask(); result: " + result);
255             }
256             return result;
257         } catch (RuntimeException e) {
258             throw e;
259         } catch (ExertionException e) {
260             throw e;
261         } catch (Exception e) {
262             throw new ExertionException(e.getMessage(), e);
263         }
264     }

267      * This method delegates the doTask method to the ExertionDelegate.
268      *
269      * @param task
270      *            ServiceTask
271      * @return ServiceTask
272      * @throws RemoteException
273      *             exception from ExertionDelegate
274      * @throws ExertionException
275      *             exception from ExertionDelegate
276      * @throws SignatureException
277      *             exception from ExertionDelegate
278      * @throws TransactionException
279      * @throws ContextException
280      */
281     public Task doTask(Task task) throws RemoteException, ExertionException,
282             SignatureException, TransactionException, ContextException {
283         Task result;
284         if (task.getControlContext().getAccessType() == Access.PULL) {
285             result = (Task)doRendezvousExertion(task);
286         } else if (delegate != null) {
287             result = delegate.doTask(task, null);
288         }
289         else if (task.isConditional())
290             result = doConditional(task);
291         else
292             result = doBatchTask(task);
293 
294         return result;
295     }

328     public Exertion doRendezvousExertion(ServiceExertion xrt) throws RemoteException, ExertionException {
329         try {
330             if (xrt.isSpacable()) {
331                 logger.info("exertion isSpacable");
332 
333                 if (spacer == null) {
334                     String spacerName = xrt.getRendezvousName();
335                     Spacer spacerService;
336                     if (spacerName != null) {
337                         spacerService = SpacerAccessor.getSpacer(spacerName);
338                     } else {
339                         try {
340                             spacerService = SpacerAccessor.getSpacer();
341                         } catch (Exception x) {
342                             throw new ExertionException("Could not find Spacer", x);
343                         }
344                     }
345                     logger.info("Got Spacer: " + spacerService);
346                     return spacerService.service(xrt, null);
347                 }
348 				Exertion job = ((Executor)spacer).execute(xrt, null);
349                 logger.info("spacable exerted = " + job);
350                 return job;
351             }
352             else {
353                 logger.info("exertion NOT Spacable");
354                 if (jobber == null) {
355                     String jobberName = xrt.getRendezvousName();
356 					Jobber jobberService;
357                     if (jobberName != null)
358                         jobberService = JobberAccessor.getJobber(jobberName);
359                     else
360                         try {
361                             jobberService = JobberAccessor.getJobber();
362                         } catch (AccessorException e) {
363                             throw new ExertionException("Could not find Jobber", e);
364                         }
365                     logger.info("Got Jobber: " + jobber);
366                     return jobberService.service(xrt, null);
367                 }
368 				Exertion job = ((Executor)jobber).execute(xrt, null);
369                 logger.info("job exerted = " + job);
370                 return job;
371             }
372         } catch (TransactionException e) {
373             logger.error( "Error", e);
374         }
375         return null;
376     }

519     public Task doBatchTask(Task task) throws ExertionException,
520             SignatureException, RemoteException, ContextException {
521 		ServiceFidelity alls = task.getFidelity();
522 		Signature lastSig = alls.get(alls.size()-1);
523 		if (alls.size() > 1 &&  task.isBatch() && !(lastSig instanceof NetSignature)) {
524 			for (int i = 0; i< alls.size()-1; i++) {
525 				alls.get(i).setType(Signature.PRE);
526 			}
527 		}
528         task.startExecTime();
529         // append context from Contexters
530         if (task.getApdProcessSignatures().size() > 0) {
531             Context cxt = apdProcess(task);
532             cxt.setExertion(task);
533             task.setContext(cxt);
534         }
535         // do preprocessing
536         if (task.getPreprocessSignatures().size() > 0) {
537             Context cxt = preprocess(task);
538             cxt.setExertion(task);
539             task.setContext(cxt);
540         }
541         // execute service task
542 		ServiceFidelity ts = new ServiceFidelity(1);
543         Signature tsig = task.getProcessSignature();
544         ((ServiceContext)task.getContext()).setCurrentSelector(tsig.getSelector());
545         ((ServiceContext)task.getContext()).setCurrentPrefix(tsig.getPrefix());
546 
547         ts.add(tsig);
548         task.setFidelity(ts);
549         if (tsig.getReturnPath() != null)
550             ((ServiceContext)task.getContext()).setReturnPath(tsig.getReturnPath());
551 
552         task = task.doTask();
553         if (task.getStatus() <= Exec.FAILED) {
554             task.stopExecTime();
555             ExertionException ex = new ExertionException("Batch service task failed: "
556                     + task.getName());
557             task.reportException(ex);
558             task.setStatus(Exec.FAILED);
559             task.setFidelity(alls);
560             return task;
561         }
562         task.setFidelity(alls);
563         // do postprocessing
564         if (task.getPostprocessSignatures().size() > 0) {
565             Context cxt = postprocess(task);
566             cxt.setExertion(task);
567             task.setContext(cxt);
568         }
569         if (task.getStatus() <= Exec.FAILED) {
570             task.stopExecTime();
571             ExertionException ex = new ExertionException("Batch service task failed: "
572                     + task.getName());
573             task.reportException(ex);
574             task.setStatus(Exec.FAILED);
575             task.setFidelity(alls);
576             return task;
577         }
578         task.setFidelity(alls);
579         task.stopExecTime();
580         return task;
581     }

595     public Context processContinousely(Task task, List<Signature> signatures)
596             throws ExertionException, ContextException {
597         Signature.Type type = signatures.get(0).getType();
598         Task t;
599         Context shared = task.getContext();
600         for (int i = 0; i < signatures.size(); i++) {
601             try {
602                 t = task(task.getName() + "-" + i, signatures.get(i), shared);
603                 signatures.get(i).setType(Signature.SRV);
604                 ((ServiceContext)shared).setCurrentSelector(signatures.get(i).getSelector());
605                 ((ServiceContext)shared).setCurrentPrefix(signatures.get(i).getPrefix());
606 
607                 ServiceFidelity tmp = new ServiceFidelity(1);
608                 tmp.add(signatures.get(i));
609                 t.setFidelity(tmp);
610                 t.setContinous(true);
611                 t.setContext(shared);
612 
613                 logger.info("Sending one of the batch tasks to exert: " + t.getName() + " " + t.getFidelity().toString());
614                 t = t.doTask();
615                 signatures.get(i).setType(type);
616                 shared = t.getContext();
617                 if (t.getStatus() <= Exec.FAILED) {
618                     task.setStatus(Exec.FAILED);
619                     ExertionException ne = new ExertionException(
620                             "Batch signature failed: " + signatures.get(i));
621                     task.reportException(ne);
622                     task.setContext(shared);
623                     return shared;
624                 }
625             } catch (Exception e) {
626                 logger.error( "Error", e);
627                 task.setStatus(Exec.FAILED);
628                 task.reportException(e);
629                 task.setContext(shared);
630                 return shared;
631             }
632         }
633         // return the service context of the last exertion
634         return shared;
635     }
