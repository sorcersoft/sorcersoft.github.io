81  /**
82   * The ServiceProvider class is a type of {@link Provider} with dependency
83   * injection defined by a Jini 2 configuration, proxy management, and own
84   * service discovery management for registering its proxies. In the simplest
85   * case, the provider exports and registers its own (outer) proxy with the
86   * primary method {@link sorcer.service.Service#service(Exertion, Transaction)}. The functionality of an
87   * outer proxy can be extended by its inner server functionality with its Remote
88   * inner proxy. In this case, the outer proxies have to implement {@link sorcer.core.provider.proxy.Outer}
89   * interface and each outer proxy is registered with the inner proxy allocated
90   * with {@link Outer#getInner} invoked on the outer proxy. Obviously an outer
91   * proxy can implement own interfaces with the help of its embedded inner proxy
92   * that in turn can consit of multiple own inner proxies as needed. This class
93   * implements the {@link sorcer.core.provider.proxy.Outer} interface, so can extend its functionality by
94   * inner proxying.
95   * <p>
96   * A smart proxy can be defined in the provider's Jini configuration by the
97   * <code>smartProxy</code> entry. This proxy does not represent directly any
98   * exported server, it is registered with lookup services as is. However, the
99   * smart proxy (implementing {@link sorcer.core.provider.proxy.Outer} interface) can extend its
100  * functionality by setting the providers outer proxy as its inner proxy. Thus,
101  * smart proxies that implement{@link sorcer.core.provider.proxy.Outer}, called <i>semismart</i> contain
102  * this provider's proxy as its inner proxy. Smart proxies that do not extend
103  * its functionality via its inner proxy are called fat proxies. Fat proxies do
104  * not make remote calls back to their providers and the providers just maintain
105  * lookup service registrations.
106  * <p>
107  * An inner or outer proxy can be a surrogate of a service partner defined by
108  * this provider using <code>server</code> configuration entry or two realted
109  * entries: <code>serverType</code> and <code>serverName</code>. If the entry
110  * <code>server</code> is defined and its exporter is defined by the entry
111  * <code>serverExporter</code> then this provider will use the server's proxy as
112  * the outer (primary proxy) and itself as the inner proxy. However if the
113  * exporter is not defined then the provider's proxy is primary (outer) and the
114  * server's proxy is the inner one.
115  * <p>
116  * On the other hand, if the <code>server</code> entry is not defined but at
117  * least <code>serverType</code> is defined then the instance of server is
118  * created and exported if the entry <code>serverExporter</code> is given. The
119  * exported server proxy becomes the primary provider's proxy. However, if no
120  * exporter is defined then server proxy becomes the inner proxy of this
121  * provider's proxy (outer proxy). Thus, exported servers use outer proxies
122  * while not exported user inner proxies of this provider. In this context, a
123  * smart proxy implementing {@link sorcer.core.provider.proxy.Outer} interface can get the outer proxy and
124  * its inner proxy composed in either direction provider/server proxy
125  * relationship.
126  * <p>
127  * A service method is a method returning {@link sorcer.core.context.ServiceContext} and having a
128  * single parameter as {@link sorcer.core.context.ServiceContext}. Service beans are components that
129  * implement interfaces in terms of service methods only. A list of service
130  * beans can be specified in a provider's Jini configuration as the
131  * <code>beans</code> entry. In this case a proxy implementing all interfaces
132  * implemented by service beans are dynamically created and registered with
133  * lookup services. Multiple SORECER servers can be deployed within a single
134  * {@link sorcer.core.provider.ServiceProvider} as its own service beans.
135  * 
136  * @see sorcer.core.Provider
137  * @see net.jini.lookup.ServiceIDListener
138  * @see ReferentUuid
139  * @see sorcer.core.AdministratableProvider
140  * @see net.jini.export.ProxyAccessor
141  * @see net.jini.security.proxytrust.ServerProxyTrust
142  * @see net.jini.core.constraint.RemoteMethodControl
143  * @see com.sun.jini.start.LifeCycle
144  * @see Partner
145  * @see sorcer.core.provider.proxy.Partnership
146  * @see sorcer.core.context.RemoteContextManagement
147  * @see sorcer.core.SorcerConstants
148  */

614 	/**
615 	 * Portion of construction that is common between the activatable and not
616 	 * activatable cases. This method performs the minimum number of operations
617 	 * before establishing the Subject, and logs errors.
618 	 */
619 	public void init(String[] configOptions, LifeCycle lifeCycle)
620 			throws Exception {
621 		logger.entering(this.getClass().toString(), "init");
622 		this.lifeCycle = lifeCycle;
623 		try {
624 			// Take the login context entry from the configuration file, if this
625 			// entry is null, server will start without a subject
626 			loginContext = (LoginContext) delegate.getDeploymentConfig().getEntry(
627 					PROVIDER, "loginContext", LoginContext.class, null);
628 			logger.finer("loginContext " + loginContext);
629 			if (loginContext == null) {
630 				logger.finer("Login Context was null when the service was Started");
631 				// Starting the Service with NO subject provided
632 				initAsSubject();
633 			} else {
634 				logger.finer("Login Context was not null when the service was Started");
635 				loginContext.login();
636 				logger.finer("Login Context subject= "
637 						+ loginContext.getSubject());
638 
639 				try {
640 					// Starting the Service with a subject
641 					Subject.doAsPrivileged(loginContext.getSubject(),
642 							new PrivilegedExceptionAction() {
643 								public Object run() throws Exception {
644 									initAsSubject();
645 									return null;
646 								}
647 							}, null);

810 	final static class ProxyVerifier implements TrustVerifier, Serializable {
811 		private final RemoteMethodControl serverProxy;
812 
813 		private final Uuid serverUuid;
814 
815 		/**
816 		 * Create the verifier, throwing UnsupportedOperationException if the
817 		 * server proxy does not implement both RemoteMethodControl and
818 		 * TrustEquivalence.
819 		 */
820 		public ProxyVerifier(Object serverProxy, Uuid serverUuid) {
821 			if (serverProxy instanceof RemoteMethodControl
822 					&& serverProxy instanceof TrustEquivalence) {
823 				this.serverProxy = (RemoteMethodControl) serverProxy;
824 			} else {
825 				throw new UnsupportedOperationException();
826 			}
827 			this.serverUuid = serverUuid;
828 		}

830 		/** Implement TrustVerifier */
831 		public boolean isTrustedObject(Object obj, Context ctx)
832 				throws RemoteException {
833 			if (obj == null || ctx == null) {
834 				throw new NullPointerException();
835 			} else if (!(obj instanceof ProxyAccessor)) {
836 				return false;
837 			} else if (!(obj instanceof ReferentUuid))
838 				return false;
839 
840 			if (!serverUuid.equals(((ReferentUuid) obj).getReferentUuid()))
841 				return false;
842 
843 			RemoteMethodControl otherServerProxy = (RemoteMethodControl) ((ProxyAccessor) obj)
844 					.getProxy();
845 			MethodConstraints mc = otherServerProxy.getConstraints();
846 			TrustEquivalence trusted = (TrustEquivalence) serverProxy
847 					.setConstraints(mc);
848 			return trusted.checkTrustEquivalence(otherServerProxy);
849 		}
850 	}
