59  	@Test
60  	public void batchTask3Test() throws Exception {
61  		// batch for the composition f1(f2(f3((x1, x2), f4(x1, x2)), f5(x1, x2))
62  		// shared dataContext with named paths
63  		Task batch3 = task("batch3",
64  				type(sig("multiply", MultiplierImpl.class, result("subtract/x1", Direction.IN)), Signature.PRE),
65  				type(sig("add", AdderImpl.class, result("subtract/x2", Direction.IN)), Signature.PRE),
66  				sig("subtract", SubtractorImpl.class, result("result/y", from("subtract/x1", "subtract/x2"))),
67  				context(in("multiply/x1", 10.0), in("multiply/x2", 50.0), 
68  						in("add/x1", 20.0), in("add/x2", 80.0)));
69  		
70  		batch3 = exert(batch3);
71  		//logger.info("task result/y: " + get(batch3, "result/y"));
72  		assertEquals("Wrong value for 400.0", 400.0, get(batch3, "result/y"));
73  	}
