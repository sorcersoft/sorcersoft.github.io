135     static void waitForDeployment(OperationalStringManager mgr) throws RemoteException, OperationalStringException, InterruptedException {
136         OperationalString opstring  = mgr.getOperationalString();
137         Map<ServiceElement, Integer> deploy = new HashMap<ServiceElement, Integer>();
138         int total = 0;
139         for (ServiceElement elem: opstring.getServices()) {
140             deploy.put(elem, 0);
141             total += elem.getPlanned();
142         }
143         int deployed = 0;
144         long sleptFor = 0;
145         List<String> deployedServices = new ArrayList<String>();
146         while (deployed < total && sleptFor< MAX_TIMEOUT) {
147             deployed = 0;
148             for (Map.Entry<ServiceElement, Integer> entry: deploy.entrySet()) {
149                 int numDeployed = entry.getValue();
150                 ServiceElement elem = entry.getKey();
151                 if (numDeployed < elem.getPlanned()) {
152                     numDeployed = mgr.getServiceBeanInstances(elem).length;
153                     deploy.put(elem, numDeployed);
154                 } else {
155                     String name = String.format("%s/%s", elem.getOperationalStringName(), elem.getName());
156                     if(!deployedServices.contains(name)) {
157                         System.out.println(String.format("Service %s/%-24s is deployed. Planned [%s], deployed [%d]",
158                                                          elem.getOperationalStringName(),
159                                                          elem.getName(),
160                                                          elem.getPlanned(),
161                                                          numDeployed));
162                         deployedServices.add(name);
163                     }
164                     deployed += elem.getPlanned();
165                 }
166             }
167             if(sleptFor==MAX_TIMEOUT)
168                 break;
169             if (deployed < total) {
170                 Thread.sleep(1000);
171                 sleptFor += 1000;
172             }
173         }
174 
175         if(sleptFor>=MAX_TIMEOUT && deployed < total)
176             throw new RuntimeException("Timeout waiting for service to be deployed");
177     }

184     @SuppressWarnings("unchecked")
185     static <T> T waitForService(Class<T> serviceType, long timeout) throws Exception{
186         int waited = 0;
187         JiniClient client = new JiniClient();
188         client.addRegistrarGroups(Sorcer.getLookupGroups());
189         Listener listener = new Listener(serviceType);
190         client.getDiscoveryManager().addDiscoveryListener(listener);
191 
192         while(listener.monitor.get()==null && waited < timeout) {
193             listener.lookup();
194             Thread.sleep(500);
195             waited++;
196         }
197         return (T) listener.monitor.get();
198     }

200     static class Listener implements DiscoveryListener {
201         final AtomicReference<Object> monitor = new AtomicReference<Object>();
202         private Class<?> serviceType;
203         private final List<ServiceRegistrar> lookups = new ArrayList<ServiceRegistrar>();
204         
205         public Listener(Class<?> providerInterface) {
206         	serviceType = providerInterface;
207         }
208 
209         void lookup() {
210             final ServiceTemplate template = new ServiceTemplate(null, new Class[]{serviceType}, null);
211             for(ServiceRegistrar registrar : lookups) {
212                 try {
213                     Object service = registrar.lookup(template);
214                     if(service!=null) {
215                         monitor.set(service);
216                         break;
217                     }
218                 } catch (RemoteException e) {
219                     e.printStackTrace();
220                 }
221             }
222         }
223 
224         public void discovered(DiscoveryEvent discoveryEvent) {
225             Collections.addAll(lookups, discoveryEvent.getRegistrars());
226         }
227 
228         public void discarded(DiscoveryEvent discoveryEvent) {
229         }
230     }
