60      //@Ignore("hangs")
61      // This type of tasks only work if there is a separate provider for the SRV signature and others for other sigs
62      @Test
63      public void batchTaskTest() throws Exception {
64          // batch for the composition f1(f2(f3((x1, x2), f4(x1, x2)), f5(x1, x2))
65          // shared context with named paths
66          Task batch3 = task("batch3",
67                  type(sig("multiply", Multiplier.class, result("subtract/x1", Direction.IN)), Signature.PRE),
68                  type(sig("add", Adder.class, result("subtract/x2", Direction.IN)), Signature.PRE),
69                  sig("subtract", Subtractor.class, result("result/y", from("subtract/x1", "subtract/x2"))),
70                  context(in("multiply/x1", 10.0), in("multiply/x2", 50.0),
71                          in("add/x1", 20.0), in("add/x2", 80.0)));
72  
73          batch3 = exert(batch3);
74          //logger.info("task result/y: " + get(batch3, "result/y"));
75          assertEquals("Wrong value for 400.0", 400.0, get(batch3, "result/y"));
76      }
