116 	public static Job getJobComposition() throws Exception {
117 		Task task1 = getAddTask();
118 		Task task2 = getMultiplyTask();
119 		Task task3 = getSubtractTask();
120 
121 		Job internal = new NetJob("2tasks");
122 		internal.addExertion(task2);
123 		internal.addExertion(task1);
124 		
125 		Job job = new NetJob("1job1task");
126 		job.addExertion(internal);
127 		job.addExertion(task3);
128 		
129 		// make the result of second task as the first argument of task
130 		// three
131 		task2.getContext().connect("out/value", "arg1/value", task3.getContext());
132 		// make the result of the first task as the second argument of task
133 		// three
134 		task1.getContext().connect("out/value", "arg2/value", task3.getContext());
135 		
136 		return job;

158 	private static Task getAddTask() throws Exception {
159 		Context context = new PositionalContext("add");
160 		context.putInValue("arg1/value", 20.0);
161 		context.putInValue("arg2/value", 80.0);
162 		// We know that the output is gonna be placed in this path
163 		context.putOutValue("out/value", 0);
164 		NetSignature method = new NetSignature("add", Adder.class);
165 		Task task = new NetTask("add", method);
166 		task.setContext(context);
167 		return task;
168 	}

182 	private static Task getSubtractTask() throws Exception {
183 		PositionalContext context = new PositionalContext("subtract");
184 		// We want to stick in the result of multiply in here
