40  public class ReggieHelper {
41      public static final int MAX_MATCHES = 64;
42  
43      public static ServiceItem[] lookup(
44              Class[] serviceTypes) throws RemoteException {
45          return lookup(serviceTypes, (String)null);
46      }
47  
48      public static ServiceItem[] lookup(
49              Class[] serviceTypes, String serviceName) throws RemoteException {
50          return lookup(null, serviceTypes, serviceName);
51      }
52  
53      public static ServiceItem[] lookup(ServiceRegistrar registrar,
54              Class[] serviceTypes, String serviceName) throws RemoteException {
55          ServiceRegistrar regie = null;
56          if (registrar == null) {
57              regie = NetworkShellAccessor.getNetworkShell(Thread.currentThread().getContextClassLoader()).getSelectedRegistrar();
58              if (regie == null)
59                  return null;
60          } else {
61              regie = registrar;
62          }
63  
64          ArrayList<ServiceItem> serviceItems = new ArrayList<ServiceItem>();
65          ServiceMatches matches = null;
66          Entry myAttrib[] = null;
67          if (serviceName != null) {
68              myAttrib = new Entry[1];
69              myAttrib[0] = new Name(serviceName);
70          }
71          ServiceTemplate myTmpl = new ServiceTemplate(null, serviceTypes,
72                  myAttrib);
73  
74          matches = regie.lookup(myTmpl, MAX_MATCHES);
75          serviceItems.addAll(Arrays.asList(matches.items).subList(0, Math.min(MAX_MATCHES, matches.totalMatches)));
76          ServiceItem[] sItems = new ServiceItem[serviceItems.size()];
77          return serviceItems.toArray(sItems);
78      }
