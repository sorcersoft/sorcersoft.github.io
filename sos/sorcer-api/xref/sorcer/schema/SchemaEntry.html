28  public class SchemaEntry implements Serializable {
29      private static final long serialVersionUID = -809999449914128883L;
30      public String path;
31      public Direction direction;
32      public boolean required;
33      public Class type;
34      public String tag;
35      public String description;
36      public boolean result;
37      public String[] returnArgPaths;
38  
39      protected SchemaEntry() {
40      }
41  
42      public SchemaEntry(Class type, String path, Direction direction, boolean required, String tag, String description, boolean result, String[] returnArgPaths) {
43          if (type == null) throw new IllegalArgumentException("type is null");
44          if (path == null && tag == null) throw new IllegalArgumentException("path is null and no tags");
45          if (direction == null) throw new IllegalArgumentException("direction is null");
46          this.type = type;
47          this.path = path;
48          this.direction = direction;
49          this.required = required;
50          this.tag = tag;
51          this.description = description;
52          this.result = result;
53          this.returnArgPaths = returnArgPaths;
54      }
55  
56      @Override
57      public boolean equals(Object obj) {
58          return (obj != null && obj == this) || (obj instanceof SchemaEntry && equals((SchemaEntry) obj));
59      }
60  
61      protected boolean equals(SchemaEntry entry) {
62          return compare(path, entry.path) && equals(tag, entry.tag);
63      }
64  
65      @Override
66      public int hashCode() {
67          return 13 * (path == null ? 0 : path.hashCode()) + 37 * (tag == null ? 0 : tag.hashCode());
68      }
69  
70      @Override
71      public String toString() {
72          return (required ? "required" : "optional") + " " + type.getName() + " " + path + " " + tag + " " + direction;
73      }
74  
75      private static boolean compare(Object a, Object b) {
76          return a == null ? b == null : a.equals(b);
77      }
78  
79      private static boolean equals(Object a, Object b) {
80          return a == null ? b == null : a.equals(b);
81      }
82  }
