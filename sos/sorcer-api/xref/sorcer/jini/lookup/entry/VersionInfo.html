18  package sorcer.jini.lookup.entry;

20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import net.jini.core.entry.Entry;
24  import net.jini.lookup.entry.ServiceInfo;
25  import sorcer.core.SorcerEnv;

27  /**
28   * @author Mike Sobolewski
29   * 
30   */
31  public class VersionInfo {
32  	public static final String PRODUCT_NAME = "SORCER Runtime Environmnet";
33  	public static final String EMAIL_CONTACT = "sobol@sorcersoft.org";
34  	public static final String SUPPLIER_NAME = "Sorcersoft.com S.A.";
35  	public static final String VERSION = SorcerEnv.getSorcerVersion();
36  	public static final String MODEL = "Sorcersoft.com";
37  
38  	public static String getVersion() {
39  		return PRODUCT_NAME + ", " + EMAIL_CONTACT + ", " + SUPPLIER_NAME
40  				+ ", " + VERSION + ", "
41  				+ System.getProperty("java.runtime.name") + " v."
42  				+ System.getProperty("java.runtime.version");
43  	}
44  
45  	/**
46  	 * Create the service owned attributes for a SORCER service provider
47  	 */
48  	public static List<Entry> productAttributesFor(String serviceName) {
49  		final Entry info = new ServiceInfo(serviceName == null
50  				|| serviceName.length() == 0 ? PRODUCT_NAME : serviceName,
51  				VersionInfo.EMAIL_CONTACT, VersionInfo.SUPPLIER_NAME,
52  				VersionInfo.VERSION, MODEL, "");
53  
54  		final Entry type = new com.sun.jini.lookup.entry.BasicServiceType(
55  				serviceName);
56  
57  		List<Entry> al = new ArrayList<Entry>(2);
58  		al.add(info);
59  		al.add(type);
60  		return al;
61  	}
