67      public static void initialize(String providerType) {
68          try {
69              logger.fine("SORCER DynamicAccessor provider: " + providerType);
70              Class type = Class.forName(providerType,true, Thread.currentThread().getContextClassLoader());
71              if(!DynamicAccessor.class.isAssignableFrom(type)){
72                  throw new IllegalArgumentException("Configured class must implement DynamicAccessor: "+providerType);
73              }
74              accessor = (DynamicAccessor) type.newInstance();
75          } catch (ClassNotFoundException e) {
76              throw new RuntimeException("No service accessor available for: " + providerType,e);
77          } catch (InstantiationException e) {
78              throw new RuntimeException("No service accessor available for: " + providerType,e);
79          } catch (IllegalAccessException e) {
80              throw new RuntimeException("No service accessor available for: " + providerType,e);
81          }
82      }

92      public static ServiceTemplate getServiceTemplate(ServiceID serviceID,
93              String providerName, Class[] serviceTypes,
94              String[] publishedServiceTypes) {
95          Class[] types;
96          List<Entry> attributes = new ArrayList<Entry>(2);
97  
98          if (providerName != null && !providerName.isEmpty() && !ANY.equals(providerName))
99              attributes.add(new Name(providerName));
100 
101         if (publishedServiceTypes != null) {
102             SorcerServiceInfo st = new SorcerServiceInfo();
103             st.publishedServices = publishedServiceTypes;
104             attributes.add(st);
105         }
106 
107         if (serviceTypes == null) {
108             types = new Class[] { Provider.class };
109         } else {
110             types = serviceTypes;
111         }
112 
113         logger.info("getServiceTemplate >> \n serviceID: " + serviceID
114                 + "\nproviderName: " + providerName + "\nserviceTypes: "
115                 + StringUtils.arrayToString(serviceTypes)
116                 + "\npublishedServiceTypes: "
117                 + StringUtils.arrayToString(publishedServiceTypes));
118 
119         return new ServiceTemplate(serviceID, types, attributes.toArray(new Entry[attributes.size()]));
120     }

122     private static String overrideName(String providerName, Class serviceType) {
123         if (providerName == null || "*".equals(providerName)) return providerName;
124         if (SorcerConstants.NAME_DEFAULT.equals(providerName))
125             providerName = providerNameUtil.getName(serviceType);
126         return SorcerEnv.getActualName(providerName);
127     }

217     private static void checkNullName(ServiceTemplate template) {
218         if (template.attributeSetTemplates == null)
219             return;
220         for (Entry attr : template.attributeSetTemplates) {
221             if (attr instanceof Name) {
222                 Name name = (Name) attr;
223                 if (ANY.equals(name.name)) {
224                     name.name = null;
225                     logger.log(Level.WARNING, "Requested service with name '*'", new IllegalArgumentException());
226                 }
227             } else if (attr instanceof SorcerServiceInfo) {
228                 SorcerServiceInfo info = (SorcerServiceInfo) attr;
229                 if (ANY.equals(info.providerName)) {
230                     info.providerName = null;
231                     logger.log(Level.WARNING, "Requested service with name '*'", new IllegalArgumentException());
232                 }
233             }
234         }
235     }

237     /**
238      * Test if provider is still replying.
239      *
240      * @param provider the provider to check
241      * @return true if a provider is alive, otherwise false
242      * @throws java.rmi.RemoteException
243      */
244     public static boolean isAlive(Provider provider)
245             throws RemoteException {
246         if (provider == null)
247             return false;
248         try {
249             provider.getProviderName();
250             return true;
251         } catch (RemoteException e) {
252             throw e;
253         }
