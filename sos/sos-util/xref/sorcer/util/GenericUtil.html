183      * This is a function that appends a File object to open jar archive
184      *
185      * @param fileObj
186      *            File or directory
187      * @param parentPath
188      *            Relative parent path of fileObj to be stored in the archive
189      * @param jarOut
190      *            Output stream connected to an open archive
191      */
192     public static void addToArchive(File fileObj, String parentPath,
193                                     JarOutputStream jarOut) {
194         FileInputStream in;
195         byte[] buffer = new byte[1024];
196         if (fileObj.isFile()) {
197             // dirObj is a file
198             try {
199                 String fileName = null;
200                 if (parentPath == null) {
201                     fileName = fileObj.getName();
202                 } else {
203                     fileName = parentPath + fileObj.getName();
204                 }
205                 in = new FileInputStream(fileObj);
206                 System.out.println("Adding: " + fileName);
207                 jarOut.putNextEntry(new JarEntry(fileName));
208                 int len = in.read(buffer);
209                 while (len > 0) {
210                     jarOut.write(buffer, 0, len);
211                     len = in.read(buffer);
212                 }
213                 jarOut.closeEntry();
214                 in.close();
215             } catch (IOException e) {
216                 e.printStackTrace();
217             }
218         } else if (fileObj.isDirectory()) {
219             // dirObj is a directory
220             File[] files = fileObj.listFiles();
221 
222             for (int i = 0; i < files.length; i++) {
223                 if (files[i].isDirectory()) {
224                     if (parentPath == null) {
225                         parentPath = fileObj.getName() + File.separator
226                                 + files[i].getName();
227                     } else {
228                         parentPath = parentPath + File.separator
229                                 + fileObj.getName() + File.separator
230                                 + files[i].getName();
231                     }
232                     addToArchive(files[i], parentPath, jarOut);
233                     continue;
234                 }
235 
236                 try {
237                     in = new FileInputStream(files[i]);
238                     String fileName = null;
239                     if (parentPath == null) {
240                         fileName = fileObj.getName() + File.separator
241                                 + files[i].getName();
242                     } else {
243                         fileName = parentPath + File.separator
244                                 + fileObj.getName() + File.separator
245                                 + files[i].getName();
246                     }
247                     System.out.println("Adding: " + fileName);
248                     jarOut.putNextEntry(new JarEntry(fileName));
249                     int len = in.read(buffer);
250                     while (len > 0) {
251                         jarOut.write(buffer, 0, len);
252                         len = in.read(buffer);
253                     }
254                     jarOut.closeEntry();
255                     in.close();
256                 } catch (FileNotFoundException e) {
257                     e.printStackTrace();
258                 } catch (IOException e) {
259                     e.printStackTrace();
260                 }
261             }
262         }
263         return;
264     }

451      * @param sourceFile Source file 
452      * @param destinationFile Destination file 
453      * @throws IOException
454      * @see redirectInputStream2File
455      */
456     public static void copyFile(File sourceFile, File destinationFile) throws IOException {
457         // If the destination file's parent directory does not exit, create it
458         destinationFile.getParentFile().mkdirs();
459 
460         //  Creating a new input file stream that is connected to the source file
461         InputStream is = new FileInputStream(sourceFile);
462         try {
463             // Attempting to redirect the input file stream of the source file to the destination file path 
464             redirectInputStream2File(is, destinationFile);
465         } catch (IOException e) {
466             // Caught an IOException 
467 
468             // Check if the destination file exist
469             if(destinationFile.exists()) {
470                 // If the destination file exist delete the file 
471                 destinationFile.delete();
472             }
473             // The input file stream is reset
474             is.reset();
475             // Making a second attempt to copy the source file 
476             redirectInputStream2File(is, destinationFile);
477             // If the redirection fails a second time an IO Exception is thrown
478         }
479     }
