	/**
	 * Return an index of this ListContext path.
	 * 
	 * @param path
	 *            ListContext dataContext path
	 * @return an index of the ListContext path
	 */
	public int pathIndex(String path) {
		int i1 = path.indexOf('[');
		if (i1 >= 0) {
			int i2 = path.indexOf(']');
			return Integer.parseInt(path.substring(i1 + 1, i2));
		}
		return -1;
	}

		String path;
		int i;
		elements.add(index, element);
		Enumeration en = contextPaths();
		while (en.hasMoreElements()) {
			path = (String)en.nextElement();
			i = pathIndex(path);
			if (i > index) {
				putValue(pathFor(i+1), elements.get(i+1));
			}
		}
		i = elements.size();
		// append last shifted already element
		putValue(pathFor(i), elements.get(i-1));
	}
