185 			NoSuchMethodException, IllegalArgumentException,
186 			InstantiationException, IllegalAccessException,
187 			InvocationTargetException {
188 		Constructor<? extends Object> constructor;
189 		Class<? extends Object> sigClass = signature.getClass();
190 		constructor = sigClass.getConstructor(signature.getClass());
191 		Object sig = constructor.newInstance(signature);

242 		String[] newContextTemplateIDs;
243 		if (contextTemplateIDs == null) {
244 			newContextTemplateIDs = new String[1];
245 			newContextTemplateIDs[0] = id;
246 		} else {
247 			newContextTemplateIDs = new String[contextTemplateIDs.length + 1];
248 			int i;
249 			for (i = 0; i < contextTemplateIDs.length; i++)
250 				newContextTemplateIDs[i] = contextTemplateIDs[i];
251 			newContextTemplateIDs[i] = id;
252 		}
253 		contextTemplateIDs = newContextTemplateIDs;
254 		// selfModified();
255 	}

257 	public void removeContextTemplateID(String id) {
258 		if (contextTemplateIDs == null)
259 			return;
260 		List<String> v = new ArrayList<String>();
261 		for (int i = 0; i < contextTemplateIDs.length; i++)
262 			v.add(contextTemplateIDs[i]);
263 		v.remove(id);
264 		if (contextTemplateIDs.length != v.size()) {
265 			contextTemplateIDs = new String[v.size()];
266 			v.toArray(contextTemplateIDs);
267 			// selfModified();
268 		}
269 	}

271 	public void setSelector(String selector) {
272 		if (selector != null) {
273 			if (selector.indexOf("#") > 0) {
274 				StringTokenizer token = new StringTokenizer(selector, "#");
275 				this.selector = token.nextToken();
276 				prefix = token.nextToken();
277 			} else if (selector.equals("new")) {
278 				this.selector = null;
279 			} else {
280 				this.selector = selector;
281 			}
282 		} else {
283 			this.selector = null;
284 		}
285 	}

353 	/**
354 	 * Returns a method provided by the requestor itself to substitute the
355 	 * existing provider'smethod. The implementation of this requestor's method
356 	 * overrides the existing implementation of the provider if one exists.Thus
357 	 * a new functionality is inserted into the executing provider if the
358 	 * alternative inserted method is acceptable (valid).
359 	 */
360 	public Method getSubstituteMethod(Class<?>[] argTypes) {
361 		Method m = null;
362 		try {
363 			Class<?> clazz = getSubstituteClass();
364 			if (clazz == null)
365 				return null;
366 			m = clazz.getMethod(selector, argTypes);
367 		} catch (NoSuchMethodException nsme) {
368 			nsme.printStackTrace();
369 			// logger.info("The method: \"" + selector
370 			// + "\" doesn't exist in requestor's signature code");
371 		}
372 		return m;
373 	}
