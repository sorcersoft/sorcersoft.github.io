35  	public SpaceTaskDispatcher(final NetTask task, 
36              final Set<Context> sharedContexts,
37              final boolean isSpawned, 
38              final LokiMemberUtil myMemberUtil,
39              final ProvisionManager provisionManager) throws ExertionException, SignatureException {
40  
41  		this.provisionManager = provisionManager;
42          this.xrt = task;
43  		subject = task.getSubject();
44  		this.sharedContexts = sharedContexts;
45  		this.isSpawned = isSpawned;
46  		isMonitored = task.isMonitorable();
47  		state = RUNNING;
48  		dispatchers.put(task.getId(), this);
49  		dispatchExertions();
50  		
51  		disatchGroup = new ThreadGroup("task-"+ task.getId());
52  		disatchGroup.setDaemon(true);
53  		disatchGroup.setMaxPriority(Thread.NORM_PRIORITY - 1);
54  
55  		CollectResultThread crThread = new CollectResultThread(disatchGroup);
56  		crThread.start();
57  
58  		CollectFailThread cfThread = new CollectFailThread(disatchGroup);
59  		cfThread.start();
60  
61  		CollectErrorThread efThread = new CollectErrorThread(disatchGroup);
62  		efThread.start();
63  
64  		this.myMemberUtil = myMemberUtil;
65  	}

67  	public void dispatchExertions() throws ExertionException,
68  			SignatureException {
69  		reconcileInputExertions(xrt);
70  		logger.finer("space task: " + xrt);
71  		try {
72  			writeEnvelop(xrt);
73  			logger.finer("written task ==> SPACE EXECUTE TASK: "
74  					+ xrt.getName());
75  		} catch (ProvisioningException pe) {
76              xrt.setStatus(FAILED);
77              throw new ExertionException(pe.getLocalizedMessage());
78          } catch (RemoteException re) {
79  			re.printStackTrace();
80  			logger.severe("Space not reachable... resetting space");
81  			space = SpaceAccessor.getSpace();
82  			if (space == null) {
83  				xrt.setStatus(FAILED);
84  				throw new ExertionException("NO exertion space available!");
85  			}
86  		}
87  	}

89  	public void collectResults() throws ExertionException, SignatureException {
90  		ExertionEnvelop temp;
91  		temp = ExertionEnvelop.getTemplate();
92  		temp.exertionID = xrt.getId();
93  		temp.state = new Integer(DONE);
94  
95  		logger.finer("<===================== template for space task to be collected: \n"
96  				+ temp.describe());
97  
98  		ExertionEnvelop resultEnvelop = takeEnvelop(temp);
99  		logger.finer("collected result envelope  <===================== \n"
100 				+ resultEnvelop.describe());
101 
102 		NetTask result = (NetTask) resultEnvelop.exertion;
103 		state = DONE;
104 		notifyExertionExecution(xrt, xrt, result);
105 		result.setStatus(DONE);
106 		dispatchers.remove(xrt.getId());
107 		xrt = result;
108 	}

115

117 				+ template.describe());
118 
119 		ExertionEnvelop resultEnvelop = takeEnvelop(template);
120 		if (resultEnvelop != null) {
121 			NetTask result = (NetTask) resultEnvelop.exertion;
122 			state = FAILED;
123 			notifyExertionExecution(xrt, xrt, result);
124 			result.setStatus(FAILED);
125 			xrt = result;
126 		}
127 		dispatchers.remove(xrt.getId());
128 	}
