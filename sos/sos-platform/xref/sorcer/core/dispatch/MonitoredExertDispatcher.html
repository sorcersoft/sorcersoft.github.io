57      public MonitoredExertDispatcher(Exertion exertion,
58                                      Set<Context> sharedContext, boolean isSpawned, Provider provider)
59  			throws Throwable {
60  		super(exertion, sharedContext, isSpawned, provider);
61  				
62  		if (sessionMonitor == null)
63  			sessionMonitor = Accessor.getService(SorcerConstants.NAME_DEFAULT, MonitoringManagement.class);
64  		if (lrm == null)
65  			lrm = new LeaseRenewalManager();
66  
67  		// get the exertion with the monitor session
68  		this.xrt = (ServiceExertion)register(exertion);
69  
70  		// make the monitor session of this exertion active
71  		ExertDispatcher.logger.log(Level.FINER, "Dispatching task now: " + xrt.getName());
72  		MonitoringSession session = xrt.getMonitorSession();
73  		session.init((Monitorable) provider.getProxy(), LEASE_RENEWAL_PERIOD,
74  				DEFAULT_TIMEOUT_PERIOD);
75  		lrm.renewUntil(session.getLease(), Lease.ANY, null);
76  		
77  		dThread = new DispatchThread();
78  		try {
79  			dThread.start();
80  			dThread.join();
81  		} catch (InterruptedException ie) {
82  			ie.printStackTrace();
83  			state = ExecState.FAILED;
84  		}
85  	}

87  	private Exertion register(Exertion exertion) throws Exception {
88  		RemoteEventListener l = (RemoteEventListener) (new ResultListener()
89  				.export());
90  		ServiceExertion registeredExertion = (ServiceExertion) (sessionMonitor.register(l,
91  				exertion, LEASE_RENEWAL_PERIOD));
92  
93  		MonitoringSession session = registeredExertion.getMonitorSession();
94  		ExertDispatcher.logger.info("Session for the exertion =" + session);
95  		ExertDispatcher.logger.info("Lease to be renewed for duration ="
96  				+ (session.getLease().getExpiration() - System
97  						.currentTimeMillis()));
98  		lrm.renewUntil(session.getLease(), Lease.ANY, null);
99  		return registeredExertion;
100 	}

102 	protected void preExecExertion(Exertion exertion) throws ExertionException {
103 		try {
104 			exertion.getControlContext().appendTrace(provider.getProviderName() 
105 					+ " dispatcher: " + getClass().getName());
106 		} catch (RemoteException e) {
107 			// ignore it, local call
108 			}
109 		ExertionSessionInfo.add((ServiceExertion)exertion);
110 		//Provider is expecting exertion field in Context to be set.
111 		xrt.getDataContext().setExertion(xrt);
112 		updateInputs(exertion);
113 		((ServiceExertion)exertion).startExecTime();
114 		((ServiceExertion) exertion).setStatus(ExecState.RUNNING);
115 	}

117 	protected void postExecExertion(Exertion result) throws ExertionException {
118 		ServiceExertion sxrt = (ServiceExertion)result;
119 		if (sxrt instanceof NetJob )
120 			((NetJob)xrt).setExertionAt(result, result.getIndex());
121 		else
122 			xrt = sxrt;
123 		
124 		if (sxrt.getStatus() > ExecState.ERROR && sxrt.getStatus() != ExecState.SUSPENDED) {
125 			sxrt.setStatus(ExecState.DONE);
126 			collectOutputs(result);
127 		}
128 		if (sxrt.getStatus() != ExecState.DONE)
129 			state = sxrt.getStatus();
130 		
131 		xrt.stopExecTime();
132 		ExertDispatcher.dispatchers.remove(xrt.getId());
133 		ExertionSessionInfo.removeLease();
134 		dThread.stop = true;
135 	}

137 	protected void execExertion(Exertion exertion) throws ExertionException,
138 			SignatureException, RemoteException {
139 		ServiceExertion ei = (ServiceExertion) exertion;
140 		preExecExertion(exertion);
141 
142 		ExertDispatcher.logger.log(Level.INFO,
143 				"Prexec Exertion done .... noe executing exertion");
144 		try {
145 			if (ei.isTask())
146 				execTask((NetTask) exertion);
147 			else
148 				throw new ExertionException("Functionality not yet supported");
149 		} catch (Exception ex) {
150 			ex.printStackTrace();
151 			ei.reportException(ex);
152 			try {
153 				if (ei.isTask())
154 					((ServiceExertion) exertion).getMonitorSession()
155 							.changed(exertion.getDataContext(), ExecState.Category.FAILED);
156 			} catch (Exception ex0) {
157 				ex0.printStackTrace();
158 			} finally {
159 				((ServiceExertion)exertion).setStatus(ExecState.FAILED);
160 			}
161 		}
162 
163 	}

167 	private void execTask(NetTask task) throws ExertionException,
168 			SignatureException {
169 
170 		ExertDispatcher.logger.log(Level.INFO, "start executing task");
171 		try {
172 			Service provider = (Service) Accessor.getService(task
173                     .getProcessSignature().getProviderName(), task
174                     .getServiceType());
175 			ExertDispatcher.logger.log(Level.INFO, "got a provider:" + provider);
176 
177 			if (provider == null) {
178 				String msg;
179 				// get the PROCESS Method and grab provider name + interface
180 				NetSignature method = (NetSignature) task
181 						.getProcessSignature();
182 				msg = "No Provider available. Provider Name: "
183 						+ method.getProviderName() + " Provider interface: "
184 						+ method.getServiceType();
185 				System.err.println(msg);
186 				throw new ExertionException(msg, task);
187 			} else {
188 				// setTaskProvider(task, provider.getProviderName());
189 				ExertDispatcher.logger.log(Level.INFO, "Servicing task now ..............");
190 				MonitoringSession session = task
191 						.getMonitorSession();
192 				session.init((Monitorable) provider, LEASE_RENEWAL_PERIOD,
193 						DEFAULT_TIMEOUT_PERIOD);
194 				lrm.renewUntil(session.getLease(), Lease.ANY, null);
195 
196 				task.setServicer(provider);
197 				provider.service(task, null);
198 			}
199 		} catch (RemoteException re) {
200 			re.printStackTrace();
201 			ExertDispatcher.logger.log(Level.SEVERE, "dispatcher execution failed for task: "
202 					+ task);
203 			throw new ExertionException("Remote Exception while executing task");
204 		} catch (MonitorException mse) {
205 			mse.printStackTrace();
206 			logger.log(Level.SEVERE, "dispatcher execution failed for task: "
207 					+ task);
208 			throw new ExertionException("Remote Exception while executing task");
209 		} catch (TransactionException te) {
210 			te.printStackTrace();
211 			logger.log(Level.SEVERE, "dispatcher execution failed for task: "
212 					+ task);
213 			throw new ExertionException("Remote Exception while executing task");
214 		}
215 	}
