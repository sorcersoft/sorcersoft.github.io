271 			throws ExertionException {
272 		ServiceContext fromContext;
273 		String toPath = null, newToPath = null, toPathcp, fromPath = null;
274 		int argIndex = -1;
275 		try {
276 			Hashtable toInMap = Contexts.getInPathsMap(toContext);
277 //			logger.info("**************** updating inputs in dataContext toContext = " + toContext);
278 //			logger.info("**************** updating based on = " + toInMap);

281 				// find argument for parametric dataContext

293 				fromContext = getSharedContext(fromPath, Contexts.getContextParameterID(toPathcp));
294 //				logger.info("**************** fromContext = " + fromContext);
295 //				logger.info("**************** before updating toContext: " + toContext
296 //						+ "\n>>> TO path: " + toPath + "\nfromContext: "
297 //						+ fromContext + "\n>>> FROM path: " + fromPath);
298 				if (fromContext != null) {
299 //					logger.info("**************** updating toContext: " + toContext
300 //							+ "\n>>> TO path: " + toPath + "\nfromContext: "
301 //							+ fromContext + "\n>>> FROM path: " + fromPath);
302 					// make parametric substitution if needed
303 					if (argIndex >=0 ) {
304 						Object args = toContext.getValue(Context.PARAMETER_VALUES);
305 						if (args.getClass().isArray()) {
306 							if (Array.getLength(args) > 0) {

314 						}
315 					} else {
316 						// make contextual substitution
317 						Contexts.copyValue(fromContext, fromPath, toContext, toPath);
318 					}
319 //					logger.info("**************** updated dataContext:\n" + toContext);
320 				}
321 			}
322 		} catch (Exception ex) {
323 			throw new ExertionException("Failed to update data dataContext: " + toContext.getName()
324 					+ " at: " + toPath + " from: " + fromPath, ex);

328 	private Tuple2<String, Integer> getPathIndex(String path) {
329 		int index = -1;
330 		String newPath = null;
331 		int i1 = path.lastIndexOf('/');
332 		String lastAttribute = path.substring(i1+1);
333 		if (lastAttribute.charAt(0) == '[' && lastAttribute.charAt(lastAttribute.length()-1) == ']') {
334 			index = Integer.parseInt(lastAttribute.substring(1, lastAttribute.length()-1));
335 			newPath = path.substring(0, i1+1);
336 		}
337 		return new Tuple2<String, Integer>(newPath, index);
338 	}

340 	protected ServiceContext getSharedContext(String path, String id) {
341 		// try to get the dataContext with particular id.
342 		// If not found, then find a dataContext with particular path.
343 		Context hc;
344 		if (Context.EMPTY_LEAF.equals(path) || "".equals(path))
345 			return null;
346 		if (id != null && id.length() > 0) {
347 			Iterator<Context> it = sharedContexts.iterator();
348 			while (it.hasNext()) {
349 				hc = it.next();
350 				if (UuidFactory.create(id).equals(hc.getId()))
351 					return (ServiceContext)hc;
352 			}
353 		}
354 		else {
355 			Iterator<Context> it = sharedContexts.iterator();
356 			while (it.hasNext()) {
357 				hc = it.next();
358 				if (hc.containsPath(path))
359 					return (ServiceContext)hc;
360 			}
361 		}
362 		return null;
363 	}

378 			ServiceExertion outTask) {
379 		// notify o MASTER task completion
380 		Vector recipients = null;
381 		String notifyees = parent.getControlContext().getNotifyList(inTask);
382 		if (notifyees != null) {
383 			String[] list = StringUtils.tokenize(notifyees, SorcerConstants.MAIL_SEP);
384 			recipients = new Vector(list.length);
385 			for (int i = 0; i < list.length; i++)
386 				recipients.addElement(list[i]);
387 		}
388 
389 		String to = "", admin = SorcerEnv.getProperty("sorcer.admin");
390 		if (recipients == null) {
391 			if (admin != null) {
392 				recipients = new Vector();
393 				recipients.addElement(admin);
394 			}
395 		} else if (admin != null && !recipients.contains(admin))
396 			recipients.addElement(admin);
397 
398 		if (recipients == null || recipients.size() == 0)
399 			return;
400 
401 		StringBuffer sb;
402 		if (inTask == masterXrt)
403 			sb = new StringBuffer("SORCER Master Task ");
404 		else
405 			sb = new StringBuffer("SORCER Task ");
406 
407 		sb.append(outTask.getName()).append("\n\nDescription:\n").append(
408 				outTask.getDescription());
409 		if (outTask.getStatus() > 0)
410 			sb
411 					.append("\n\nTask executed sucessfully with the input dataContext:\n");
412 		else
413 			sb.append("\n\nTask execution FAILED; The input dataContext was:\n");
414 		sb.append(inTask.contextToString()).append(
415 				"\n\nincluding the following output dataContext:\n").append(
416 				outTask.getDataContext());
417 		sb.append("\n\nincluding the specific output paths:\n").append(
418 				Contexts.getFormattedOut(outTask.getDataContext(), true));
419 
420 		for (int i = 0; i < recipients.size() - 1; i++)
421 			to = to + recipients.elementAt(i) + ",";
422 
423 		to = to + recipients.lastElement();
424 
425 		// sendMailWithSubject(sb.toString(), outTask.getName(), to);
426 	}

466 	protected void reconcileInputExertions(Exertion ex) {
467 		ServiceExertion ext = (ServiceExertion)ex;
468 		if (ext.getStatus() == DONE) {
469 			collectOutputs(ex);
470 			if (inputXrts != null)
471 				inputXrts.removeElement(ex);
472 		} else {
473 			ext.setStatus(INITIAL);
474 			if (ex.isJob()) {
475 				for (int i = 0; i < ((Job) ex).size(); i++)
476 					reconcileInputExertions(((Job) ex).exertionAt(i));
477 			}
