50  /**
51   * The Invoker defines context driven invocations on a parameter context
52   * (invoke context) containing its parameter names (paths) and arguments
53   * (values). The requested invocation is specified by the own invoke context and
54   * eventual context of parameters (Par).
55   * 
56   * The semantics for how parameters can be declared and how the arguments get
57   * passed to the parameters of callable unit are defined by the language, but
58   * the details of how this is represented in any particular computing system
59   * depend on the calling conventions of that system. A context-driven computing
60   * system defines callable unit called invokers used within a scope of service
61   * contexts, data structures defined in SORCER.
62   * 
63   * An invoke context is dictionary (associative array) composed of a collection
64   * of (key, value) pairs, such that each possible key appears at most once in
65   * the collection. Keys are considered as parameters and values as arguments of
66   * the service invokers accepting service contexts as their input data. A key is
67   * expressed by a path of attributes like directories in paths of a file system.
68   * Paths define a namespace of the context parameters. A context argument is any
69   * object referenced by its path or returned by a context invoker referenced by
70   * its path inside the context. An ordered list of parameters is usually
71   * included in the definition of an invoker, so that, each time the invoker is
72   * called, the context arguments for that call can be assigned to the
73   * corresponding parameters of the invoker. The context values for all paths
74   * inside the context are defined explicitly by corresponding objects or
75   * calculated by corresponding invokers. Thus, requesting a value for a path in
76   * a context is a computation defined by a invoker composition within the scope
77   * of the context.
78   */

236 	 * @param name

244 	public Invoker addPar(Object par) throws EvaluationException,
245 			RemoteException {
246 		if (par instanceof Par) {
247 			invokeContext.put(((Par)par).getName(), par);
248 			if (((Par)par).asis() instanceof Invoker) {
249 			((Invoker) ((Par)par).getValue()).addObserver(this);
250 			pars.add((Par)par);
251 			value = null;
252 			setChanged();
253 			notifyObservers(this);
254 			valueValid(false);
255 			}
256 		} else if (par instanceof Identifiable) {
257 			Par p = new ParImpl(((Identifiable)par).getName(), par, invokeContext);
258 			try {
259 				invokeContext.putValue(p.getName(), p);
260 			} catch (ContextException e) {
261 				throw new EvaluationException(e);
262 			}
263 		}
264 		return this;
265 	}

303 	public T invoke(Context context, Arg... entries)
304 			throws RemoteException, InvocationException {
305 		try {
306 			if (invokeContext == null)
307 				invokeContext = (ParModel) context;
308 			else {
309 				invokeContext.append(context);
310 			}
311 		} catch (ContextException e) {
312 			throw new InvocationException(e);
313 		}
314 		if (evaluator != null)
315 			return (T) invokeEvaluator();
316 		else
317 			return invoke(entries);
318 	}

320 	public T invoke(Arg... entries) throws RemoteException,
321 			InvocationException {
322 		try {
323 			if (entries != null && entries.length > 0) {
324 				valueIsValid = false;
325 				if (invokeContext == null)
326 					invokeContext = new ParModel("model/par");
327 					
328 				invokeContext.substitute(entries);
329 			}
330 			if (invokeContext.isContextChanged()) {
331 				valueIsValid = false;
332 				pars.clearArgs();
333 			}
334 			if (valueIsValid)
335 				return value;
336 			else {
337 				if (evaluator != null)
338 					value = (T) invokeEvaluator(entries);
339 				else
340 					value = getValue(entries);
341 
342 				valueValid(true);
343 			}
344 		} catch (Exception e) {
345 			throw new InvocationException(e);
346 		}
347 		return value;
348 	}

350 	private Object invokeEvaluator(Arg... entries) throws InvocationException {
351 		init(pars);
352 		try {
353 			evaluator.addArgs(pars);
354 			return evaluator.getValue(entries);
355 		} catch (Exception e) {
356 			throw new InvocationException(e);
357 		}
358 	}
