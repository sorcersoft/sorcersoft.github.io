36  /**
37   * A <code>Task</code> is an elementary service-oriented message
38   * {@link sorcer.service.Exertion} (with its own service {@link sorcer.service.Context} and a collection of
39   * service {@link Signature}s. Signatures of four
40   * {@link sorcer.service.Signature.Type}s can be associated with each task:
41   * <code>SERVICE</code>, <code>PREPROCESS</code>, <code>POSTROCESS</code>, and
42   * <code>APPEND</code>. However, only a single <code>PROCESS</code> signature
43   * can be associated with a task but multiple preprocessing, postprocessing, and
44   * context appending methods can be added.
45   * 
46   * @see Exertion
47   * @see Job
48   * 
49   * @author Mike Sobolewski
50   */

100 	public static Task newTask(String name, Signature signature, Context context)
101 			throws SecurityException, NoSuchMethodException,
102 			IllegalArgumentException, InstantiationException,
103 			IllegalAccessException, InvocationTargetException {
104 		Class<? extends Task> taskClass = null;
105 		if (signature.getClass() == ObjectSignature.class) {
106 			taskClass = ObjectTask.class;
107 		} else if (signature.getClass() == NetSignature.class) {
108 			taskClass = NetTask.class;
109 		}
110 		Constructor<? extends Task> constructor;
111 		constructor = taskClass.getConstructor(String.class,
112 				Signature.class, Context.class);
113 		if (signature.getPrefix() != null)
114             ((ServiceContext)context).setPrefix(signature.getPrefix());
115         if (signature.getReturnPath() != null)
116             ((ServiceContext) context)
117                     .setReturnPath(signature.getReturnPath());
118 
119 		return constructor.newInstance(name, signature, context);
120 	}

176 	public String toString() {
177 		if (innerTask != null) {
178 			return innerTask.toString();
179 		}
180 		StringBuilder sb = new StringBuilder(
181 				"\n=== START PRINTNIG TASK ===\nExertion Description: "
182 						+ getClass().getName() + ":" + name);
183 		sb.append("\n\tstatus: ").append(getStatus());
184 		sb.append(", task ID=");
185 		sb.append(getId());
186 		sb.append(", description: ");
187 		sb.append(description);
188 		sb.append(", priority: ");
189 		sb.append(priority);
190 		// .append( ", Index=" + getIndex())
191 		sb.append(", AccessClass=");
192 		sb.append(getAccessClass()).append(
193 				// ", isExportControlled=" + isExportControlled()).append(
194 				", providerName: ");
195 		sb.append(getProcessSignature().getProviderName());
196 		sb.append(", principal: ").append(getPrincipal());
197 		sb.append(", serviceType: ").append(getServiceType());
198 		sb.append(", selector: ").append(getSelector());
199 		sb.append(", parent ID: ").append(parentId);
200 
201 		if (signatures.size() == 1) {
202 			sb.append(getProcessSignature().getProviderName());
203 		} else {
204 			for (Signature s : signatures) {
205 				sb.append("\n  ").append(s);
206 			}
207 		}
208 		String time = getControlContext().getExecTime();
209 		if (time != null && time.length() > 0)
210 			sb.append("\n\texec time=").append(time);
211 		sb.append(controlContext).append("\n");
212 		sb.append(dataContext);
213 		sb.append("\n=== DONE PRINTING TASK ===\n");
214 
215 		return sb.toString();
216 	}

218 	public String describe() {
219 		StringBuilder sb = new StringBuilder(this.getClass().getName() + ": "
220 				+ name);
221 		sb.append(" task ID: ").append(getId()).append("\n  process sig: ")
222 				.append(getProcessSignature());
223 		sb.append("\n  status: ").append(getStatus());
224 		String time = getControlContext().getExecTime();
225 		if (time != null && time.length() > 0)
226 			sb.append("\n  exec time: ").append(time);
227 		return sb.toString();
228 	}
