196         // Changes below fix master exertions 03.09.2013 PR
197 		Uuid contextId = null;
198 		try {
199 			//contextName = (String) controlContext.getValue(ControlContext.MASTER_EXERTION);

212 					master = get(i);

215 			}
216 			return master;

498         String[] attributes = StringUtils.tokenizerSplit(path, CPS);
499 		// remove the leading attribute of the current exertion
500 		if (attributes[0].equals(getName())) {
501 			String[] attributes1 = new String[attributes.length - 1];
502 			System.arraycopy(attributes, 1, attributes1, 0,
503 					attributes.length - 1);
504 			attributes = attributes1;
505 		}
506 		String last = attributes[0];
507 		Exertion exti = this;
508 		for (String attribute : attributes) {
509 			if (((ServiceExertion) exti).hasChild(attribute)) {
510 				exti = ((Job) exti).getChild(attribute);
511 				if (exti instanceof Task) {
512 					last = attribute;
513 					break;
514 				}
515 			} else {
516 				break;
517 			}
518 		}
519 		int index = path.indexOf(last);
520 		String contextPath = path.substring(index + last.length() + 1);
521 
522 		return exti.getContext().getValue(contextPath);
