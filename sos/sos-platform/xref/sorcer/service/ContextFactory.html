203     public static void popultePositionalContext(PositionalContext pcxt,
204                                                  List<Tuple2<String, ?>> entryList) throws ContextException {
205         for (int i = 0; i < entryList.size(); i++) {
206             if (entryList.get(i) instanceof InEntry) {
207                 Object par = ((InEntry)entryList.get(i)).value();
208                 if (par instanceof Scopable) {
209                     try {
210                         ((Scopable)par).setScope(pcxt);
211                     } catch (RemoteException e) {
212                         throw new ContextException(e);
213                     }
214                 }
215                 if (((InEntry) entryList.get(i)).isPersistant) {
216                     setPar(pcxt, (InEntry) entryList.get(i), i);
217                 } else {
218                     pcxt.putInValueAt(((InEntry) entryList.get(i)).path(),
219                             ((InEntry) entryList.get(i)).value(), i + 1);
220                 }
221             } else if (entryList.get(i) instanceof OutEntry) {
222                 if (((OutEntry) entryList.get(i)).isPersistant) {
223                     setPar(pcxt, (OutEntry) entryList.get(i), i);
224                 } else {
225                     pcxt.putOutValueAt(((OutEntry) entryList.get(i)).path(),
226                             ((OutEntry) entryList.get(i)).value(), i + 1);
227                 }
228             } else if (entryList.get(i) instanceof InoutEntry) {
229                 if (((InoutEntry) entryList.get(i)).isPersistant) {
230                     setPar(pcxt, (InoutEntry) entryList.get(i), i);
231                 } else {
232                     pcxt.putInoutValueAt(
233                             ((InoutEntry) entryList.get(i)).path(),
234                             ((InoutEntry) entryList.get(i)).value(), i + 1);
235                 }
236             } else if (entryList.get(i) instanceof Entry) {
237                 if (((Entry) entryList.get(i)).isPersistant) {
238                     setPar(pcxt, (Entry) entryList.get(i), i);
239                 } else {
240                     pcxt.putValueAt((entryList.get(i)).path(),
241                             (entryList.get(i)).value(), i + 1);
242                 }
243             } else if (entryList.get(i) instanceof DataEntry) {
244                 pcxt.putValueAt(Context.DSD_PATH,
245                         ((DataEntry) entryList.get(i)).value(), i + 1);
246             } else if (entryList.get(i) instanceof Tuple2) {

254                     setPar(pcxt, (Tuple2<String, ?>) entryList.get(i), i);
255                 } else {

266     public static void populteContext(Context cxt,
267                                        List<Tuple2<String, ?>> entryList) throws ContextException {
268         for (int i = 0; i < entryList.size(); i++) {
269             if (entryList.get(i) instanceof InEntry) {
270                 if (((InEntry) entryList.get(i)).isPersistant) {
271                     setPar(cxt, (InEntry) entryList.get(i));
272                 } else {
273                     cxt.putInValue((entryList.get(i)).path(),
274                             (entryList.get(i)).value());
275                 }
276             } else if (entryList.get(i) instanceof OutEntry) {
277                 if (((OutEntry) entryList.get(i)).isPersistant) {
278                     setPar(cxt, (OutEntry) entryList.get(i));
279                 } else {
280                     cxt.putOutValue((entryList.get(i)).path(),
281                             (entryList.get(i)).value());
282                 }
283             } else if (entryList.get(i) instanceof InoutEntry) {
284                 if (((InoutEntry) entryList.get(i)).isPersistant) {
285                     setPar(cxt, (InoutEntry) entryList.get(i));
286                 } else {
287                     cxt.putInoutValue((entryList.get(i)).path(),
288                             (entryList.get(i)).value());
289                 }
290             } else if (entryList.get(i) instanceof Entry) {
291                 if (((Entry) entryList.get(i)).isPersistant) {
292                     setPar(cxt, entryList.get(i));
293                 } else {
294                     cxt.putValue((entryList.get(i)).path(),
295                             (entryList.get(i)).value());
296                 }
297             } else if (entryList.get(i) instanceof DataEntry) {
298                 cxt.putValue(Context.DSD_PATH,

318     private static void setPar(PositionalContext pcxt, Tuple2 entry, int i)
319             throws ContextException {
320 
321         Par p = new ParImpl(entry.path(), entry.value());
322         p.setPersistent(true);
323         if (entry.datastoreURL != null)
324             p.setDbURL(entry.datastoreURL);
325         if (entry instanceof InEntry)
326             pcxt.putInValueAt(entry.path(), p, i + 1);
327         else if (entry instanceof OutEntry)
328             pcxt.putOutValueAt(entry.path(), p, i + 1);
329         else if (entry instanceof InoutEntry)
330             pcxt.putInoutValueAt(entry.path(), p, i + 1);
331         else
332             pcxt.putValueAt(entry.path(), p, i + 1);
333     }

347             cxt.putInoutValue(entry.path(), p);
348         else
349             cxt.putValue(entry.path(), p);
