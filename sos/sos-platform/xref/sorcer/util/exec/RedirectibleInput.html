6   package sorcer.util.exec;
7   
8   import java.io.IOException;
9   import java.io.OutputStream;
10  
11  /**
12   * Input source capable of redirecting
13   * the data to an output stream. It means that instead of reading data into an
14   * array and then feeding it to the output stream, the data may be moved
15   * directly from the input to the output.
16   *
17   * @author Dawid Kurzyniec
18   * @version 1.0
19   */
20  public interface RedirectibleInput extends Input {
21  
22      /**
23       * Reads and redirects up to the <code>len</code> bytes of data to a
24       * specified output stream. Returns the number of bytes actually redirected.
25       *
26       * @param dest the destination stream
27       * @param len the maximum number of bytes to redirect
28       * @return number of bytes actually redirected
29       * @throws IOException if I/O error occurs
30       */
31      int redirect(OutputStream dest, int len) throws IOException;
32  
33      /**
34       * Readirects all further data from this input into the specified output
35       * stream, until EOF. Returns the number of bytes actually redirected.
36       *
37       * @param dest the destination stream
38       * @return number of bytes actually redirected
39       * @throws IOException if I/O error occurs
40       */
41      int redirectAll(OutputStream dest) throws IOException;
42  }
