59  	public void makeConnection() {
60  		// create the connection to the servlet
61  		try {
62  
63  			if (principal == null) {
64  				System.out
65  						.println(getClass().getName()
66  								+ "::makeConnection() NULL PRINCIPAL getting from Launcher");
67  
68  				if (Launcher.model != null)
69  					if (Launcher.model.getPrincipal() == null)
70  						System.out
71  								.println(getClass().getName()
72  										+ "::makeConnection() Launcher.model.principal == null");
73  					else
74  						principal = Launcher.model.getPrincipal();
75  				if (principal != null)
76  					System.out.println(getClass().getName()
77  							+ "::makeConnection() principal:"
78  							+ principal.asString());
79  			} else if (principal.isAnonymous()) {
80  
81  				System.out
82  						.println(getClass().getName()
83  								+ "::makeConnection() NULLANONYMOUS PRINCIPAL getting from Launcher");
84  				if (Launcher.model != null)
85  					if (Launcher.model.getPrincipal() == null)
86  						System.out
87  								.println(getClass().getName()
88  										+ "::makeConnection() Launcher.model.principal == null");
89  					else
90  						principal = Launcher.model.getPrincipal();
91  				if (principal != null)
92  					System.out.println(getClass().getName()
93  							+ "::makeConnection() principal:"
94  							+ principal.asString());
95  			}
96  
97  			if (isController)
98  				urlConnection = new URL(controllerURL + "?cmd=gapp")
99  						.openConnection();
100 			else
101 				urlConnection = new URL(dispatcherURL + "?cmd=gapp")
102 						.openConnection();
103 
104 			urlConnection.setDoInput(true);
105 			urlConnection.setDoOutput(true);
106 			urlConnection.setUseCaches(false);
107 			urlConnection.setRequestProperty("Content-Type",
108 					"application/x-www-form-urlencoded");
109 
110 			((ServletProtocolStream) stream).out = new PrintWriter(
111 					urlConnection.getOutputStream());
112 		} catch (Exception e) {
113 			System.err.println("Servlet output stream error");
114 			e.printStackTrace();
115 			((ServletProtocolStream) stream).out = null;
116 		}
117 
118 		if (previousUrlStr != null) {
119 			// Util.debug(this, "makeConnection:previousUrlStr=" +
120 			// previousUrlStr);
121 			controllerURL = previousUrlStr;
122 			previousUrlStr = null;
123 		}
124 	}

126 	public void sendCmd(int cmd, String inline) {
127 		checkConnection();
128 		// handling serialized objects
129 		if (!isController) {
130 			executeCmd(cmd, inline);
131 			return;
132 		}
133 
134 		// handling char streams
135 		try {
136 			stream.writeInt(cmd);
137 			if (inline != null)
138 				stream.writeEscapedLine(inline);
139 			if (principal != null)
140 				stream.writeEscapedLine(principal.asString());
141 			stream.flush();
142 		} catch (java.io.IOException e) {
143 			result.removeAllElements();
144 			e.printStackTrace();
145 			return;
146 		}
147 		readData();
148 	}

150 	public void sendCmd(int cmd, int subCmd, String[] data) throws IOException {
151 		checkConnection();
152 		// handling serialized objects
153 		if (!isController) {
154 			Object[] args = new Object[data.length + 1];
155 			args[0] = new Integer(subCmd);
156 			for (int i = 0; i < data.length; i++)
157 				args[i + 1] = data[i];
158 			executeCmd(cmd, args);
159 			return;
160 		}
161 		try {
162 			stream.writeInt(cmd);
163 			stream.writeInt(subCmd);
164 			if (data != null) {
165 				stream.writeInt((principal != null) ? data.length + 1
166 						: data.length);
167 
168 				for (int i = 0; i < data.length; i++) {
169 					stream.writeEscapedLine(data[i]);
170 				}
171 				if (principal != null)
172 					stream.writeLine(principal.asString());
173 			} else {
174 				if (principal != null) {
175 					stream.writeInt(1);
176 					stream.writeLine(Launcher.model.getPrincipal().asString());
177 				} else
178 					stream.writeInt(0);
179 			}
180 			stream.flush();
181 		} catch (java.io.IOException e) {
182 			e.printStackTrace();
183 			throw e;
184 		}
185 		readData();
186 	}

189 		try {
190 			((ServletProtocolStream) stream).in = new BufferedReader(
191 					new InputStreamReader(urlConnection.getInputStream()));
192 			super.readData();

194 		} catch (Exception e) {
195 			System.err.println("Servlet input stream error");
196 			e.printStackTrace();
197 			((ServletProtocolStream) stream).in = null;

271 	public Object executeCmd(int command, Serializable[] data) {
272 		try {
273 			URLConnection con;
274 			Serializable[] newData = new Serializable[(principal != null) ? data.length + 1
275 					: data.length];
276 
277 			for (int i = 0; i < data.length; i++)
278 				newData[i] = data[i];
279 			if (principal != null)
280 				newData[data.length] = principal;
281 
282 			if (command == DO_JOB || command == DO_TASK
283 					|| command == EXEC_MANDATE || !isController) {
284 				// Util.debug(this, "dispatcherURL=" + dispatcherURL);
285 				con = new URL(dispatcherURL + "?cmd=" + command)
286 						.openConnection();
287 			} else {
288 				// Util.debug(this, "controllerURL=" + controllerURL);
289 				con = new URL(controllerURL + "?cmd=" + command)
290 						.openConnection();
291 			}
292 			con.setDoInput(true);
293 			con.setDoOutput(true);
294 			con.setUseCaches(false);
295 			con.setRequestProperty("Content-Type",
296 					"application/x-www-form-urlencoded");
297 
298 			// Write the arguments as post data
299 			ObjectOutputStream out = new ObjectOutputStream(con
300 					.getOutputStream());
301 			out.writeObject(newData);
302 			out.flush();
303 			out.close();
304 
305 			ObjectInputStream ois = new ObjectInputStream(con.getInputStream());
306 			outcome = ois.readObject();
307 			return outcome;
308 		} catch (Exception e) {
309 			System.err.println("Servlet stream communication error");
310 			e.printStackTrace();
311 			return e;
312 		}
313 	}
