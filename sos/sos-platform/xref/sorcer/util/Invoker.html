20  /**
21   * Invoker interface is used by CmdManager in the Command pattern. Each GApp
22   * application should implement a factory <appName>.<type>.<appName>CmdProvider
23   * class providing its commands that implement gapp.lnch.Command interface.
24   * Invokers can dispatch commands using a CmdManager or execute default commands
25   * themselves via this interface.
26   */
27  
28  public interface Invoker {
29  
30  	// should be implemented is classes when CmdFactory does not have
31  	// a selected command to be executed
32  	public boolean executeSelect(String action);
33  
34  	// should be implemented is classes when CmdFactory does not have
35  	// an action command to be executed
36  	public boolean executeAction(String cmd);
37  
38  	// should be implemented is classes when CmdFactory does not have
39  	// a command of mandate to be executed
40  	public boolean executeMandate(Mandate mandate);
41  }
