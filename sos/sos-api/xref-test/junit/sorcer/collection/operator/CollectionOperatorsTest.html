74  		Map<Object, Object> map1 = dictionary(entry("name", "Mike"), entry("height", 174.0));

77  		
78  		// keys and values of entries
79  		String k = key(entry("name", "Mike"));
80  		Double v = value(entry("height", 174.0));
81  		assertEquals("name", k);
82  		assertEquals((Object) 174.0, v);
83  		
84  		// casts are needed for dictionary: Map<Object, Object>
85  		k = (String)map1.get("name");
86  		v = (Double)map1.get("height");
87  		assertEquals(k, "Mike");
88  		assertEquals((Object) 174.0, v);
89  		
90  		// casts are NOT needed for map: Map<K, V>
91  		v = map2.get("length");
92  		assertTrue(v.equals(248.0));
93  		
94  		// check map keys
95  		assertEquals(bag("name", "height"), map1.keySet());
96  
97          // check map values
98          Collection<? extends Serializable> c1 = Arrays.asList(174.0, "Mike");
99          Collection<Object> c2 = map1.values();
100         assertTrue(c1.containsAll(c2) && c2.containsAll(c1));
