116 	public static String getRepoDir() {
117 		String repo = props.getProperty(S_SORCER_REPO);
118         if (repo!=null) repo = repo.trim();
119 		
120 		if (repo != null && !repo.isEmpty()) {
121 			try {
122 				File repoDir = new File(repo);
123 				if (repoDir.exists() && repoDir.isDirectory())
124 					return repo;
125 			} catch (Throwable t) {
126 				logger.throwing(
127 						SorcerEnv.class.getName(),
128 						"The given Sorcer Jar Repo Location: " + repo + " does not exist or is not a directory!",
129 						t);
130 			}			
131 		} else {
132 			// Fall back to default location in user's home/.m2
133 			try {
134 				File repoDir = new File(System.getProperty("user.home")+"/.m2/repository");
135                 if (repoDir.exists() && repoDir.isDirectory())
136 					return repoDir.getAbsolutePath();
137                 else {
138                     FileUtils.forceMkdir(repoDir);
139                     return repoDir.getAbsolutePath();
140                 }
141             } catch (Throwable t) {
142 				logger.throwing(
143 						SorcerEnv.class.getName(),
144 						"The given Sorcer Jar Repo Location: " + repo + " does not exist or is not a directory!",
145 						t);
146 			}						
147 		}
148 		return null;
149 	}

194 	 * Tries to load properties from a <code>filename</code>, first in a local
195 	 * directory. If there is no file in the local directory then load the file
196 	 * from the classpath at sorcer/util/sorcer.env.
197 	 * 
198 	 * @throws IOException
199 	 * @throws ConfigurationException
200 	 */
201 	public static Properties loadProperties(String filename)
202 			throws ConfigurationException {
203 		Properties properties = new Properties();
204 		try {
205 			// Try in user home directory first
206 			properties.load((new FileInputStream(new File(filename))));
207 			logger.fine("loaded properties from: " + filename);
208 
209 		} catch (Exception e) {
210 			try {
211 				// try to look for sorcer.env in SORCER_HOME/configs
212 				properties.load((new FileInputStream(new File(System.getenv("SORCER_HOME")+"/configs/" + filename))));
213 				logger.fine("loaded properties from: " + System.getenv("SORCER_HOME") +"/configs/" + filename);
214 			} catch (Exception ee) {
215 				try {
216 					// No file give, try as resource sorcer/util/sorcer.env
217 					InputStream stream = SorcerEnv.class.getResourceAsStream(filename);
218 					if (stream != null)
219 						properties.load(stream);
220 					else
221 						logger.severe("could not load properties as Sorcer resource file>"
222 								+ filename + "<" );
223 				} catch (Throwable t2) {
224 					throw new ConfigurationException(e);
225 				}
226 			}
227 		}
228 		reconcileProperties(properties);
229 
230 		return properties;
231 	}

305 	 * Overwrites defined properties in sorcer.env (sorcer.home,
306 	 * provider.webster.interface, provider.webster.port) with those defined as
307 	 * JVM system properties.
308 	 * 
309 	 * @param properties
310 	 * @throws ConfigurationException
311 	 */
312 	private static void update(Properties properties)
313 			throws ConfigurationException {
314 		Enumeration<?> e = properties.propertyNames();
315 		String key, value, evalue = null;
316 		String pattern = "${" + "localhost" + "}";
317 //		String userDirPattern = "${user.home}";
318 		// first substitute for this localhost
319 		while (e.hasMoreElements()) {
320 			key = (String) e.nextElement();
321 			value = properties.getProperty(key);
322 			if (value.equals(pattern)) {
323 				try {
324 					value = getHostAddress();
325 				} catch (UnknownHostException ex) {
326 					ex.printStackTrace();
327 				}
328 				properties.put(key, value);
329 			}
330 	/*		if (value.equals(userDirPattern)) {
331 				value = System.getProperty("user.home");				
332 				properties.put(key, value);
333 			}*/
334 		}
335 		// now substitute other entries accordingly 
336 		e = properties.propertyNames();
337 		while (e.hasMoreElements()) {
338 			key = (String) e.nextElement();
339 			value = properties.getProperty(key);
340 			evalue = expandStringProperties(value, true);
341 			// try SORCER env properties
342 			if (evalue == null)
343 				evalue = expandStringProperties(value, false);
344 			if (evalue != null)
345 				properties.put(key, evalue);
346 			if (value.equals(pattern)) {
347 				try {
348 					evalue = getHostAddress();
349 				} catch (UnknownHostException e1) {
350 					e1.printStackTrace();
351 				}
352 				properties.put(key, evalue);
353 			}
354 		}
355 	}

368 	 * Expands properties embedded in a string with ${some.property.name}. Also
369 	 * treats ${/} as ${file.separator}.
370 	 */
371 	public static String expandStringProperties(String value,
372 			boolean isSystemProperty) throws ConfigurationException {
373 		int p = value.indexOf("${", 0);
374 		if (p == -1) {
375 			return value;
376 		}
377 		int max = value.length();
378 		StringBuffer sb = new StringBuffer(max);
379 		int i = 0; /* Index of last character we copied */
380 		while (true) {
381 			if (p > i) {
382 				/* Copy in anything before the special stuff */
383 				sb.append(value.substring(i, p));
384 				i = p;
385 			}
386 			int pe = value.indexOf('}', p + 2);
387 			if (pe == -1) {
388 				/* No matching '}' found, just add in as normal text */
389 				sb.append(value.substring(p, max));
390 				break;
391 			}
392 			String prop = value.substring(p + 2, pe);
393 			if (prop.equals("/")) {
394 				sb.append(File.separatorChar);
395 			} else {
396 				try {
397 					String val = null;
398 					if (isSystemProperty) {
399 						val = prop.length() == 0 ? null : System
400 								.getProperty(prop);
401 						if (val == null) {
402 							// try System env
403 							val = prop.length() == 0 ? null : System
404 									.getenv(prop);
405 						}
406 					} else {
407 						val = prop.length() == 0 ? null : props
408 								.getProperty(prop);
409 					}
410 					if (val != null) {
411 						sb.append(val);
412 					} else {
413 						return null;
414 					}
415 				} catch (SecurityException e) {
416 					throw new ConfigurationException(e);
417 				}
418 			}
419 			i = pe + 1;
420 			p = value.indexOf("${", i);
421 			if (p == -1) {
422 				/* No more to expand -- copy in any extra. */
423 				if (i < max) {
424 					sb.append(value.substring(i, max));
425 				}
426 				break;
427 			}
428 		}
429 		return sb.toString();
430 	}
