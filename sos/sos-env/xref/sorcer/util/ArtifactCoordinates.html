23  public class ArtifactCoordinates {
24  	private static final String DEFAULT_PACKAGING = "jar";
25  	private static final char MVN_SEP = ':';
26  	private String groupId;
27  	private String artifactId;
28  	private String version;
29  	private String classifier;
30  	private String packaging;
31  
32  	/**
33  	 * @param coords artifact coordinates in the form of
34  	 *               groupId:artifactId[[:packaging[:classifier]]:version]
35  	 * @throws IllegalArgumentException
36  	 */
37  	public static ArtifactCoordinates coords(String coords) {
38  		String[] coordSplit = coords.split(":");
39  		int length = coordSplit.length;
40  		if (length < 2 || length > 5) {
41  			throw new IllegalArgumentException(
42  					"Artifact coordinates must be in a form of groupId:artifactId[[:packaging[:classifier]]:version]");
43  		}
44  		String groupId = coordSplit[0];
45  		String artifactId = coordSplit[1];
46  		String packaging = DEFAULT_PACKAGING;
47  		String classifier = null;
48  		//if version is not specified it will be resolved by the Resolver#resolveVersion
49  		String version = null;
50  
51  		if (length == 3) {
52  			version = coordSplit[2];
53  		} else if (length == 4) {
54  			packaging = coordSplit[2];
55  			version = coordSplit[3];
56  		} else if (length == 5) {
57  			packaging = coordSplit[2];
58  			classifier = coordSplit[3];
59  			version = coordSplit[4];
60  		}
61  
62  		return new ArtifactCoordinates(groupId, artifactId, packaging, version, classifier);
63  	}
64  
65  	public ArtifactCoordinates(String groupId, String artifactId, String packaging, String version, String classifier) {
66  		this.groupId = groupId;
67  		this.artifactId = artifactId;
68  		this.packaging = packaging;
69  		this.version = version;
70  		this.classifier = classifier;
71  	}
72  
73  	public static ArtifactCoordinates coords(String groupId, String artifactId, String version) {
74  		return new ArtifactCoordinates(groupId, artifactId, DEFAULT_PACKAGING, version, null);
75  	}
76  
77  	public ArtifactCoordinates(String groupId, String artifactId, String version) {
78  		this(groupId, artifactId, DEFAULT_PACKAGING, version, null);
79  	}
80  
81  	@Override
82  	public String toString() {
83  		StringBuilder result = new StringBuilder(groupId).append(MVN_SEP).append(artifactId).append(MVN_SEP).append(packaging);
84  		if (classifier != null) {
85  			result.append(MVN_SEP).append(classifier);
86  		}
87  		if (version != null) {
88  			result.append(MVN_SEP).append(version);
89  		}
90  		return result.toString();
91  	}
