179 	 * 
180 	 * @param fileObj
181 	 *            File or directory
182 	 * @param parentPath
183 	 *            Relative parent path of fileObj to be stored in the archive
184 	 * @param jarOut
185 	 *            Output stream connected to an open archive
186 	 */
187 	public static void addToArchive(File fileObj, String parentPath,
188 			JarOutputStream jarOut) {
189 		FileInputStream in;
190 		byte[] buffer = new byte[1024];
191 		if (fileObj.isFile()) {
192 			// dirObj is a file
193 			try {
194 				String fileName = null;
195 				if (parentPath == null) {
196 					fileName = fileObj.getName();
197 				} else {
198 					fileName = parentPath + fileObj.getName();
199 				}
200 				in = new FileInputStream(fileObj);
201 				System.out.println("Adding: " + fileName);
202 				jarOut.putNextEntry(new JarEntry(fileName));
203 				int len = in.read(buffer);
204 				while (len > 0) {
205 					jarOut.write(buffer, 0, len);
206 					len = in.read(buffer);
207 				}
208 				jarOut.closeEntry();
209 				in.close();
210 			} catch (IOException e) {
211 				e.printStackTrace();
212 			}
213 		} else if (fileObj.isDirectory()) {
214 			// dirObj is a directory
215 			File[] files = fileObj.listFiles();
216 
217 			for (int i = 0; i < files.length; i++) {
218 				if (files[i].isDirectory()) {
219 					if (parentPath == null) {
220 						parentPath = fileObj.getName() + File.separator
221 								+ files[i].getName();
222 					} else {
223 						parentPath = parentPath + File.separator
224 								+ fileObj.getName() + File.separator
225 								+ files[i].getName();
226 					}
227 					addToArchive(files[i], parentPath, jarOut);
228 					continue;
229 				}
230 
231 				try {
232 					in = new FileInputStream(files[i]);
233 					String fileName = null;
234 					if (parentPath == null) {
235 						fileName = fileObj.getName() + File.separator
236 								+ files[i].getName();
237 					} else {
238 						fileName = parentPath + File.separator
239 								+ fileObj.getName() + File.separator
240 								+ files[i].getName();
241 					}
242 					System.out.println("Adding: " + fileName);
243 					jarOut.putNextEntry(new JarEntry(fileName));
244 					int len = in.read(buffer);
245 					while (len > 0) {
246 						jarOut.write(buffer, 0, len);
247 						len = in.read(buffer);
248 					}
249 					jarOut.closeEntry();
250 					in.close();
251 				} catch (FileNotFoundException e) {
252 					e.printStackTrace();
253 				} catch (IOException e) {
254 					e.printStackTrace();
255 				}
256 			}
257 		}

401 			// Check if the destination file exist

408 			// Making a second attempt to copy the source file
