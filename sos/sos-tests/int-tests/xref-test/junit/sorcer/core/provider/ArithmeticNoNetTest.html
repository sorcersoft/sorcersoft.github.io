150 	@SuppressWarnings("unchecked")
151 	private Job createSrv() throws Exception {
152 		Task t3 = srv("t3", sig("subtract", SubtractorImpl.class), 
153 				cxt("subtract", in("arg/x1"), in("arg/x2"),
154 						out("result/y")));
155 
156 		Task t4 = srv("t4", sig("multiply", MultiplierImpl.class), 
157 				//cxt("multiply", in("super/arg/x1"), in("arg/x2", 50.0),
158 				cxt("multiply", in("arg/x1", 10.0), in("arg/x2", 50.0),
159 						out("result/y")));
160 
161 		Task t5 = srv("t5", sig("add", AdderImpl.class), 
162 				cxt("add", in("arg/x1", 20.0), in("arg/x2", 80.0),
163 						out("result/y")));
164 
165 		// Service Composition j1(j2(t4(x1, x2), t5(x1, x2)), t3(x1, x2))
166 		//Job j1= job("j1", job("j2", t4, t5, strategy(Flow.PARALLEL, Access.PULL)), t3,
167 		Job job = srv("j1", sig("execute", ServiceJobber.class),
168 					cxt(in("arg/x1", 10.0), result("job/result", from("j1/t3/result/y"))), 
169 				srv("j2", sig("execute", ServiceJobber.class), t4, t5),
170 				t3,
171 				pipe(out(t4, "result/y"), in(t3, "arg/x1")),
172 				pipe(out(t5, "result/y"), in(t3, "arg/x2")));
173 				
174 		return job;
175 	}

252 	private Job createJob(Flow flow) throws Exception {
253 		Task t3 = task("t3", sig("subtract", Subtractor.class), 
254 				context("subtract", in("arg/x1", null), in("arg/x2", null),
255 						out("result/y", null)));
256 
257 		Task t4 = task("t4", sig("multiply", Multiplier.class), 
258 				context("multiply", in("arg/x1", 10.0), in("arg/x2", 50.0),
259 						out("result/y", null)));
260 
261 		Task t5 = task("t5", sig("add", Adder.class), 
262 				context("add", in("arg/x1", 20.0), in("arg/x2", 80.0),
263 						out("result/y", null)));
264 
265 		// Service Composition j1(j2(t4(x1, x2), t5(x1, x2)), t3(x1, x2))
266 		//Job job = job("j1",
267 		Job job = job("j1", //sig("service", RemoteJobber.class), 
268 				//job("j2", t4, t5),
269 				job("j2", t4, t5, strategy(flow, Access.PULL)), 
270 				t3,
271 				pipe(out(t4, "result/y"), in(t3, "arg/x1")),
272 				pipe(out(t5, "result/y"), in(t3, "arg/x2")));
