56      /**
57       * There is a java API for starting SORCER in in-process or forked modes with WAIT options, so there shouldn't be any need to call this method from java code.
58       */
59      public static void main(String[] args) {
60          new Sorcer().run(args);

63      private void run(String[] args) {
64          try {
65              JavaSystemProperties.ensure(SORCER_HOME);
66              Options options = buildOptions();
67              CommandLineParser parser = new PosixParser();
68              CommandLine cmd = parser.parse(options, args);
69  
70              if (cmd.hasOption('h')) {
71                  HelpFormatter helpFormatter = new HelpFormatter();
72                  helpFormatter.printHelp(160,
73                          "sorcer-boot [options] [service list files]\n"
74                                  + "Service list file may be:\n" +
75                                  "- a .config file with list of service descriptors,\n"
76                                  + "- an .opstring or .groovy Operational String file,\n"
77                                  + "- an .oar or .jar file compliant with OAR specification,\n"
78                                  + "- an :artifactId of a module that output is compliant with OAR specification (artifact*jar file is searched under $SORCER_HOME)",
79                          "Start sorcer", options, null
80                  );
81                  return;
82              }
83  
84              ILauncher launcher = parseCommandLine(cmd);
85              WaitingListener listener = new WaitingListener();
86              launcher.addSorcerListener(listener);
87              launcher.preConfigure();
88  
89              launcher.start();
90              listener.wait(waitMode);
91              System.exit(0);
92          } catch (Exception x) {
93              x.printStackTrace();
94              System.exit(-1);
95          }
96      }

149     private ILauncher parseCommandLine(CommandLine cmd) throws ParseException, IOException {
150         Mode mode = null;
151         if (cmd.hasOption(MODE)) {
152             String modeValue = cmd.getOptionValue(MODE);
153             for (Mode m : Mode.values()) {
154                 if (m.paramValue.equalsIgnoreCase(modeValue))
155                     mode = m;
156             }
157             if (mode == null)
158                 throw new IllegalAccessError("Illegal mode " + modeValue);
159         } else
160             mode = Mode.preferDirect;
161 
162         Integer debugPort = null;
163         if (cmd.hasOption(DEBUG)) {
164             debugPort = Integer.parseInt(cmd.getOptionValue(DEBUG));
165         }
166 
167         try {
168             waitMode = cmd.hasOption(WAIT) ? WaitMode.valueOf(cmd.getOptionValue(WAIT)) : WaitMode.end;
169         } catch (IllegalArgumentException x) {
170             throw new IllegalArgumentException("Illegal wait option " + cmd.getOptionValue(WAIT) + ". Use one of " + Arrays.toString(WaitMode.values()), x);
171         }
172 
173         ILauncher launcher = null;
174         if (!mode.fork) {
175             boolean envOk = SorcerLauncher.checkEnvironment();
176             if (envOk && debugPort == null) {
177                 if (waitMode != WaitMode.end)
178                     System.err.println("WARN Starting SORCER with " + waitMode + " mode will result with early exit.");
179                 launcher = createSorcerLauncher();
180             } else {
181                 if (debugPort != null)
182                     report(mode, "Cannot run in {} mode with debug", mode.paramValue);
183                 else
184                     report(mode, "Cannot run in {} mode; see above", mode.paramValue);
185             }
186         }
187 
188         if (launcher == null) {
189             IForkingLauncher forkingLauncher = null;
190             try {
191                 forkingLauncher = createForkingLauncher(debugPort, waitMode);
192             } catch (IllegalStateException e) {
193                 if ((mode.fork && mode.force) || (!mode.fork))
194                     throw e;
195             }
196             launcher = forkingLauncher;
197         }
198 
199         // called prefer-fork but didn't make it
200         // fallback to direct
201         if (launcher == null && mode.fork && !mode.force)
202             launcher = createSorcerLauncher();
203 
204         if (launcher == null)
205             throw new IllegalStateException("Could not start SORCER");
206 
207         String homePath = System.getProperty(SORCER_HOME);
208         File home = null;
209         if (homePath != null) {
210             home = new File(homePath).getCanonicalFile();
211             launcher.setHome(home);
212         }
213 
214         File logDir = FileUtils.getFile(home, cmd.hasOption(LOGS) ? cmd.getOptionValue(LOGS) : "logs");
215 
216         launcher.setLogDir(logDir);
217 
218         @SuppressWarnings("unchecked")
219         List<String> userConfigFiles = cmd.getArgList();
220         launcher.setConfigs(userConfigFiles);
221 
222         if (cmd.hasOption(PROFILE))
223             launcher.setProfile(cmd.getOptionValue(PROFILE));
224 
225         if (cmd.hasOption(RIO)) {
226             String[] rioConfigs = StringUtils.tokenizerSplit(cmd.getOptionValue(RIO), File.pathSeparator);
227             launcher.setRioConfigs(new ArrayList<String>(Arrays.asList(rioConfigs)));
228         }
229 
230         return launcher;
231     }

233     private static void report(Mode mode, String message, Object... args) {
234         if (mode.force)
235             throw new IllegalArgumentException(MessageFormatter.arrayFormat(message, args).getMessage());
236         else
237             LoggerFactory.getLogger(Sorcer.class).warn(message, args);
238     }

240     private static IForkingLauncher createForkingLauncher(Integer debugPort, WaitMode mode) {
241         IForkingLauncher forkingLauncher;
242         try {
243             forkingLauncher = (IForkingLauncher) Class.forName("sorcer.launcher.process.ForkingLauncher").newInstance();
244             if (debugPort != null)
245                 forkingLauncher.setDebugPort(debugPort);
246 
247             forkingLauncher.addSorcerListener(new DestroyingListener(ProcessDestroyer.installShutdownHook(), mode == WaitMode.start));
248         } catch (InstantiationException e) {
249             throw new IllegalStateException("Could not instantiate ForkingLauncher", e);
250         } catch (IllegalAccessException e) {
251             throw new IllegalStateException("Could not access ForkingLauncher", e);
252         } catch (ClassNotFoundException e) {
253             throw new IllegalStateException("Could not instantiate ForkingLauncher", e);
254         }
255         forkingLauncher.setOut(System.out);
256         forkingLauncher.setErr(System.err);
257         return forkingLauncher;
258     }

260     private static ILauncher createSorcerLauncher() {
261         SorcerLauncher.installLogging();
262         SorcerLauncher launcher = new SorcerLauncher();
263         launcher.addSorcerListener(new ExitingCallback());
264 
265         SorcerLauncher.installSecurityManager();
266         return launcher;
267     }
