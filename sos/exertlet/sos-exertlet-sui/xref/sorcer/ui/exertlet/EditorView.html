341 	class EditActionListener implements ActionListener {
342 		public void actionPerformed(ActionEvent event) {
343 			String url;
344 			if (event.getSource() == urlField) {
345 				url = urlField.getText();
346                 try {
347                     displayUrl(new URL(url));
348                 } catch (MalformedURLException me) {
349                     warnUser("can't load: " + url);
350                 }
351 				return;
352 			}
353 
354 			String command = event.getActionCommand();
355 			if (OPEN_LABEL.equals(command)) {
356 				openFile();
357 				return;
358 			} else if (EDIT_LABEL.equals(command)) {
359 				openEditor(urlField.getText());
360 				return;
361 			} else if (EXERT_LABEL.equals(command)) {
362 				String script = editPane.getText();
363 				if (model != null) {
364 					runTaskScript(script);
365 				} else {
366 					StringBuilder sb = new StringBuilder(
367 							staticImports.toString());
368 					Scanner scanner = new Scanner(script);
369 					while (scanner.hasNextLine()) {
370 						String line = scanner.nextLine().trim();
371 						if (line.length() > 0 && line.charAt(0) != '#') {
372 							sb.append(line);
373 						}
374 						sb.append("\n");
375 					}
376 					logger.finer(">>> executing script: " + sb.toString());
377                     try {
378                         scriptExerter.readScriptWithHeaders(sb.toString());
379                         scriptExerter.parse();
380                         Object result = scriptExerter.execute();
381                         if (result instanceof Exertion)
382                             processExerion((Exertion) result);
383                         else if (result != null) {
384                             openOutPanel(result.toString());
385                         }
386                     } catch (IOException io) {
387                         logger.severe("Caught exception while executing script: " + io.getMessage());
388                         openOutPanel(StringUtils.stackTraceToString(io));
389                     } catch (Throwable th) {
390                         openOutPanel(StringUtils.stackTraceToString(th));
391                         logger.severe("Caught exception while executing script: " + th.getMessage());
392                     }
393 				}
394 				return;
395 			} else if (SAVE_LABEL.equals(command)) {
396 				saveFile();
397 				return;
398 			} else if (SAVE_AS_LABEL.equals(command)) {
399 				saveAsFile();
400 				return;
401 			} else if (EXIT_LABEL.equals(command)) {
402 				// tabbedPane.remove(tabbedPane.getTabCount()-1);
403 				tabbedPane.remove(EditorView.this);
404 				return;
405 			} else if (GET_CONTEXT_LABEL.equals(command)) {
406 				try {
407 					getContextFromProvider();
408 				} catch (RemoteException e) {
409 					openEditor(StringUtils.stackTraceToString(e));
410 					e.printStackTrace();
411 				}
412 				return;
413 			}
414 			// Clicked "home" button instead of entering URL
415 			// url = initialURL;
416 			url = "http://sorcersoft.org";
417             try {
418 			    displayUrl(new URL(url));
419             } catch (MalformedURLException me) {
420                 warnUser("can't load: " + url);
421             }
422 		}
423 	}

466 	private ExertionThread runTaskScript(String script) {
467 		String serviceType = model.getSelectedInterfaceName();
468 		String selector = model.getSelectedMethod();
469 		StringBuilder sb = new StringBuilder();
470 		sb.append(staticImports.toString()).append("\nimport ")
471 				.append(serviceType).append(";\n").append("task(sig(\"")
472 				.append(selector).append("\",").append(serviceType)
473 				.append(".class),\n").append(script).append(");");
474 
475 		// logger.info(">>> executing task script: " + sb.toString());
476 		ExertionThread eThread = new ExertionThread(sb.toString());
477 		eThread.start();
478 		return eThread;
479 	}

481 	private class ExertionThread extends Thread {
482 		String script;
483 		Object result;
484 
485 		public ExertionThread(String script) {
486 			this.script = script;
487 		}
488 
489 		public void run() {
490 			try {
491 				shell = new GroovyShell();
492 				Object result = shell.evaluate(script);
493 				if (result instanceof Exertion)
494 					processExerion((Exertion) result);
495 				else if (result != null) {
496 					openOutPanel(result.toString());
497 				}
498 			} catch (Exception e) {
499 				openOutPanel(StringUtils.stackTraceToString(e));
500 			}
501 		}
502 
503 		public Object getResult() {
504 			return result;
505 		}
506 	}

508 	private void processExerion(Exertion exertion) throws ContextException {
509 		String codebase = System.getProperty("java.rmi.server.codebase");
510 		logger.finer("Using exertlet codebase: " + codebase);
511 		
512 		if (((ServiceExertion) exertion).getStatus() == Exec.DONE) {
513 			// logger.finer(">>> done by Groovy Shell");
514 			showResults(exertion);
515 			return;
516 		}
517 		Exertion out = null;
518 		boolean done = false;
519 		try {
520 			Class<?>[] interfaces = provider.getClass().getInterfaces();
521 			for (int i = 0; i < interfaces.length; i++) {
522 				if (interfaces[i] == exertion.getProcessSignature()
523 						.getServiceType()) {
524 					out = provider.service(exertion, null);
525 					done = true;
526 					break;
527 				}
528 			}
529 			if (!done) {
530 				logger.finer(">> executing by exert: " + exertion.getName());
531 				// inspect class loader tree
532 //				com.sun.jini.start.ClassLoaderUtil
533 //						.displayContextClassLoaderTree();
534 				// com.sun.jini.start.ClassLoaderUtil.displayClassLoaderTree(exertion
535 				// .getClass().getClassLoader());
536 
537 				out = exertion.exert();
538 			}
539 		} catch (RemoteException e) {
540 			openOutPanel(StringUtils.stackTraceToString(e));
541 			e.printStackTrace();
542 			return;
543 		} catch (TransactionException e) {
544 			openOutPanel(StringUtils.stackTraceToString(e));
545 			e.printStackTrace();
546 			return;
547 		} catch (ExertionException e) {
548 			openOutPanel(StringUtils.stackTraceToString(e));
549 			e.printStackTrace();
550 			return;
551 		}
552 		showResults(out);
553 	}

555 	private void showResults(Exertion exertion)  throws ContextException {
556 		if (exertion == null) {
557 			openOutPanel("Failed to process the exertlet!");
558 			return;
559 		}
560 		if (exertion.getExceptions().size() > 0) {
561 			openOutPanel(exertion.getExceptions().toString());
562 		}
563 		else {
564 			StringBuilder sb = new StringBuilder(exertion.getContext().toString());
565 			if (debug) {
566 				sb.append("\n");
567 				sb.append(((ServiceExertion)exertion).getControlInfo().toString());
568 			}
569 			openOutPanel(sb.toString());
570 		}
571 	}

573 	private void saveFile() {
574 		BufferedWriter br = null;
575 		String fn = source;
576 		if (source != null && source.length() != 0) {
577 			int index = source.indexOf("file://");
578 			if (index == 0)
579 				fn = source.substring("file://".length());
580 		}
581 		// logger.info("Saving edited file: " + fn);
582 		File file = new File(fn);
583 		try {
584 			String content = editPane.getText();
585 			br = new BufferedWriter(new FileWriter(file));
586 			br.write(content);
587 		} catch (IOException e) {
588 			e.printStackTrace();
589 		} finally {
590 			if (br != null)
591 				try {
592 					br.close();
593 				} catch (IOException e) {
594 					e.printStackTrace();
595 				}
596 		}
597 	}

623 	private void getContextFromProvider() throws RemoteException {
624 		String script = ((RemoteContextManagement) provider).getContextScript();
625 
626 		if (script == null || script.length() == 0) {
627 			script = "No dataContext script availble from the provider: \n"
628 					+ provider;
629 		}
630 		if (model != null)
631 			editor = new EditorView(script, false, true, true, true, false,
