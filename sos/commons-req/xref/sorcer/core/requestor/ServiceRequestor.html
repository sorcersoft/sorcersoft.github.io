179     /**
180      * Loads service requestor properties from a <code>filename</code> file. By
181      * default a service requestor loads its properties from
182      * <code>requestor.properties</code> file located in the requestor's
183      * package. Also, a service requestor properties file name can be specified
184      * as a system property when starting the requestor with
185      * <code>-DrequestorProperties=&ltfilename&gt<code>. In this case the requestor loads
186      * properties from <code>filename</code> file. Properties are accessible
187      * calling the <code>
188      * getProperty(String)</code> method.
189      *
190      * @param filename
191      *            the properties file name see #getProperty
192      */
193     public void loadProperties(String filename) {
194         logger.info("loading requestor properties:" + filename);
195         String propsFile = System.getProperty("requestor.properties.file");
196 
197         try {
198             if (propsFile != null) {
199                 props.load(new FileInputStream(propsFile));
200             } else {
201                 // check the class resource
202                 InputStream is = this.getClass().getResourceAsStream(filename);
203                 // check local resource
204                 if (is == null)
205                     is = (InputStream) (new FileInputStream(filename));
206                 if (is != null) {
207                     props = new Properties();
208                     props.load(is);
209                 } else {
210                     System.err
211                             .println("Not able to open stream on properties: "
212                                     + filename);
213                     System.err.println("Service runner class: "
214                             + this.getClass());
215                     return;
216                 }
217             }
218         } catch (IOException ioe) {
219             logger.info("Not able to load requestor properties");
220             // ioe.printStackTrace();
221         }
222 
223     }

233     private static String[] toArray(String arg) {
234         StringTokenizer token = new StringTokenizer(arg, " ,;");
235         String[] array = new String[token.countTokens()];
236         int i = 0;
237         while (token.hasMoreTokens()) {
238             array[i] = token.nextToken();
239             i++;
240         }
241         return (array);
242 	}

248     public static Webster prepareCodebase(String[] artifactCoords) {
249         if (artifactCoords==null)
250             return prepareCodebase((ArtifactCoordinates[])null);
251         ArtifactCoordinates[] acc = new ArtifactCoordinates[artifactCoords.length];
252         int i = 0;
253         for (String ac : artifactCoords) {
254             acc[i] = ArtifactCoordinates.coords(ac);
255             i++;
256         }
257         return prepareCodebase(acc);
258     }

260     public static Webster prepareCodebase(ArtifactCoordinates[] artifactCoords) {
261         // Initialize system properties: configs/sorcer.env
262         SorcerEnv.getEnvProperties();
263         String val = System.getProperty(SorcerConstants.SORCER_WEBSTER_INTERNAL);
264         if (val != null && val.length() != 0) {
265             isWebsterInt = val.equals("true");
266         }
267         String exertrun = System.getProperty(SorcerConstants.R_CODEBASE);
268         StringBuilder codebase = new StringBuilder();
269         if (exertrun!=null || artifactCoords!=null) {
270             if (exertrun!=null && !exertrun.isEmpty()) {
271                 String[] artifacts = exertrun.split(" ");
272                 for (String artifact : artifacts) {
273                     if (codebase.length() > 0)
274                         codebase.append(" ");
275                     codebase.append(resolve(coords(artifact)));
276                 }
277             }
278             if (artifactCoords!=null)
279                 for (ArtifactCoordinates artCord : artifactCoords) {
280                     codebase.append(' ').append(resolve(artCord));
281                 }
282 
283             // Add default codebase sos-platform and sorcer-api
284             codebase.append(' ').append(resolve(Artifact.getSorcerApi()));
285             codebase.append(' ').append(resolve(Artifact.getSosPlatform()));
286 
287             logger.fine("ServiceRequestor generated codebase: " + codebase.toString());
288             if (isWebsterInt)
289                 System.setProperty(SorcerConstants.CODEBASE_JARS, codebase.toString());
290             else
291                 System.setProperty(JavaSystemProperties.RMI_SERVER_CODEBASE, codebase.toString());
292 
293             codebaseJars = toArray(codebase.toString());
294         }
295 
296         if (isWebsterInt) {
297             String roots = System.getProperty(SorcerConstants.WEBSTER_ROOTS);
298             String[] tokens = null;
299             if (roots != null)
300                 tokens = toArray(roots);
301             try {
302                 return InternalWebster.startWebster(codebaseJars, tokens);
303             } catch (IOException e) {
304                 e.printStackTrace();
305                 throw new RuntimeException(e);
306             }
307         }
308         return null;
309     }
