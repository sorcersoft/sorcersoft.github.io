402 	private Exertion f5xP(String poolSizeStr, String size) throws Exception {
403 		int poolSize = new Integer(size);
404 		int tally = new Integer(size);
405 		Task task = null;
406 		ExertionCallable ec = null;
407 		long start = System.currentTimeMillis();
408 		List<Future<Exertion>> fList = new ArrayList<Future<Exertion>>(tally);
409 		ExecutorService pool = Executors.newFixedThreadPool(poolSize);
410 		for (int i = 0; i < tally; i++) {
411 			task = getTask();
412 			task.getControlContext().setAccessType(Access.PULL);
413 			task.setName("f5-" + i);
414 			ec = new ExertionCallable(task);
415 			logger.info("exertion submit: " + task.getName());
416 			Future<Exertion> future = pool.submit(ec);
417 			fList.add(future);
418 		}
419 		pool.shutdown();
420 		for (int i = 0; i < tally; i++) {
421 			logger.info("got future value for: " + fList.get(i).get().getName());
422 		}
423 		long end = System.currentTimeMillis();
424 		System.out.println("Execution time: " + (end - start) + " ms.");
425 		logger.info("got last dataContext #" + tally + "\n" + fList.get(tally-1).get().getDataContext());
426 		logger.info("run in parallel: " + tally);
427 		return fList.get(tally-1).get();
428 	}
