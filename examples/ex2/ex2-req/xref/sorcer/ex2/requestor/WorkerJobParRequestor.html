24  import sorcer.core.context.ServiceContext;

28  import sorcer.service.Context;

30  import sorcer.service.Job;

34  import sorcer.util.Log;
35  import sorcer.util.Sorcer;

37  public class WorkerJobParRequestor {
38  
39  	private static Logger logger = Log.getTestLog();
40  
41  	public static void main(String[] args) throws Exception {
42  		System.setSecurityManager(new RMISecurityManager());
43  		// initialize system properties
44  		Sorcer.getEnvProperties();
45  
46  		Exertion result = new WorkerJobParRequestor()
47  			.getExertion().exert();
48  		// get contexts of component exertions - in this case tasks
49  		logger.info("Output context1: \n" + result.getContext("work1"));
50  		logger.info("Output context2: \n" + result.getContext("work2"));
51  		logger.info("Output context3: \n" + result.getContext("work3"));
52  	}
53  
54  	private Exertion getExertion() throws Exception {
55  		String hostname = InetAddress.getLocalHost().getHostName();
56  
57  		Context context1 = new ServiceContext("work1");
58  		context1.putValue("requstor/name", hostname);
59  		context1.putValue("requestor/operand/1", 1);
60  		context1.putValue("requestor/operand/2", 1);
61  		
62  		Context context2 = new ServiceContext("work2");
63  		context2.putValue("requstor/name", hostname);
64  		context2.putValue("requestor/operand/1", 2);
65  		context2.putValue("requestor/operand/2", 2);
66  		
67  		Context context3 = new ServiceContext("work3");
68  		context3.putValue("requstor/name", hostname);
69  		context3.putValue("requestor/operand/1", 3);
70  		context3.putValue("requestor/operand/2", 3);
71  		
72  		NetSignature signature1 = new NetSignature("doWork",
73  				sorcer.ex2.provider.Worker.class);
74  		NetSignature signature2 = new NetSignature("doWork",
75  				sorcer.ex2.provider.Worker.class);
76  		NetSignature signature3 = new NetSignature("doWork",
77  				sorcer.ex2.provider.Worker.class);
78  		
79  		Task task1 = new NetTask("work1", signature1, context1);
80  		Task task2 = new NetTask("work2", signature2, context2);
81  		Task task3 = new NetTask("work3", signature3, context3);
82  		Job job = new NetJob();
83  		job.addExertion(task1);
84  		job.addExertion(task2);
85  		job.addExertion(task3);
86  		job.setFlow(Flow.PAR);
87  		job.setAccess(Access.PULL);
88  		return job;
89  	}
