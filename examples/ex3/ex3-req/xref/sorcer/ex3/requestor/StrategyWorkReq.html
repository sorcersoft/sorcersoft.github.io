18  package sorcer.ex3.requestor;
19  
20  import sorcer.core.context.ServiceContext;
21  import sorcer.core.exertion.NetJob;
22  import sorcer.core.exertion.NetTask;
23  import sorcer.core.requestor.ServiceRequestor;
24  import sorcer.core.signature.NetSignature;
25  import sorcer.service.*;
26  import sorcer.service.Strategy.Access;
27  import sorcer.service.Strategy.Flow;
28  import sorcer.util.Sorcer;
29  import sorcer.ex2.requestor.Works;
30  
31  public class StrategyWorkReq extends ServiceRequestor {
32  
33      public Exertion getExertion(String... args) throws ExertionException {
34          String requestorName = getProperty("requestor.name");
35          String pn1, pn2, pn3;
36          pn1 = Sorcer.getSuffixedName(getProperty("provider.name.1"));
37          pn2 = Sorcer.getSuffixedName(getProperty("provider.name.2"));
38          pn3 = Sorcer.getSuffixedName(getProperty("provider.name.3"));
39  
40          try {
41              Context context1 = new ServiceContext("work1");
42              context1.putValue("requestor/name", requestorName);
43              context1.putValue("requestor/operand/1", 1);
44              context1.putValue("requestor/operand/2", 1);
45              context1.putValue("requestor/work", Works.work1);
46              context1.putValue("to/provider/name", pn1);
47  
48              Context context2 = new ServiceContext("work2");
49              context2.putValue("requestor/name", requestorName);
50              context2.putValue("requestor/operand/1", 2);
51              context2.putValue("requestor/operand/2", 2);
52              context2.putValue("requestor/work", Works.work2);
53              context2.putValue("to/provider/name", pn2);
54  
55              Context context3 = new ServiceContext("work3");
56              context3.putValue("requestor/name", requestorName);
57              context3.putValue("requestor/operand/1", 3);
58              context3.putValue("requestor/operand/2", 3);
59              context3.putValue("requestor/work", Works.work3);
60              context3.putValue("to/provider/name", pn3);
61  
62              // define required signatures
63              NetSignature signature1 = new NetSignature("doWork",
64                      sorcer.ex2.provider.Worker.class, pn1);
65              NetSignature signature2 = new NetSignature("doWork",
66                      sorcer.ex2.provider.Worker.class, pn2);
67              NetSignature signature3 = new NetSignature("doWork",
68                      sorcer.ex2.provider.Worker.class, pn3);
69  
70              // define required services
71              Task task1 = new NetTask("work1", signature1, context1);
72              task1.setExecTimeRequested(true);
73              Task task2 = new NetTask("work2", signature2, context2);
74              Task task3 = new NetTask("work3", signature3, context3);
75  
76              // define a job
77              Job job = new NetJob("flow");
78              job.setExecTimeRequested(true);
79              job.addExertion(task1);
80              job.addExertion(task2);
81              job.addExertion(task3);
82  
83              // PUSH or PULL provider access
84              boolean isPushAccess = getProperty("provider.access.type", "PUSH").equals("PUSH");
85              if (isPushAccess)
86                  job.setAccessType(Access.PUSH);
87              else
88                  job.setAccessType(Access.PULL);
89  
90              // Exertion control flow PARALLEL or SEQUENTIAL
91              boolean iSequential = getProperty("provider.control.flow", "SEQUENTIAL").equals("SEQUENTIAL");
92              if (iSequential)
93                  job.setFlowType(Flow.SEQ);
94              else
95                  job.setFlowType(Flow.PAR);
96  
97              logger.info("isPushAccess: " + isPushAccess + " iSequential: " + iSequential);
98  
99              return job;
100         } catch (Exception e) {
101             throw new ExertionException("Failed to create exertion", e);
102         }
103     }
