41  	 * @see sorcer.ex1.provider.WhoIsIt#getHostName(sorcer.service.Context)
42  	 */
43  	@Override
44  	public Context getHostName(Context context) throws RemoteException,
45  			ContextException {
46  		String hostname;
47  		logger.info("Got dataContext to process: " + context);
48  		try {
49  			hostname = InetAddress.getLocalHost().getHostName();
50  			context.putValue("provider/hostname", hostname);
51  			String rhn = (String) context.getValue("requestor/hostname");
52  			Message msg = (Message) context.getValue("requestor/message");
53  			context.putValue("provider/message", new ProviderMessage(msg
54  					.getMessage(), getProviderName(), rhn));
55  			
56  			Thread.sleep(1000);
57  			context.reportException(new RuntimeException("getHostName: slept for 1 sec"));
58  			context.appendTrace(getClass().getName() + ":" + getProviderName());
59  			
60  			Logger cl = getContextLogger();
61  			if (cl != null)
62  				cl.info("executed getHostName: " + context);
63  			
64  			Logger pl = getProviderLogger();
65  			if (pl != null)
66  				pl.info("executed getHostName: " + context);
67  			
68  			Logger rl = getRemoteLogger();
69  			if (rl != null)
70  				rl.info(">>>>>>>>>>>> out dataContext: " + context);
71  		} catch (UnknownHostException e1) {
72  			context.reportException(e1);
73  			e1.printStackTrace();
74  		} catch (InterruptedException e2) {
75  			context.reportException(e2);
76  			e2.printStackTrace();
77  		}
78  		return context;
79  	}
