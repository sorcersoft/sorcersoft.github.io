310 	/**
311 	 * This recursive function is used to change the directions of all elements
312 	 * underneath the current node. The function works but can lead to doing
313 	 * more than the user was probably attempting to do.
314 	 * 
315 	 * @param current
316 	 *            DefaultMutableTreeNode the current node we are working on
317 	 * @param direction
318 	 *            String specifying the current direction to set the children
319 	 *            nodes to
320 	 */
321 	void SetChildDirection(DefaultMutableTreeNode current, String direction) {
322 		for (int i = 0; i < current.getChildCount(); i++) // do it for each
323 		// child
324 		{
325 			DefaultMutableTreeNode child = (DefaultMutableTreeNode) current
326 					.getChildAt(i);
327 			if (child.getUserObject() instanceof ContextNodeType) // make sure
328 			// its of
329 			// our
330 			// special
331 			// type
332 			{
333 				((ContextNodeType) child.getUserObject())
334 						.setDirection(direction);
335 				theTree.updateModel(child);
336 				if (child.getChildCount() != 0)
337 					SetChildDirection(child, direction);
338 			}
339 		}
